# 3516.找到最近的人

```java
class Solution {
    public int findClosest(int x, int y, int z) {
        int xLen = Math.abs(z - x);
        int yLen = Math.abs(z - y);
        return xLen == yLen ? 0 : xLen > yLen ? 2 : 1;
    }
}
```

# 3521.使数组和能被K整除的最少操作次数

```java
class Solution {
    public int minOperations(int[] nums, int k) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        return sum % k;
    }
}
```

# 3536.两个数字的最大乘积

```java
class Solution {
    public int maxProduct(int n) {
        char[] nChar = String.valueOf(n).toCharArray();
        Arrays.sort(nChar);
        int number1 = nChar[nChar.length - 2] - '0';
        int number2 = nChar[nChar.length - 1] - '0';
        return number1 * number2;
    }
}
```

# 3541.找到频率最高的元音和辅音

```java
class Solution {
    public int maxFreqSum(String s) {
        Map<Character, Integer> yuanMap = new HashMap<>();
        Map<Character, Integer> fuMap = new HashMap<>();
        Set<Character> yuanSet = new HashSet<>();
        yuanSet.add('a');
        yuanSet.add('e');
        yuanSet.add('i');
        yuanSet.add('o');
        yuanSet.add('u');
        for (int i = 0; i < s.length(); i++) {
            char temp = s.charAt(i);
            if (yuanSet.contains(temp)) {
                yuanMap.put(temp, yuanMap.getOrDefault(temp, 0) + 1);
            } else {
                fuMap.put(temp, fuMap.getOrDefault(temp, 0) + 1);
            }
        }
        int yuanMaxCnt = 0;
        int fuMaxCnt = 0;
        for (Character key : yuanMap.keySet()) {
            yuanMaxCnt = Math.max(yuanMaxCnt, yuanMap.get(key));
        }
        for (Character key : fuMap.keySet()) {
            fuMaxCnt = Math.max(fuMaxCnt, fuMap.get(key));
        }
        return yuanMaxCnt + fuMaxCnt;
    }
}
```



# 3545.不同字符数量最多为K时的最少删除数

```java
class Solution {
    public int minDeletion(String s, int k) {
        char[] sChar = s.toCharArray();
        int[] tempList = new int[26];
        for (int i = 0; i < sChar.length; i++) {
            tempList[sChar[i] - 'a']++;
        }
        Arrays.sort(tempList);
        int result = 0;
        for (int i = 0; i < tempList.length - k; i++) {
            result += tempList[i];
        }
        return result;
    }
}
```

# 3550.数位和等于下标的最小下标

```java
class Solution {
    public int smallestIndex(int[] nums) {
        for (int i = 0; i < nums.length; i++) {
            int temp = nums[i];
            int sum = 0;
            while (temp > 0) {
                sum += temp % 10;
                temp /= 10;
            }
            if (i == sum) {
                return i;
            }
        }
        return -1;
    }
}
```



# 3582.为视频标题生成标签

```java
class Solution {
    public String generateTag(String caption) {
        String[] words = caption.trim().split("\\s+");
        StringBuilder sb = new StringBuilder();
        sb.append('#');
        sb.append(words[0].toLowerCase());
        for (int i = 1; i < words.length; i++) {
            String tempString = words[i].toLowerCase();
            sb.append(tempString.substring(0, 1).toUpperCase());
            for (int j = 1; j < tempString.length(); j++) {
                if (sb.length() == 100) {
                    break;
                }
                sb.append(tempString.charAt(j));
            }
            if (sb.length() == 100) {
                break;
            }
        }
        return sb.length() > 100 ? sb.substring(0, 100) : sb.toString();
    }
}
```



# 3591.检查元素频次是否为质数

```java
class Solution {
    public boolean checkPrimeFrequency(int[] nums) {
        int[] cnt = new int[101];
        for (int i = 0; i < nums.length; i++) {
            cnt[nums[i]]++;
        }
        for (int i = 0; i < cnt.length; i++) {
            if (isZ(cnt[i])) {
                return true;
            }
        }
        return false;
    }
    public boolean isZ(int number) {
        if (number < 2) {
            return false;
        }
        for (int i = 2; i <= Math.sqrt(number); i++) {
            if (number % i == 0) {
                return false;
            }
        }
        return true;
    }
}
```
