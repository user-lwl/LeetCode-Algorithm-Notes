# 3507.移除最小数对使数组有序Ⅰ

```java
class Solution {
    public int minimumPairRemoval(int[] nums) {
        int cnt = 0;
        while (nums.length > 1) {
            boolean isUp = true;
            int minIndex = 0;
            int minSum = nums[0] + nums[1];
            for (int i = 0; i < nums.length - 1; i++) {
                if (nums[i + 1] < nums[i]) {
                    isUp = false;
                }
                if (nums[i] + nums[i + 1] < minSum) {
                    minIndex = i;
                    minSum = nums[i] + nums[i + 1];
                }
            }
            if (isUp) {
                break;
            }
            int[] tempNums = new int[nums.length - 1];
            for (int i = 0; i < minIndex; i++) {
                tempNums[i] = nums[i];
            }
            tempNums[minIndex] = minSum;
            for (int i = minIndex + 1; i < tempNums.length; i++) {
                tempNums[i] = nums[i + 1];
            }
            cnt++;
            nums = tempNums;
        }
        return cnt;
    }
}
```

# 3527.找到最常见的回答

```java
class Solution {
    public String findCommonResponse(List<List<String>> responses) {
        Map<String, Integer> tempMap = new HashMap<>();
        for (int i = 0; i < responses.size(); i++) {
            List<String> tempList = responses.get(i);
            Set<String> tempSet = new HashSet<>();
            for (int j = 0; j < tempList.size(); j++) {
                if (tempSet.add(tempList.get(j))) {
                    tempMap.put(tempList.get(j), tempMap.getOrDefault(tempList.get(j), 0) + 1);
                }
            }
        }
        String[][] tempResult = new String[tempMap.size()][2];
        int index = 0;
        for (String key : tempMap.keySet()) {
            tempResult[index][0] = key;
            tempResult[index][1] = String.valueOf(tempMap.get(key));
            index++;
        }
        Arrays.sort(tempResult, new Comparator<String[]>() {
            @Override
            public int compare(String[] o1, String[] o2) {
                if (o1[1].equals(o2[1])) {
                    return o1[0].compareTo(o2[0]);
                } else {
                    return Integer.compare(Integer.valueOf(o2[1]), Integer.valueOf(o1[1]));
                }
            }
        });
        return tempResult[0][0];
    }
}
```

# 3531.统计被覆盖的建筑

```java
class Solution {
    public int countCoveredBuildings(int n, int[][] buildings) {
        int[] minX = new int[n + 1];
        int[] minY = new int[n + 1];
        int[] maxX = new int[n + 1];
        int[] maxY = new int[n + 1];
        for (int i = 0; i < buildings.length; i++) {
            int x = buildings[i][0];
            int y = buildings[i][1];
            minX[x] = (minX[x] == 0 ? y : Math.min(minX[x], y));
            maxX[x] = (maxX[x] == 0 ? y : Math.max(maxX[x], y));
            minY[y] = (minY[y] == 0 ? x : Math.min(minY[y], x));
            maxY[y] = (maxY[y] == 0 ? x : Math.max(maxY[y], x));
        }
        int result = 0;
        for (int i = 0; i < buildings.length; i++) {
            int x = buildings[i][0];
            int y = buildings[i][1];
            if (y > minX[x] && y < maxX[x] && x > minY[y] && x < maxY[y]) {
                result++;
            }
        }
        return result;
    }
}
```

# 3541.找到频率最高的元音和辅音

```java
class Solution {
    public int maxFreqSum(String s) {
        Map<Character, Integer> yuanMap = new HashMap<>();
        Map<Character, Integer> fuMap = new HashMap<>();
        Set<Character> yuanSet = new HashSet<>();
        yuanSet.add('a');
        yuanSet.add('e');
        yuanSet.add('i');
        yuanSet.add('o');
        yuanSet.add('u');
        for (int i = 0; i < s.length(); i++) {
            char temp = s.charAt(i);
            if (yuanSet.contains(temp)) {
                yuanMap.put(temp, yuanMap.getOrDefault(temp, 0) + 1);
            } else {
                fuMap.put(temp, fuMap.getOrDefault(temp, 0) + 1);
            }
        }
        int yuanMaxCnt = 0;
        int fuMaxCnt = 0;
        for (Character key : yuanMap.keySet()) {
            yuanMaxCnt = Math.max(yuanMaxCnt, yuanMap.get(key));
        }
        for (Character key : fuMap.keySet()) {
            fuMaxCnt = Math.max(fuMaxCnt, fuMap.get(key));
        }
        return yuanMaxCnt + fuMaxCnt;
    }
}
```

```java
class Solution {
    public int minDeletion(String s, int k) {
        char[] sChar = s.toCharArray();
        int[] tempList = new int[26];
        for (int i = 0; i < sChar.length; i++) {
            tempList[sChar[i] - 'a']++;
        }
        Arrays.sort(tempList);
        int result = 0;
        for (int i = 0; i < tempList.length - k; i++) {

        }
```

# 3550.数位和等于下标的最小下标

```java
class Solution {
    public int smallestIndex(int[] nums) {
        for (int i = 0; i < nums.length; i++) {
            int temp = nums[i];
            int sum = 0;
            while (temp > 0) {
                sum += temp % 10;
                temp /= 10;
            }
            if (i == sum) {
                return i;
            }
        }
        return -1;
    }
}
```

# 3582.为视频标题生成标签

```java
class Solution {
    public String generateTag(String caption) {
        String[] words = caption.trim().split("\\s+");
        StringBuilder sb = new StringBuilder();
        sb.append('#');
        sb.append(words[0].toLowerCase());
        for (int i = 1; i < words.length; i++) {
            String tempString = words[i].toLowerCase();
            sb.append(tempString.substring(0, 1).toUpperCase());
            for (int j = 1; j < tempString.length(); j++) {
                if (sb.length() == 100) {
                    break;
                }
                sb.append(tempString.charAt(j));
            }
            if (sb.length() == 100) {
                break;
            }
        }
        return sb.length() > 100 ? sb.substring(0, 100) : sb.toString();
    }
}
```

# 3591.检查元素频次是否为质数

```java
class Solution {
    public boolean checkPrimeFrequency(int[] nums) {
        int[] cnt = new int[101];
        for (int i = 0; i < nums.length; i++) {
            cnt[nums[i]]++;
        }
        for (int i = 0; i < cnt.length; i++) {
            if (isZ(cnt[i])) {
                return true;
            }
        }
        return false;
    }
    public boolean isZ(int number) {
        if (number < 2) {
            return false;
        }
        for (int i = 2; i <= Math.sqrt(number); i++) {
            if (number % i == 0) {
                return false;
            }
        }
        return true;
    }
}
```
