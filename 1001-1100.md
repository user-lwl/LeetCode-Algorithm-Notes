# 1002.查找共用字符

```java
class Solution {
    public List<String> commonChars(String[] words) {
        int[][] tempList = new int[words.length][26];
        List<String> resultList = new ArrayList<>();
        for (int i = 0; i < words.length; i++) {
            char[] wordChar = words[i].toCharArray();
            for (int j = 0; j < wordChar.length; j++) {
                tempList[i][wordChar[j] - 'a']++;
            }
        }
        for (int i = 0; i < 26; i++) {
            boolean isT = true;
            int min = 999;
            for (int j = 0; j < tempList.length; j++) {
                int cnt = tempList[j][i];
                if (cnt == 0) {
                    isT = false;
                    break;
                }
                min = Math.min(min, cnt);
            }
            while(isT && min > 0) {
                char resultChar = (char) ('a' + i);
                resultList.add(String.valueOf(resultChar));
                min--;
            }
        }
        return resultList;
    }
}
```

# 1004.最大连续1的个数Ⅲ

```java
class Solution {
    public int longestOnes(int[] nums, int k) {
        int left = 0;
        int right = 0;
        int zeroCount = 0;
        int max = 0;
        int len = nums.length;
        while (right < len) {
            zeroCount += (nums[right] - 1) * (-1);
            while (zeroCount > k) {
                zeroCount -= (nums[left] - 1) * (-1);
                left++;
            }
            max = Math.max(max, right - left + 1);
            right++;
        }
        return max;
    }
}
```

# 1005.K次取反后最大化的数组和

```java
class Solution {
    public int largestSumAfterKNegations(int[] nums, int k) {
        Arrays.sort(nums);
        int minZ = 999;
        int maxF = -999;
        int sum = 0;
        boolean isHave = false;
        if (nums[0] < 0 && nums[nums.length - 1] > 0) {
            isHave = true;
        } else if (nums[0] > 0) {
            minZ = nums[0];
        } else if (nums[nums.length - 1] < 0) {
            maxF = nums[nums.length - 1];
        }
        for (int i = 0; i < nums.length; i++) {
            if (isHave && i < nums.length - 1 && nums[i] * nums[i + 1] <= 0) {
                minZ = nums[i + 1];
                maxF = nums[i];
            }
            if (k > 0 && nums[i] < 0) {
                sum -= nums[i];
                k--;
            } else {
                sum += nums[i];
            }
        }
        if (k % 2 != 0) {
            return Math.max(sum - minZ * 2, sum + maxF * 2);
        }
        return sum;
    }
}
```



# 1019.链表中的下一个更大节点

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public int[] nextLargerNodes(ListNode head) {
        ListNode node = head;
        Stack<ListNode> stack = new Stack<>();
        while (node != null) {
            stack.push(node);
            node = node.next;
        }
        int size = stack.size();
        int[] nodeValList = new int[size];
        int[] result = new int[size];
        int index = size - 2;
        result[size - 1] = 0;
        node = stack.pop();
        int max = node.val;
        nodeValList[size - 1] = node.val;
        while (!stack.empty()) {
            ListNode tempNode = stack.pop();
            nodeValList[index] = tempNode.val;
            if (tempNode.val >= max) {
                result[index--] = 0;
                max = tempNode.val;
            } else {
                for (int i = index; i < size; i++) {
                    if (nodeValList[i] > tempNode.val) {
                        result[index--] = nodeValList[i];
                        break;
                    }
                }
            }
        }
        return result;
    }
}
```

# 1021.删除最外层的括号

```java
class Solution {
    public String removeOuterParentheses(String s) {
        char[] sChar = s.toCharArray();
        StringBuilder sb = new StringBuilder();
        LinkedList<Character> tempList = new LinkedList<>();
        int leftCnt = 0;
        for (int i = 0; i < sChar.length; i++) {
            char temp = sChar[i];
            tempList.addLast(temp);
            if (temp == '(') {
                leftCnt++;
            } else {
                leftCnt--;
            }
            if (leftCnt == 0) {
                tempList.removeFirst();
                tempList.removeLast();
                while (!tempList.isEmpty()) {
                    sb.append(tempList.removeFirst());
                }
            }
        }
        return sb.toString();
    }
}
```

# 1022.从根到叶的二进制数之和

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int sumRootToLeaf(TreeNode root) {
        return getSum(root, 0);
    }
    public int getSum(TreeNode node, int value) {
        if (node == null) {
            return 0;
        } 
        value = value * 2 + node.val;
        if (node.left == null && node.right == null) {
            return value;
        } else {
            return getSum(node.left, value) + getSum(node.right, value);
        }
    }
}
```

# 1030.距离顺序排列矩阵单元格

```java
class Solution {
    public int[][] allCellsDistOrder(int rows, int cols, int rCenter, int cCenter) {
        int[][] result = new int[rows * cols][2];
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                result[i * cols + j][0] = i;
                result[i * cols + j][1] = j;
            }
        }
        Arrays.sort(result, new Comparator<int[]>() {
            public int compare(int[] a, int[] b) {
                return (Math.abs(a[0] - rCenter) + Math.abs(a[1] - cCenter)) - (Math.abs(b[0] - rCenter) + Math.abs(b[1] - cCenter));
            }
        });
        return result;
    }
}
```

# 1046.最后一块石头的重量

```java
class Solution {
    public int lastStoneWeight(int[] stones) {
        int n = stones.length;
        for (int i = n - 1; i > 0; i--) {
            bubblSort(stones);
            stones[i - 1] = stones[i] - stones[i - 1];
        }
        return stones[0];
    }
    public void bubblSort(int[] a) {
        int len = a.length;
        for (int i = len - 1; i > 0; i--) {
            for (int j = 0; j < i; j++) {
                if (a[j] > a[i]) {
                    int temp = a[j];
                    a[j] = a[i];
                    a[i] = temp;
                }
            }
        }
    }
}
```

# 1051.高度检查器

```java
class Solution {
    public int heightChecker(int[] heights) {
        int[] expected = countingSort(heights);
        int count = 0;
        for (int i = 0; i < heights.length; i++) {
            if (heights[i] != expected[i]) {
                count++;
            }
        }
        return count;
    }
    public int[] countingSort(int[] num) {
        int[] counting = new int[101];
        for (int i = 0; i < num.length; i++) {
             counting[num[i]]++;
        }
        int[] result = new int[num.length];
        int index = 0;
        for (int i = 0; i < counting.length; i++) {
            while(counting[i] > 0) {
                result[index++] = i;
                counting[i]--;
            }
        }
        return result;
    }
}
```
