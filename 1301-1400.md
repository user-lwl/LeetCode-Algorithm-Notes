```java
class Solution {
    public int[] sumZero(int n) {
        int[] result = new int[n];
        int index = 0;
        while (n > 1) {
            result[index++] = n;
            result[index++] = -n;
            n -= 2;
        }
        if (n == 1) {
            result[index] = 0;
        }
        return result;
    }
}
```

# 1310.子数组异或查询

```java
class Solution {
    public int[] xorQueries(int[] arr, int[][] queries) {
        int len = arr.length;
        int[] prefix = new int[len];
        int[] result = new int[queries.length];
        prefix[0] = arr[0];
        for (int i = 1; i < len; i++) {
            prefix[i] = prefix[i - 1] ^ arr[i];
        }
        for (int i = 0; i < queries.length; i++) {
            int left = queries[i][0];
            int right = queries[i][1];
            if (left == 0) {
                result[i] = prefix[right];
            } else {
                result[i] = prefix[right] ^ prefix[left - 1];
            }
        }
        return result;
    }
}
```



# 1314.矩阵区域和

```java
class Solution {
    public int[][] matrixBlockSum(int[][] mat, int k) {
        int m = mat.length;
        int n = mat[0].length;
        int[][] result = new int[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                result[i][j] = fix(mat, i, j, k);
            }
        }
        return result;
    }

    public int fix(int[][] mat, int i, int j, int k) {
        int res = 0;
        for (int r = i - k; r <= i+k ;r++) {
            for (int c = j - k; c <= j+k; c++) {
                if (0 <= r && r < mat.length && 0 <= c && c < mat[0].length) {
                    res += mat[r][c];
                }
            }
        }
        return res;
    }
}
```

```java
        int[] result = new int[]{1, n - 1};

                }
            }

                    break;
                }
            }

        }
        return result;
    }
```

```java
        int[] tempArray = new int[arr.length];

        for (int i = 0; i < arr.length; i++) {
            if (resultSet.add(arr[i])) {
                tempArray[index++] = arr[i];
            }
        }
        for (int i = index; i < arr.length; i++) {
            tempArray[i] = Integer.MAX_VALUE;
        }
        Arrays.sort(tempArray);

            arr[i] = resultMap.get(arr[i]);
        }
        return arr;
    }
}
```

```java
class Solution {
```

```java
        int[] result = new int[mat.length];
        for (int i = 0; i < mat.length; i++) {
            int tempSum = 0;

                tempSum += mat[i][j];

            if (cnt[i] > 0) {
                List<Integer> tempList = resultMap.get(i);
                for (int j = 0; j < tempList.size(); j++) {
                    if (index >= k) {
                        break;
                    }
                    returnResult[index++] = tempList.get(j);
                }
                if (index >= k) {
                    break;
                }
            }
        }
        return returnResult;
```

```java
        int zeroCnt = 0;

            arrSet.add(arr[i]);
            if (arr[i] == 0) {
                zeroCnt++;
            }
        }

            if (tempInt == 0) {
                if (zeroCnt > 1) {
                    return true;

            }
        }
        return false;
    }
```

```java
class Solution {
    public int countNegatives(int[][] grid) {
        int number = 0;
        for (int i = 0; i < grid.length; i++) {
            number += grid[0].length - binarySearch(grid[i]);
        }
        return number;
    }
    public int binarySearch(int[] list) {
        int left = -1;
        int right = list.length;
        while (left + 1 < right) {
            int mid = (left + right) / 2;
            if (list[mid] < 0) {
                right = mid;
            } else {
                left = mid;
            }
        }
        return right;
    }
}
```

# 1367.二叉树中的链表

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    public boolean isSubPath(ListNode head, TreeNode root) {
        if (root == null) {
            return false;
        }
        return isT(head, root) || isSubPath(head, root.left) || isSubPath(head, root.right);
    }

    public boolean isT(ListNode head, TreeNode root) {
        if (root == null) {
            return false;
        }
        if (root.val == head.val) {
            if (head.next == null) {
                return true;
            } else {
                return isT(head.next, root.left) || isT(head.next, root.right);
            }
        }
        return false;
    }
}
```

# 1370.上升下降字符串

```java
class Solution {
    public String sortString(String s) {
        char[] initChar = s.toCharArray();
        int[] initList = new int[26];
        for (int i = 0; i < initChar.length; i++) {
            initList[initChar[i] - 'a']++;
        }
        StringBuilder sb = new StringBuilder();
        while (sb.length() < s.length()) {
            for (int i = 0; i < 26; i++) {
                if (initList[i] > 0) {
                    sb.append((char) (i + 'a'));
                    initList[i]--;
                }
            }
            for (int i = 25; i >= 0; i--) {
                if (initList[i] > 0) {
                    sb.append((char) (i + 'a'));
                    initList[i]--;
                }
            }
        }
        return sb.toString();
    }
}
```

```java
class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        List<Integer> result = new ArrayList<>();
        int[] min = new int[matrix.length];
        int[] max = new int[matrix[0].length];
        for (int i = 0; i < matrix.length; i++) {

            for (int j = 0; j < matrix[0].length; j++) {
                if (matrix[i][j] < mini) {
                    mini = matrix[i][j];
                }
            }

        }

                if (matrix[j][i] > maxi) {
                    maxi = matrix[j][i];
                }
            }

        }
        for (int i = 0; i < min.length; i++) {
            for (int j = 0; j < max.length; j++) {
                if (min[i] == max[j]) {
                    result.add(min[i]);
                }
            }
        }
        return result;
    }
```

```java
        int result = -1;
        for (int i = 0; i < arr.length; i++) {
```

# 1399.统计最大组的数目

```java
class Solution {
    public int countLargestGroup(int n) {
        Map<Integer, List<Integer>> resultMap = new HashMap<>();
        for (int i = 1; i <= n; i++) {
            int key = 0;
            int tempInt = i;
            while (tempInt > 0) {
                key += tempInt % 10;
                tempInt = tempInt / 10;
            }
            List<Integer> tempList = resultMap.getOrDefault(key, new ArrayList<>());
            tempList.add(i);
            resultMap.put(key, tempList);
        }
        int result = 0;
        int cnt = 0;
        for (Integer key : resultMap.keySet()) {
            List<Integer> tempList = resultMap.get(key);
            if (tempList.size() > result) {
                result = tempList.size();
                cnt = 1;
            } else if (tempList.size() == result) {
                cnt++;
            }
        }
        return cnt;
    }
}
```
