# 1903.字符串中的最大奇数

```java
class Solution {
    public String largestOddNumber(String num) {
        int preNode = 0;
        int max = -1;
        for (int i = num.length() - 1; i >= 0; i--) {
            int number = num.charAt(i) - '0';
            if (number % 2 == 1) {
                max = i;
                break;
            }
        }
        String result = num.substring(0, max + 1);
        return result;
    }
}
```



# 1913.两个数对之间的最大乘积差

```java
class Solution {
    public int maxProductDifference(int[] nums) {
        int len = nums.length;
        Arrays.sort(nums);
        return nums[len - 2] * nums[len - 1] - nums[0] * nums[1];
    }
}
```

```java
        int len = nums.length;
```

# 1935.可以输入的最大单词数

```java
class Solution {
    public int canBeTypedWords(String text, String brokenLetters) {
        String[] word = text.split(" ");
        char[] tempChar = brokenLetters.toCharArray();
        Set<Character> resultSet = new HashSet<>();
        for (int i = 0; i < tempChar.length; i++) {
            resultSet.add(tempChar[i]);
        }
        int result = 0;
        for (int i = 0; i < word.length; i++) {
            boolean isCan = true;
            for (int j = 0; j < word[i].length(); j++) {
                if (resultSet.contains(word[i].charAt(j))) {
                    isCan = false;
                    break;
                }
            }
            if (isCan) {
                result++;
            }
        }
        return result;
    }
}
```

# 1941.检查是否所有字符出现次数相同

```java
class Solution {
    public boolean areOccurrencesEqual(String s) {
        Map<Character, Integer> resultMap = new HashMap<>();
        char[] sChar = s.toCharArray();
        for (int i = 0; i < sChar.length; i++) {
            resultMap.put(sChar[i], resultMap.getOrDefault(sChar[i], 0) + 1);
        }
        int cnt = -1;
        for (Character key : resultMap.keySet()) {
            int number = resultMap.get(key);
            if (cnt == -1) {
                cnt = number;
            }
            if (cnt != number) {
                return false;
            }
        }
        return true;
    }
}
```

# 1984.学生分数的最小差值

```java
class Solution {
    public int minimumDifference(int[] nums, int k) {
        insertSort(nums);
        int len = nums.length;
        int left = 0;
        int right = k - 1;
        int sum = Integer.MAX_VALUE;
        while (right < len) {
            int temp = nums[right] - nums[left];
            sum = Math.min(sum, temp);
            right++;
            left++;
        }
        return sum;
    }
    public void insertSort(int[] a) {
        int len = a.length;
        for (int i = 1; i < len; i++) {
            int temp = a[i];
            int j = i - 1; 
            while(j >= 0) {
                if (a[j] > temp) {
                    a[j + 1] = a[j];
                    j--;
                } else {
                    break;
                }
            }
            a[j + 1] = temp;
        }
    }
}
```

# 1995.统计特殊四元组

```java
class Solution {
    public int countQuadruplets(int[] nums) {
        Map<Integer, Integer> resultMap = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            resultMap.put(nums[i], resultMap.getOrDefault(nums[i], 0) + 1);
        }
        int result = 0;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                for (int k = j + 1; k < nums.length; k++) {
                    for (int m = k + 1; m < nums.length; m++) {
                        if (nums[i] + nums[j] + nums[k] == nums[m]) {
                            result++;
                        }
                    }
                }
            }
        }
        return result;
    }
}
```



# 2000.反转单词前缀

```java
class Solution {
    public String reversePrefix(String word, char ch) {
        char[] wordChar = word.toCharArray();
        Stack<Character> tempStack = new Stack<>();
        int index = -1;
        for (int i = 0; i < wordChar.length; i++) {
            tempStack.push(wordChar[i]);
            if (wordChar[i] == ch) {
                index = i + 1;
                break;
            }
        }
        if (index == -1) {
            return word;
        }
        StringBuilder sb = new StringBuilder();
        while (!tempStack.isEmpty()) {
            sb.append(tempStack.pop());
        }
        for (; index < wordChar.length; index++) {
            sb.append(wordChar[index]);
        }
        return sb.toString();
    }
}
```
