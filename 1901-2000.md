# 1925.统计平方和三元组的数目

```java
class Solution {
    public int countTriples(int n) {
        int result = 0;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                int number = i * i + j * j;
                int sqrt = (int) Math.sqrt(number);
                if (sqrt <= n && number == sqrt * sqrt) {
                    result++;
                }
            }
        }
        return result;
    }
}
```

# 1941.检查是否所有字符出现次数相同

```java
class Solution {
    public boolean areOccurrencesEqual(String s) {
        Map<Character, Integer> resultMap = new HashMap<>();
        char[] sChar = s.toCharArray();
        for (int i = 0; i < sChar.length; i++) {
            resultMap.put(sChar[i], resultMap.getOrDefault(sChar[i], 0) + 1);
        }
        int cnt = -1;
        for (Character key : resultMap.keySet()) {
            int number = resultMap.get(key);
            if (cnt == -1) {
                cnt = number;
            }
            if (cnt != number) {
                return false;
            }
        }
        return true;
    }
}
```

# 1984.学生分数的最小差值

```java
class Solution {
    public int minimumDifference(int[] nums, int k) {
        insertSort(nums);
        int len = nums.length;
        int left = 0;
        int right = k - 1;
        int sum = Integer.MAX_VALUE;
        while (right < len) {
            int temp = nums[right] - nums[left];
            sum = Math.min(sum, temp);
            right++;
            left++;
        }
        return sum;
    }
    public void insertSort(int[] a) {
        int len = a.length;
        for (int i = 1; i < len; i++) {
            int temp = a[i];
            int j = i - 1; 
            while(j >= 0) {
                if (a[j] > temp) {
                    a[j + 1] = a[j];
                    j--;
                } else {
                    break;
                }
            }
            a[j + 1] = temp;
        }
    }
}
```

# 1991.找到数组的中间位置

```java
class Solution {
    public int findMiddleIndex(int[] nums) {
        int left = 0;
        int right = 0;
        for (int i = 1; i < nums.length; i++) {
            right += nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            if (left == right) {
                return i;
            }
            if (i == nums.length - 1) {
                break;
            }
            left += nums[i];
            right -= nums[i + 1];
        }
        return -1;
    }
}
```

# 2000.反转单词前缀

```java
class Solution {
    public String reversePrefix(String word, char ch) {
        char[] wordChar = word.toCharArray();
        Stack<Character> tempStack = new Stack<>();
        int index = -1;
        for (int i = 0; i < wordChar.length; i++) {
            tempStack.push(wordChar[i]);
            if (wordChar[i] == ch) {
                index = i + 1;
                break;
            }
        }
        if (index == -1) {
            return word;
        }
        StringBuilder sb = new StringBuilder();
        while (!tempStack.isEmpty()) {
            sb.append(tempStack.pop());
        }
        for (; index < wordChar.length; index++) {
            sb.append(wordChar[index]);
        }
        return sb.toString();
    }
}
```
