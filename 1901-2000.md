# 1913.两个数对之间的最大乘积差

```java
class Solution {
    public int maxProductDifference(int[] nums) {
        int len = nums.length;
        Arrays.sort(nums);
        return nums[len - 2] * nums[len - 1] - nums[0] * nums[1];
    }
}
```

# 1920.基于排列构建数组

```java
class Solution {
    public int[] buildArray(int[] nums) {
        int len = nums.length;
        int[] result = new int[len];
        for (int i = 0; i < len; i++) {
            result[i] = nums[nums[i]];
        }
        return result;
    }
}
```

# 1929.数组串联

```java
class Solution {
    public int[] getConcatenation(int[] nums) {
        int len = nums.length;
        int[] result = new int[len * 2];
        for (int i = 0; i < len; i++) {
            result[i] = nums[i];
            result[len + i] = nums[i];
        }
        return result;
    }
}
```

# 1961.检查字符串是否为数组前缀

```java
class Solution {
    public boolean isPrefixString(String s, String[] words) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < words.length; i++) {
            sb.append(words[i]);
            if (sb.toString().equals(s)) {
                return true;
            }
        }
        return false;
    }
}
```



# 1984.学生分数的最小差值

```java
class Solution {
    public int minimumDifference(int[] nums, int k) {
        insertSort(nums);
        int len = nums.length;
        int left = 0;
        int right = k - 1;
        int sum = Integer.MAX_VALUE;
        while (right < len) {
            int temp = nums[right] - nums[left];
            sum = Math.min(sum, temp);
            right++;
            left++;
        }
        return sum;
    }
    public void insertSort(int[] a) {
        int len = a.length;
        for (int i = 1; i < len; i++) {
            int temp = a[i];
            int j = i - 1; 
            while(j >= 0) {
                if (a[j] > temp) {
                    a[j + 1] = a[j];
                    j--;
                } else {
                    break;
                }
            }
            a[j + 1] = temp;
        }
    }
}
```

# 2000.反转单词前缀

```java
class Solution {
    public String reversePrefix(String word, char ch) {
        char[] wordChar = word.toCharArray();
        Stack<Character> tempStack = new Stack<>();
        int index = -1;
        for (int i = 0; i < wordChar.length; i++) {
            tempStack.push(wordChar[i]);
            if (wordChar[i] == ch) {
                index = i + 1;
                break;
            }
        }
        if (index == -1) {
            return word;
        }
        StringBuilder sb = new StringBuilder();
        while (!tempStack.isEmpty()) {
            sb.append(tempStack.pop());
        }
        for (; index < wordChar.length; index++) {
            sb.append(wordChar[index]);
        }
        return sb.toString();
    }
}
```
