# LCP 01.猜数字

```java
class Solution {
    public int game(int[] guess, int[] answer) {
        int number = 0;
        for (int i = 0; i < guess.length; i++) {
            if (guess[i] == answer[i]) {
                number++;
            }
        }
        return number;
    }
}
```

# LCP 17.速算机器人

```java
class Solution {
    public int calculate(String s) {
        int x = 1;
        int y = 0;
        char[] c = s.toCharArray();
        for (int i = 0; i < c.length; i++) {
            if (c[i] == 'A') {
                x = 2 * x + y;
            }
            if (c[i] == 'B') {
                y = 2 * y + x;
            }
        }
        return x + y;
    }
}
```

# LCP 18.早餐组合

```java
class Solution {
    public int breakfastNumber(int[] staple, int[] drinks, int x) {
        Arrays.sort(drinks);
        int sum = 0;
        for (int i = 0; i < staple.length; i++) {
            if (staple[i] >= x) {
                continue;
            }
            sum += binarySearch(drinks, x - staple[i]);
            sum %= 1000000007;
        }
        return sum;
    }
    public int binarySearch(int[] list, int x) {
        int left = -1;
        int right = list.length;
        while (left + 1 < right) {
            int mid = (int) (((long)left + (long)right) / 2);
            if (list[mid] > x) {
                right = mid;
            } else {
                left = mid;
            }
        }
        return right;
    }
}
```

# LCP 28.采购方案

```java
class Solution {
    public int purchasePlans(int[] nums, int target) {
        int result = 0;
        Arrays.sort(nums);
        int left = 0;
        int right = nums.length - 1;
        while (left < right) {
            if (nums[left] + nums[right] > target) {
                right--;
            } else {
                result += right - left;
                left++;
            }
            result %= 1000000007;
        }
        return result % 1000000007;
    }
}
```

# LCP 39.无人机方阵

```java
class Solution {
    public int minimumSwitchingTimes(int[][] source, int[][] target) {
        int[] color = new int[10001];
        for (int i = 0; i < source.length; i++) {
            for (int j = 0; j < source[0].length; j++) {
                color[source[i][j]]++;
            }
        }
        int result = 0;
        for (int i = 0; i < target.length; i++) {
            for (int j = 0; j < target[0].length; j++) {
                color[target[i][j]]--;
                if (color[target[i][j]] < 0) {
                    result++;
                }
            }
        }
        return result;
    }
}
```

# LCP 40.心算挑战

```java
class Solution {
    public int maximumScore(int[] cards, int cnt) {
        Integer[] cardsIntegerList = Arrays.stream(cards).boxed().toArray(Integer[]::new);
        Arrays.sort(cardsIntegerList, Collections.reverseOrder());
        int jiNumber = -1;
        int ouNubmer = -1;
        int result = 0;
        int index = 0;
        int csJiNumber = -1;
        int csOuNumber = -1;
        for (int i = 0; i < cnt; i++) {
            result += cardsIntegerList[i];
            if (cardsIntegerList[i] % 2 == 0) {
                ouNubmer = cardsIntegerList[i];
            } else {
                jiNumber = cardsIntegerList[i];
            }
            index = i;
        }
        for (int i = index + 1; i < cardsIntegerList.length; i++) {
            if (cardsIntegerList[i] % 2 == 0 && csOuNumber == -1) {
                csOuNumber = cardsIntegerList[i];
            } else if (cardsIntegerList[i] % 2 != 0 && csJiNumber == -1) {
                csJiNumber = cardsIntegerList[i];
            }
        }
        if (result % 2 != 0) {
            if (ouNubmer == -1 && csOuNumber == -1) {
                return 0;
            } else {
                int temp1 = -1;
                int temp2 = -1;
                if (jiNumber != -1 && csOuNumber != -1) {
                    temp1 = result - jiNumber + csOuNumber;
                }
                if (ouNubmer != -1 && csJiNumber != -1) {
                    temp2 = result - ouNubmer + csJiNumber;
                }
                result = Math.max(temp1, temp2);
            }
        }
        return Math.max(result, 0);
    }
}
```

# LCP 44.开幕式焰火

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    Set<Integer> set;
    public int numColor(TreeNode root) {
        set = new HashSet<>();
        findColor(root);
        return set.size();
    }
    public void findColor(TreeNode node) {
        if (node == null) {
            return;
        }
        set.add(node.val);
        findColor(node.left);
        findColor(node.right);
    }
}
```

# LCP 66.最小展台数量

```java
class Solution {
    public int minNumBooths(String[] demand) {
        int[] result = new int[26];
        for (int i = 0; i < demand.length; i++) {
            int[] temp = new int[26];
            for (int j = 0; j < demand[i].length(); j++) {
                int table = demand[i].charAt(j) - 'a';
                temp[table]++;
            }
            for (int j = 0; j < temp.length; j++) {
                result[j] = Math.max(result[j], temp[j]);
            }
        }
        int resultInt = 0;
        for (int i = 0; i < result.length; i++) {
            resultInt += result[i];
        }
        return resultInt;
    }
}
```
