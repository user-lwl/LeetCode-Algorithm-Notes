# LCP 02.分式化简

```java
class Solution {
    public int[] fraction(int[] cont) {
        if (cont.length == 1) {
            return new int[]{cont[0], 1};
        }
        int fm = 1;
        int fz = cont[cont.length - 1];
        for (int i = cont.length - 2; i >= 0; i--) {
            int temp = fm;
            fm = fz;
            fz = temp;
            fz += cont[i] * fm;
        }
        return new int[]{fz, fm};
    }
}
```

# LCP 07.传递信息

```java
class Solution {

    List<Integer>[] tempList;

    int result = 0;

    public int numWays(int n, int[][] relation, int k) {
        tempList = new List[n];
        for (int i = 0; i < n; i++) {
            tempList[i] = new ArrayList<>();
        }
        for (int i = 0; i < relation.length; i++) {
            tempList[relation[i][0]].add(relation[i][1]);
        }
        dfs(0, 0, k, n);
        return result;
    }

    public void dfs(int node, int level, int k, int n) {
        if (level > k) {
            return;
        }
        if (level == k && node == n - 1) {
            result++;
            return;
        }
        List<Integer> tempThisList = tempList[node];
        for (int i = 0; i < tempThisList.size(); i++) {
            dfs(tempThisList.get(i), level + 1, k, n);
        }
    }
}
```

```java
class Solution {
    public int calculate(String s) {
        int x = 1;
        int y = 0;
        char[] c = s.toCharArray();
        for (int i = 0; i < c.length; i++) {
            if (c[i] == 'A') {
                x = 2 * x + y;
            }
            if (c[i] == 'B') {
                y = 2 * y + x;
            }
        }
        return x + y;
    }
}
```

# LCP 18.早餐组合

```java
class Solution {
    public int breakfastNumber(int[] staple, int[] drinks, int x) {
        Arrays.sort(drinks);
        int sum = 0;
        for (int i = 0; i < staple.length; i++) {
            if (staple[i] >= x) {
                continue;
            }
            sum += binarySearch(drinks, x - staple[i]);
            sum %= 1000000007;
        }
        return sum;
    }
    public int binarySearch(int[] list, int x) {
        int left = -1;
        int right = list.length;
        while (left + 1 < right) {
            int mid = (int) (((long)left + (long)right) / 2);
            if (list[mid] > x) {
                right = mid;
            } else {
                left = mid;
            }
        }
        return right;
    }
}
```

# LCP 22.黑白方格画

```java
class Solution {
    public int paintingPlan(int n, int k) {
        if (k == 0 || k == n * n) {
            return 1;
        }
        int result = 0;
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= n; j++) {
                if (i * n + j * n - i * j == k) {
                    result += mathC(i, n) * mathC(j, n);
                }
            }
        }
        return result;
    }

    public int mathC(int m, int n) {
        int result = 1;
        for (int i = 0; i < m; i++) {
            result *= (n - i);
        }
        for (int i = 1; i <= m; i++) {
            result /= i;
        }
        return result;
    }
}
```



# LCP 28.采购方案

```java
class Solution {
    public int purchasePlans(int[] nums, int target) {
        int result = 0;
        Arrays.sort(nums);
        int left = 0;
        int right = nums.length - 1;
        while (left < right) {
            if (nums[left] + nums[right] > target) {
                right--;
            } else {
                result += right - left;
                left++;
            }
            result %= 1000000007;
        }
        return result % 1000000007;
    }
}
```

# LCP 33.蓄水

```java
class Solution {
    public int storeWater(int[] bucket, int[] vat) {
        int maxVat = 0;
        for (int i = 0; i < vat.length; i++) {
            maxVat = Math.max(maxVat, vat[i]);
        }
        if (maxVat == 0) {
            return 0;
        }
        int result = 9999999;
        for (int i = 1; i <= maxVat; i++) {
            int upBucket = 0;
            for (int j = 0; j < bucket.length; j++) {
                int needBucket = vat[j] / i + (vat[j] % i == 0 ? 0 : 1);
                if (needBucket > bucket[j]) {
                    upBucket += needBucket - bucket[j];
                }
            }
            result = Math.min(upBucket + i, result);
        }
        return result;
    }
}
```

```java
        int jiNumber = -1;

            } else {
                jiNumber = cardsIntegerList[i];
            }
            index = i;
        }
        for (int i = index + 1; i < cardsIntegerList.length; i++) {
            if (cardsIntegerList[i] % 2 == 0 && csOuNumber == -1) {
                csOuNumber = cardsIntegerList[i];
            } else if (cardsIntegerList[i] % 2 != 0 && csJiNumber == -1) {
                csJiNumber = cardsIntegerList[i];
            }
        }
        if (result % 2 != 0) {
            if (ouNubmer == -1 && csOuNumber == -1) {
                return 0;
            } else {
                int temp1 = -1;
                int temp2 = -1;
                if (jiNumber != -1 && csOuNumber != -1) {
                    temp1 = result - jiNumber + csOuNumber;
                }
                if (ouNubmer != -1 && csJiNumber != -1) {
                    temp2 = result - ouNubmer + csJiNumber;
                }

        }
```

```java
        for (int i = 0; i < operations.length; i++) {
            int src = operations[i][0];
            int tar = operations[i][1];
```

```java
class Solution {
    public int perfectMenu(int[] materials, int[][] cookbooks, int[][] attribute, int limit) {
        int result = -1;
        for (int i = 1; i < Math.pow(2, cookbooks.length); i++) {
            int tempBf = 0;
            int tempMv = 0;
            int[] cost = new int[5];
            for (int j = 0; j < cookbooks.length; j++) {
                if (((i >> j) & 1) == 1) {
                    cost[0] += cookbooks[j][0];
                    cost[1] += cookbooks[j][1];
                    cost[2] += cookbooks[j][2];
                    cost[3] += cookbooks[j][3];
                    cost[4] += cookbooks[j][4];
                    tempBf += attribute[j][1];
                    tempMv += attribute[j][0];
                }
            }
            if (cost[0] > materials[0] || cost[1] > materials[1] || cost[2] > materials[2] || cost[3] > materials[3] || cost[4] > materials[4] || tempBf < limit) {

        }

    }
```

```java
class Solution {
```

```java
        int[] tempRes = new int[temperatureA.length - 1];
        for (int i = 0; i < tempRes.length; i++) {
            int tempA = temperatureA[i + 1] - temperatureA[i];
            int tempB = temperatureB[i + 1] - temperatureB[i];
            if ((tempA == 0 && tempB == 0) || tempA * tempB > 0) {
                tempRes[i] = 1;

            int cnt = 0;
            while (i < tempRes.length && tempRes[i] == 1) {

                result = Math.max(result, cnt);
                i++;
            }
        }
        return result;
    }
}
```

```java
class Solution {
    public int[] supplyWagon(int[] supplies) {
        int finalLen = supplies.length / 2;
        while (supplies.length > finalLen) {
            int[] temp = new int[supplies.length - 1];
            int index = -1;
            int minSum = 99999;
            for (int i = 0; i < supplies.length - 1; i++) {
                if (supplies[i] + supplies[i + 1] < minSum) {
                    index = i;
                    minSum = supplies[i] + supplies[i + 1];
                }
            }
            for (int i = 0; i < index; i++) {
                temp[i] = supplies[i];
            }
            temp[index] = supplies[index] + supplies[index + 1];
            for (int i = index + 1; i < temp.length; i++) {
                temp[i] = supplies[i + 1];
            }
            supplies = temp;
        }
        return supplies;
    }
}
```
