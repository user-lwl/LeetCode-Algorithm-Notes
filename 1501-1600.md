# 1502.判断能否形成等差数列

```java
class Solution {
    public boolean canMakeArithmeticProgression(int[] arr) {
        int len = arr.length;
        for (int i = 0; i < len; i++) {
            int minIndex = i;
            for (int j = i; j < len; j++) {
                if (arr[j] < arr[minIndex]) {
                    minIndex = j;
                }
            }
            int temp = arr[minIndex];
            arr[minIndex] = arr[i];
            arr[i] = temp;
        }
        int sex = arr[1] - arr[0];
        for (int i = 0; i < len - 1; i++) {
            if (arr[i + 1] - arr[i] != sex) {
                return false;
            }
        }
        return true;
    }
}
```

# 1512.好数对的数目

```java
class Solution {
    public int numIdenticalPairs(int[] nums) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int num : nums) {
            map.put(num, map.getOrDefault(num, 0) + 1);
        }
        int result = 0;
        for (Integer key : map.keySet()) {
            int val = map.get(key);
            result += val * (val - 1) / 2;
        }
        return result;
    }
}
```

# 1534.统计好三元组

```java
class Solution {
    public int countGoodTriplets(int[] arr, int a, int b, int c) {
        int result = 0;
        for (int i = 0; i< arr.length; i++) {
            for (int j = i + 1; j < arr.length; j++) {
                for (int k = j + 1; k < arr.length; k++) {
                    if (Math.abs(arr[i] - arr[j]) <= a && Math.abs(arr[j] - arr[k]) <= b && Math.abs(arr[i] - arr[k]) <= c) {
                        result++;
                    }
                }
            }
        }
        return result;
    }
}
```

# 1561.你可以获得的最大硬币数目

```java
class Solution {
    public int maxCoins(int[] piles) {
        heapSort(piles);
        int len = piles.length - 2;
        int sum = 0;
        int count = len / 3;
        while (count >= 0) {
            sum += piles[len];
            len -= 2;
            count--;
        }
        return sum;
    }
    public void swap(int[] nums, int left, int right) {
        int temp = nums[left];
        nums[left] = nums[right];
        nums[right] = temp;
    }
    public void maxHeapify(int[] nums, int left, int right) {
        int son = left * 2;
        while (son <= right) {
            if (son + 1 <= right && nums[son + 1] > nums[son]) {
                son++;
            }
            if (nums[son] > nums[left]) {
                swap(nums, left, son);
                left = son;
                son = son * 2;
            } else {
                break;
            }
        }
    }
    public void heapSort(int[] nums) {
        int len = nums.length - 1;
        for (int i = len / 2; i >= 0; i--) {
            maxHeapify(nums, i, len);
        }
        for (int i = len; i > 0; i--) {
            swap(nums, 0, i);
            maxHeapify(nums, 0, --len);
        }
    }
}
```

# 1566.重复至少K次且长度为M的模式

```java
class Solution {
    public boolean containsPattern(int[] arr, int m, int k) {
        for (int i = 0; i < arr.length - m * k + 1; i++) {
            boolean isT = true;
            for (int j = 0; j < m * k; j++) {
                if (arr[i + j] != arr[i + j % m]) {
                    isT = false;
                    break;
                }
            }
            if (isT) {
                return true;
            }
        }
        return false;
    }
}
```

# 1572.矩阵对角线元素的和

```java
class Solution {
    public int diagonalSum(int[][] mat) {
        int sum = 0;
        for (int i = 0; i < mat.length; i++) {
            if (i == mat.length - 1 - i) {
                sum +=mat[i][i];
                continue;
            }
            sum += mat[i][i] + mat[i][mat.length - 1 - i];
        }
        return sum;
    }
}
```

# 1582.二进制矩阵中的特殊位置

```java
class Solution {
    public int numSpecial(int[][] mat) {
        int sum = 0;
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[0].length; j++) {
                sum += check(mat, i, j);
            }
        }
        return sum;
    }

    public int check(int[][] mat, int i, int j) {
        if (mat[i][j] == 0) {
            return 0;
        }
        for (int k = 0; k < mat.length; k++) {
            if (mat[k][j] == 1 && k != i) {
                return 0;
            }
        }
        for (int k = 0; k < mat[0].length; k++) {
            if (mat[i][k] == 1 && k != j) {
                return 0;
            }
        }
        return 1;
    }
}
```

# 1588.所有奇数长度子数组的和

```java
class Solution {
    int[] prefix;
    public int sumOddLengthSubarrays(int[] arr) {
        int n = arr.length;
        prefix = new int[n];
        prefix[0] = arr[0];
        for (int i = 1; i < n; i++) {
            prefix[i] = prefix[i - 1] + arr[i];
        }
        int sum = 0;
        for (int len = 1; len <= n; len+=2) {
            for (int left = 0; left < n; left++) {
                int right = left + len - 1;
                if (right >= n) {
                    break;
                }
                sum += getPrefixSum(right) - getPrefixSum(left - 1);
            }
        }
        return sum;
    }
    public int getPrefixSum(int index) {
        if (index == -1) {
            return 0;
        }
        return prefix[index];
    }
}
```

# 1598.文件夹操作日志搜集器

```java
class Solution {
    public int minOperations(String[] logs) {
        Stack<String> resultStack = new Stack<>();
        for (int i = 0; i < logs.length; i++) {
            if ("./".equals(logs[i])) {
                continue;
            } else if ("../".equals(logs[i])) {
                if (resultStack.size() == 0) {
                    continue;
                }
                resultStack.pop();
            } else {
                resultStack.push(logs[i]);
            }
        }
        return resultStack.size();
    }
}
```
