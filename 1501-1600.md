# 1502.判断能否形成等差数列

```java
class Solution {
    public boolean canMakeArithmeticProgression(int[] arr) {
        int len = arr.length;
        for (int i = 0; i < len; i++) {
            int minIndex = i;
            for (int j = i; j < len; j++) {
                if (arr[j] < arr[minIndex]) {
                    minIndex = j;
                }
            }
            int temp = arr[minIndex];
            arr[minIndex] = arr[i];
            arr[i] = temp;
        }
        int sex = arr[1] - arr[0];
        for (int i = 0; i < len - 1; i++) {
            if (arr[i + 1] - arr[i] != sex) {
                return false;
            }
        }
        return true;
    }
}
```

```java
class Solution {
    public String reformatDate(String date) {
        String[] dateString = date.split(" ");
        String day = dateString[0];
        String month = dateString[1];
        String year = dateString[2];
        StringBuilder sb = new StringBuilder();
        sb.append(year).append('-');
        String[] monthString = new String[]{"Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"};
        for (int i = 0; i < monthString.length; i++) {
            if (monthString[i].equals(month)) {
                if (i + 1 < 10) {
                    sb.append(0);
                }
                sb.append(i + 1).append('-');
                break;
            }
        }
        if (!Character.isLetter(day.charAt(1))) {
            sb.append(day.charAt(0)).append(day.charAt(1));
        } else {
            sb.append(0).append(day.charAt(0));
        }
        return sb.toString();
    }
}
```

# 1512.好数对的数目

```java
class Solution {
    public int numIdenticalPairs(int[] nums) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int num : nums) {
            map.put(num, map.getOrDefault(num, 0) + 1);
        }
        int result = 0;
        for (Integer key : map.keySet()) {
            int val = map.get(key);
            result += val * (val - 1) / 2;
        }
        return result;
    }
}
```

```java
class Solution {
    public String thousandSeparator(int n) {
        if (n == 0) {
            return "0";
        }
        StringBuilder sb = new StringBuilder();
        int cnt = 0;
        while (n > 0) {
            sb.append(n % 10);
            n /= 10;
            cnt++;
            if (cnt % 3 == 0 && n != 0) {
                sb.append('.');
            }
        }
        return sb.reverse().toString();
    }
}
```

# 1560.圆形赛道上经过次数最多的扇区

```java
class Solution {
    public List<Integer> mostVisited(int n, int[] rounds) {
        int[] part = new int[n + 1];
        int max = 0;
        for (int i = 0; i < rounds.length - 1; i++) {
            int left = rounds[i];
            int right = rounds[i + 1];
            if (left > right) {
                for (int j = left; j <= n; j++) {
                    part[j]++;
                    max = Math.max(part[j], max);
                }
                for (int j = 1; j < right; j++) {
                    part[j]++;
                    max = Math.max(part[j], max);
                }
            } else {
                for (int j = left; j < right; j++) {
                    part[j]++;
                    max = Math.max(part[j], max);
                }
            }
        }
        part[rounds[rounds.length - 1]]++;
        max = Math.max(part[rounds[rounds.length - 1]], max);
        List<Integer> result = new ArrayList<>();
        for (int i = 0; i < part.length; i++) {
            if (part[i] == max) {
                result.add(i);
            }
        }
        return result;
    }
}
```

# 1561.你可以获得的最大硬币数目

```java
class Solution {
    public int maxCoins(int[] piles) {
        heapSort(piles);
        int len = piles.length - 2;
        int sum = 0;
        int count = len / 3;
        while (count >= 0) {
            sum += piles[len];
            len -= 2;
            count--;
        }
        return sum;
    }
    public void swap(int[] nums, int left, int right) {
        int temp = nums[left];
        nums[left] = nums[right];
        nums[right] = temp;
    }
    public void maxHeapify(int[] nums, int left, int right) {
        int son = left * 2;
        while (son <= right) {
            if (son + 1 <= right && nums[son + 1] > nums[son]) {
                son++;
            }
            if (nums[son] > nums[left]) {
                swap(nums, left, son);
                left = son;
                son = son * 2;
            } else {
                break;
            }
        }
    }
    public void heapSort(int[] nums) {
        int len = nums.length - 1;
        for (int i = len / 2; i >= 0; i--) {
            maxHeapify(nums, i, len);
        }
        for (int i = len; i > 0; i--) {
            swap(nums, 0, i);
            maxHeapify(nums, 0, --len);
        }
    }
}
```

```java
class Solution {
    public int numSpecial(int[][] mat) {
        int sum = 0;
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[0].length; j++) {
                sum += check(mat, i, j);
            }
        }
        return sum;
    }

    public int check(int[][] mat, int i, int j) {
        if (mat[i][j] == 0) {
            return 0;
        }
        for (int k = 0; k < mat.length; k++) {
            if (mat[k][j] == 1 && k != i) {
                return 0;
            }
        }
        for (int k = 0; k < mat[0].length; k++) {
            if (mat[i][k] == 1 && k != j) {
                return 0;
            }
        }
        return 1;
    }
}
```

# 1588.所有奇数长度子数组的和

```java
class Solution {
    int[] prefix;
    public int sumOddLengthSubarrays(int[] arr) {
        int n = arr.length;
        prefix = new int[n];
        prefix[0] = arr[0];
        for (int i = 1; i < n; i++) {
            prefix[i] = prefix[i - 1] + arr[i];
        }
        int sum = 0;
        for (int len = 1; len <= n; len+=2) {
            for (int left = 0; left < n; left++) {
                int right = left + len - 1;
                if (right >= n) {
                    break;
                }
                sum += getPrefixSum(right) - getPrefixSum(left - 1);
            }
        }
        return sum;
    }
    public int getPrefixSum(int index) {
        if (index == -1) {
            return 0;
        }
        return prefix[index];
    }
}
```

# 1592.重新排列单词间的空格

```java
class Solution {
    public String reorderSpaces(String text) {
        String[] words = text.trim().split("\\s+");
        int spaceCnt = 0;
        for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) == ' ') {
                spaceCnt++;
            }
        }
        int needSpace = words.length == 1 ? 0 : spaceCnt / (words.length - 1);
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < words.length; i++) {
            sb.append(words[i]);
            if (i == words.length - 1) {
                break;
            }
            for (int j = 0; j < needSpace; j++) {
                sb.append(' ');
                spaceCnt--;
            }
        }
        while (spaceCnt > 0) {
            sb.append(' ');
            spaceCnt--;
        }
        return sb.toString();
    }
}
```

# 1598.文件夹操作日志搜集器

```java
class Solution {
    public int minOperations(String[] logs) {
        Stack<String> resultStack = new Stack<>();
        for (int i = 0; i < logs.length; i++) {
            if ("./".equals(logs[i])) {
                continue;
            } else if ("../".equals(logs[i])) {
                if (resultStack.size() == 0) {
                    continue;
                }
                resultStack.pop();
            } else {
                resultStack.push(logs[i]);
            }
        }
        return resultStack.size();
    }
}
```
