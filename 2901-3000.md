# 2903.找出满足差值条件的下标Ⅰ

```java
class Solution {
    public int[] findIndices(int[] nums, int indexDifference, int valueDifference) {
        int[] result = new int[]{-1, -1};
        for (int i = 0; i < nums.length; i++) {
            for (int j = 0; j < nums.length; j++) {
                if (Math.abs(i - j) >= indexDifference && Math.abs(nums[i] - nums[j]) >= valueDifference) {
                    result[0] = i;
                    result[1] = j;
                    return result;
                }
            }
        }
        return result;
    }
}
```

# 2908.元素和最小的山形三元组Ⅰ

```java
class Solution {
    public int minimumSum(int[] nums) {
        int result = 9999;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                for (int k = j + 1; k < nums.length; k++) {
                    if (nums[j] > nums[i] && nums[j] > nums[k]) {
                        result = Math.min(result, nums[i] + nums[j] + nums[k]);
                    }
                }
            }
        }
        return result == 9999 ? -1 : result;
    }
}
```

```java
class Solution {
    public int findKOr(int[] nums, int k) {
        int[] result = new int[32];
        for (int i = 0; i < nums.length; i++) {
            int number = nums[i];
            for (int j = 31; j >= 0; j--) {
                if (((number >> j) & 1) == 1) {
                    result[31 - j]++;
                }
            }
        }
        int answer = 0;
        for (int i = 0; i < result.length; i++) {
            answer *= 2;
            answer += result[i] >= k ? 1 : 0;
        }
        return answer;
    }
}
```

# 2937.使三个字符串相等

```java
class Solution {
    public int findMinimumOperations(String s1, String s2, String s3) {
        int index = 0;
        while (index < Math.min(s1.length(), Math.min(s2.length(), s3.length()))) {
            if (s1.charAt(index) == s2.charAt(index) && s1.charAt(index) == s3.charAt(index)) {
                index++;
            } else {
                break;
            }
        }
        return index == 0 ? -1 : s1.length() + s2.length() + s3.length() - 3 * index;
    }
}
```

# 2946.循环移位后的矩阵相似检查

```java
class Solution {
    public boolean areSimilar(int[][] mat, int k) {
        k = k % mat[0].length;
        int[][] lastMat = new int[mat.length][mat[0].length];
        for (int i = 0; i < mat.length; i++) {
            if (i % 2 == 0) {
                for (int j = 0; j < mat[0].length; j++) {
                    lastMat[i][j] = j + k >= mat[0].length ? mat[i][j + k - mat[0].length] : mat[i][j + k];
                }
            } else {
                for (int j = 0; j < mat[0].length; j++) {
                    lastMat[i][j] = j - k < 0 ? mat[i][mat[0].length + j - k] : mat[i][j - k];
                }
            }
        }
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[0].length; j++) {
                if (mat[i][j] != lastMat[i][j]) {
                    return false;
                }
            }
        }
        return true;
    }
}
```

# 2970.统计移除递增子数组的数目Ⅰ

```java
class Solution {
    public int incremovableSubarrayCount(int[] nums) {
        int result = 0;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i; j < nums.length; j++) {
                if (getIsUpArray(nums, i, j)) {
                    result++;
                }
            }
        }
        return result;
    }

    public boolean getIsUpArray(int[] nums, int left, int right) {
        int pre = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i >= left && i <= right) {
                continue;
            }
            if (nums[i] <= pre) {
                return false;
            }
            pre = nums[i];
        }
        return true;
    }
}
```

# 2996.大于等于顺序前缀和的最小缺失整数

```java
class Solution {
    public int missingInteger(int[] nums) {
        Set<Integer> tempSet = new HashSet<>();
        int stratIndex = 0;
        int endIndex = 0;
        int max = nums[0];
        tempSet.add(nums[0]);
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] - nums[i - 1] == 1) {
                endIndex++;
                if (i == nums.length - 1) {
                    int sum = (nums[stratIndex] + nums[endIndex]) * (endIndex - stratIndex + 1) / 2;
                    max = sum;
                    break;
                }
            } else {
                int sum = (nums[stratIndex] + nums[endIndex]) * (endIndex - stratIndex + 1) / 2;
                max = sum;
                break;
            }
        }
        for (int i = 0; i < nums.length; i++) {
            tempSet.add(nums[i]);
        }
        for (int i = max;; i++) {
            if (!tempSet.contains(i)) {
                return i;
            }
        }
    }
}
```

# 3000.对角线最长的矩形的面积

```java
class Solution {
    public int areaOfMaxDiagonal(int[][] dimensions) {
        double len = 0;
        int area = 0;
        for (int i = 0; i < dimensions.length; i++) {
            int x = dimensions[i][0];
            int y = dimensions[i][1];
            double tempLen = Math.sqrt((double) x * x + y * y);
            if (tempLen > len) {
                len = tempLen;
                area = x * y;
            } else if (tempLen == len) {
                area = Math.max(area, x * y);
            }
        }
        return area;
    }
}
```
