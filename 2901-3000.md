# 2928.给小朋友们分糖果Ⅰ

```java
class Solution {
    public int distributeCandies(int n, int limit) {
        int result = 0;
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= n - i; j++) {
                int k = n - i - j;
                if (i <= limit && j <= limit && k <= limit) {
                    result++;
                }
            }
        }
        return result;
    }
}
```

# 2951.找出峰值

```java
class Solution {
    public List<Integer> findPeaks(int[] mountain) {
        List<Integer> resultList = new ArrayList<>();
        for (int i = 1; i < mountain.length - 1; i++) {
            if (mountain[i] > mountain[i - 1] && mountain[i] > mountain[i + 1]) {
                resultList.add(i);
            }
        }
        return resultList;
    }
}
```

# 2970.统计移除递增子数组的数目Ⅰ

```java
class Solution {
    public int incremovableSubarrayCount(int[] nums) {
        int result = 0;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i; j < nums.length; j++) {
                if (getIsUpArray(nums, i, j)) {
                    result++;
                }
            }
        }
        return result;
    }

    public boolean getIsUpArray(int[] nums, int left, int right) {
        int pre = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i >= left && i <= right) {
                continue;
            }
            if (nums[i] <= pre) {
                return false;
            }
            pre = nums[i];
        }
        return true;
    }
}
```



# 2974.最小数字游戏

```java
class Solution {
    public int[] numberGame(int[] nums) {
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i += 2) {
            int temp = nums[i];
            nums[i] = nums[i + 1];
            nums[i + 1] = temp;
        }
        return nums;
    }
}
```

# 2996.大于等于顺序前缀和的最小缺失整数

```java
class Solution {
    public int missingInteger(int[] nums) {
        Set<Integer> tempSet = new HashSet<>();
        int stratIndex = 0;
        int endIndex = 0;
        int max = nums[0];
        tempSet.add(nums[0]);
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] - nums[i - 1] == 1) {
                endIndex++;
                if (i == nums.length - 1) {
                    int sum = (nums[stratIndex] + nums[endIndex]) * (endIndex - stratIndex + 1) / 2;
                    max = sum;
                    break;
                }
            } else {
                int sum = (nums[stratIndex] + nums[endIndex]) * (endIndex - stratIndex + 1) / 2;
                max = sum;
                break;
            }
        }
        for (int i = 0; i < nums.length; i++) {
            tempSet.add(nums[i]);
        }
        for (int i = max;; i++) {
            if (!tempSet.contains(i)) {
                return i;
            }
        }
    }
}
```
