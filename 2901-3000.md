# 2974.最小数字游戏

```java
class Solution {
    public int[] numberGame(int[] nums) {
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i += 2) {
            int temp = nums[i];
            nums[i] = nums[i + 1];
            nums[i + 1] = temp;
        }
        return nums;
    }
}
```

# 2996.大于等于顺序前缀和的最小缺失整数

```java
class Solution {
    public int missingInteger(int[] nums) {
        Set<Integer> tempSet = new HashSet<>();
        int stratIndex = 0;
        int endIndex = 0;
        int max = nums[0];
        tempSet.add(nums[0]);
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] - nums[i - 1] == 1) {
                endIndex++;
                if (i == nums.length - 1) {
                    int sum = (nums[stratIndex] + nums[endIndex]) * (endIndex - stratIndex + 1) / 2;
                    max = sum;
                    break;
                }
            } else {
                int sum = (nums[stratIndex] + nums[endIndex]) * (endIndex - stratIndex + 1) / 2;
                max = sum;
                break;
            }
        }
        for (int i = 0; i < nums.length; i++) {
            tempSet.add(nums[i]);
        }
        for (int i = max;; i++) {
            if (!tempSet.contains(i)) {
                return i;
            }
        }
    }
}
```
