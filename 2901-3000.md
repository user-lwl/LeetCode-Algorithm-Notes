# 2917.找出数组中的K-or值

```java
class Solution {
    public int findKOr(int[] nums, int k) {
        int[] result = new int[32];
        for (int i = 0; i < nums.length; i++) {
            int number = nums[i];
            for (int j = 31; j >= 0; j--) {
                if (((number >> j) & 1) == 1) {
                    result[31 - j]++;
                }
            }
        }
        int answer = 0;
        for (int i = 0; i < result.length; i++) {
            answer *= 2;
            answer += result[i] >= k ? 1 : 0;
        }
        return answer;
    }
}
```

```java
class Solution {
    public int findChampion(int[][] grid) {
        for (int i = 0; i < grid.length; i++) {
            int cnt = 0;
            for (int j = 0; j < grid[0].length; j++) {
                cnt += grid[i][j];
            }

            }
        }
```

# 2928.给小朋友们分糖果Ⅰ

```java
class Solution {
    public int distributeCandies(int n, int limit) {
        int result = 0;
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= n - i; j++) {
                int k = n - i - j;
                if (i <= limit && j <= limit && k <= limit) {
                    result++;
                }
            }
        }
        return result;
    }
}
```

# 2932.找出强数对的最大异或值Ⅰ

```java
class Solution {
    public int maximumStrongPairXor(int[] nums) {
        int result = 0;
        for (int i = 0; i < nums.length; i++) {
            for (int j = 0; j < nums.length; j++) {
                if (Math.abs(nums[i] - nums[j]) <= Math.min(nums[i], nums[j])) {
                    result = Math.max(result, nums[i] ^ nums[j]);
                }
            }
        }
        return result;
    }
}
```



# 2946.循环移位后的矩阵相似检查

```java
class Solution {
    public boolean areSimilar(int[][] mat, int k) {
        k = k % mat[0].length;
        int[][] lastMat = new int[mat.length][mat[0].length];
        for (int i = 0; i < mat.length; i++) {
            if (i % 2 == 0) {
                for (int j = 0; j < mat[0].length; j++) {
                    lastMat[i][j] = j + k >= mat[0].length ? mat[i][j + k - mat[0].length] : mat[i][j + k];
                }
            } else {
                for (int j = 0; j < mat[0].length; j++) {
                    lastMat[i][j] = j - k < 0 ? mat[i][mat[0].length + j - k] : mat[i][j - k];
                }
            }
        }
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[0].length; j++) {
                if (mat[i][j] != lastMat[i][j]) {
                    return false;
                }
            }
        }
        return true;
    }
}
```

# 2965.找出缺失和重复的数字

```java
class Solution {
    public int[] findMissingAndRepeatedValues(int[][] grid) {
        int[] result = new int[grid.length * grid.length + 1];
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                result[grid[i][j]]++;
            }
        }
        int two = -1;
        int no = -1;
        for (int i = 1; i < result.length; i++) {
            if (result[i] == 2) {
                two = i;
            } else if (result[i] == 0) {
                no = i;
            }
        }
        return new int[]{two, no};
    }
}
```

# 2970.统计移除递增子数组的数目Ⅰ

```java
class Solution {
    public int incremovableSubarrayCount(int[] nums) {
        int result = 0;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i; j < nums.length; j++) {
                if (getIsUpArray(nums, i, j)) {
                    result++;
                }
            }
        }
        return result;
    }

    public boolean getIsUpArray(int[] nums, int left, int right) {
        int pre = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i >= left && i <= right) {
                continue;
            }
            if (nums[i] <= pre) {
                return false;
            }
            pre = nums[i];
        }
        return true;
    }
}
```

```java
    public boolean hasTrailingZeros(int[] nums) {
        int cnt = 0;

            if (nums[i] % 2 == 0) {
```

# 2996.大于等于顺序前缀和的最小缺失整数

```java
class Solution {
    public int missingInteger(int[] nums) {
        Set<Integer> tempSet = new HashSet<>();
        int stratIndex = 0;
        int endIndex = 0;
        int max = nums[0];
        tempSet.add(nums[0]);
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] - nums[i - 1] == 1) {
                endIndex++;
                if (i == nums.length - 1) {
                    int sum = (nums[stratIndex] + nums[endIndex]) * (endIndex - stratIndex + 1) / 2;
                    max = sum;
                    break;
                }
            } else {
                int sum = (nums[stratIndex] + nums[endIndex]) * (endIndex - stratIndex + 1) / 2;
                max = sum;
                break;
            }
        }
        for (int i = 0; i < nums.length; i++) {
            tempSet.add(nums[i]);
        }
        for (int i = max;; i++) {
            if (!tempSet.contains(i)) {
                return i;
            }
        }
    }
}
```
