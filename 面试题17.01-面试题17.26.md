# 面试题 17.01.不用加号的加法

```java
class Solution {
    public int add(int a, int b) {
        return a+b;
    }
}
```

```java
class Solution {
    public int add(int a, int b) {
        while (b != 0) {
            int temp = (a & b) << 1;
            a = a ^ b;
            b = temp;
        }
        return a;
    }
}
```

# 面试题 17.04.消失的数字

```java
class Solution {
    public int missingNumber(int[] nums) {
        int n = nums.length;
        int sum = n * (n + 1) / 2;
        for (int i = 0; i < nums.length; i++) {
            sum -= nums[i];
        }
        return sum;
    }
}
```

# 面试题 17.10.主要元素

```java
class Solution {
    public int majorityElement(int[] nums) {
        Arrays.sort(nums);
        int number = nums[nums.length / 2];
        int cnt = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == number) {
                cnt++;
            }
            if (cnt != 0 && nums[i] != number) {
                break;
            }
        }
        return cnt > nums.length / 2 ? number : -1;
    }
}
```

# 面试题 17.12.BiNode

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    TreeNode minNode = null;

    public TreeNode convertBiNode(TreeNode root) {
        helper(root);
        return minNode;
    }
    public void helper(TreeNode node) {
        if (node == null) {
            return;
        }
        helper(node.right);
        node.right = minNode;
        minNode = node;
        helper(node.left);
        node.left = null;
    }
}
```



# 面试题 17.16.按摩师

```java
class Solution {
    public int massage(int[] nums) {
        int len = nums.length;
        int[] result = new int[len];
        if (len == 0) {
            return 0;
        } else if (len == 1) {
            return nums[0];
        }
        result[0] = nums[0];
        result[1] = Math.max(nums[1], nums[0]);
        for (int i = 2; i < len; i++) {
            result[i] = Math.max(result[i - 2] + nums[i], result[i - 1]);
        }
        return result[len - 1];
    }
}
```
