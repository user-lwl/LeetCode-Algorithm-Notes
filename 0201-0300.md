```java
            if (head.val == val) {
                head = head.next;
            } else {
                break;
            }

            result = head;
        }

            if (head.next.val == val) {
                head.next = head.next.next;
            } else {
                head = head.next;
            }
        }
        return result;
    }
}
```

```java
class Solution {
    public int[] findOrder(int numCourses, int[][] prerequisites) {
        int len = numCourses;
        int[][] graph = new int[len][len];
        int[] in = new int[len];
        for (int[] prerequisit: prerequisites) {
            int b = prerequisit[0];
            int a = prerequisit[1];
            graph[a][b] = 1;
            in[b]++;
        }
        Deque<Integer> queue = new LinkedList<>();
        for (int i = 0; i < len; i++) {
            if (in[i] == 0) {
                queue.addLast(i);
            }
        }
        int index = 0;
        int[] result = new int[len];
        Arrays.fill(result, -1);
        while (!queue.isEmpty()) {
            int t = queue.poll();
            result[index] = t;
            index++;
            for (int u = 0; u < len; u++) {
                if (graph[t][u] == 1) {
                    in[u]--;
                    if (in[u] == 0) {
                        queue.addLast(u);
                    }
                }
            }
        }
        if (result[len - 1] == -1) {
            return new int[0];
        }
        return result;
    }
}
```

```java
class Solution {
    public int rob(int[] nums) {
        if (nums.length == 1) {
            return nums[0];
        }
        // dp1:偷第一间，不偷最后一间
        int[] dp1 = new int[nums.length];
        // dp2:不偷第一间，可能偷最后一间
        int[] dp2 = new int[nums.length];
        dp1[0] = nums[0];
        dp1[1] = nums[0];
        dp2[0] = 0;
        dp2[1] = nums[1];
        for (int i = 2; i < nums.length - 1; i++) {
            dp1[i] = Math.max(dp1[i - 1], dp1[i - 2] + nums[i]);
        }
        for (int i = 2; i < nums.length; i++) {
            dp2[i] = Math.max(dp2[i - 1], dp2[i - 2] + nums[i]);
        }
        return Math.max(dp1[dp1.length - 2], dp2[dp2.length - 1]);
    }
}
```

```java
class Solution {
    public boolean containsDuplicate(int[] nums) {
        int len = nums.length;
        quickSort(nums, 0, len - 1);
        for (int i = 1; i < len; i++) {
            if (nums[i] == nums[i - 1]) {
                return true;
            }
        }
        return false;
    }
    public void swap(int[] nums, int left, int right) {
        int temp = nums[left];
        nums[left] = nums[right];
        nums[right] = temp;
    }
    public int partition(int[] nums, int left, int right) {
        int mid = new Random().nextInt(right - left + 1) + left;
        swap(nums, left, mid);
        int j = left + 1;
        for (int i = left + 1; i <= right; i++) {
            if (nums[i] < nums[left]) {
                swap(nums, i, j++);
            }
        }
        swap(nums, j - 1, left);
        return j - 1;
    }
    public void quickSort(int[] nums, int left, int right) {
        if (left < right) {
            int mid = partition(nums, left, right);
            quickSort(nums, left, mid - 1);
            quickSort(nums, mid + 1, right);
        }
    }
}
```

# 221.最大正方形

```java
class Solution {
    public int maximalSquare(char[][] matrix) {
        int[][] dp = new int[matrix.length][matrix[0].length];
        int max = 0;
        for (int i = 0; i < matrix.length; i++) {
            dp[i][0] = matrix[i][0] - '0';
            max = Math.max(dp[i][0], max);
        }
        for (int i = 0; i < matrix[0].length; i++) {
            dp[0][i] = matrix[0][i] - '0';
            max = Math.max(dp[0][i], max);
        }
        for (int i = 1; i < matrix.length; i++) {
            for (int j = 1; j < matrix[0].length; j++) {
                if (matrix[i][j] == '0') {
                    dp[i][j] = 0;
                    continue;
                }
                dp[i][j] = Math.min(dp[i - 1][j], Math.min(dp[i][j - 1], dp[i - 1][j - 1])) + 1;
                max = Math.max(dp[i][j], max);
            }
        }
        return max * max;
    }
}
```

```java
class Solution {
    public boolean isPalindrome(ListNode head) {
        Deque<Integer> queue = new LinkedList<>();
        while (head != null) {
            queue.offer(head.val);
            head = head.next;
        }
        while (queue.size() > 0) {
            int first = queue.pollFirst();
            if (queue.size() == 0) {
                return true;
            }
            int last = queue.pollLast();
            if (first != last) {
                return false;
            }
        }
        return true;
    }
```

```java
class Solution {
    public int[] maxSlidingWindow(int[] nums, int k) {
        int len = nums.length;
        int[] result = new int[len - k + 1];
        Deque<Integer> queue = new LinkedList<>();
        for (int i = 0; i < len; i++) {
            while (!queue.isEmpty() && nums[queue.peekLast()] <= nums[i]) {
                queue.pollLast();
            }
            queue.addLast(i);
            while (!queue.isEmpty() && i - k >= queue.peekFirst()) {
                queue.pollFirst();
            }
            if (i >= k - 1) {
                result[i - k + 1] = nums[queue.peekFirst()];
            }
        }
        return result;
    }
```

# 287.寻找重复数

```java
class Solution {
    public int findDuplicate(int[] nums) {
        /**
         * 求带环链表的环的入口节点
         * 快指针走两步慢指针走一步
         * 快慢指针相遇时，慢指针回到原点
         * 快慢指针均走一步，相遇时即为环的入口
         */
        int slow = 0;
        int fast = 0;
        do {
            slow = nums[slow];
            fast = nums[nums[fast]];
        } while (slow != fast);
        slow = 0;
        while (slow != fast) {
            slow = nums[slow];
            fast = nums[fast];
        }
        return fast;
    }
}
```

```java
class Solution {
    public int findDuplicate(int[] nums) {
        int len = nums.length;
        quickSort(nums, 0, len - 1);
        for (int i = 1; i < len; i++) {
            if (nums[i] == nums[i - 1]) {
                return nums[i];
            }
        }
        return -1;
    }
    public void swap(int[] nums, int left, int right) {
        int temp = nums[left];
        nums[left] = nums[right];
        nums[right] = temp;
    }
    public int partition(int[] nums, int left, int right) {
        int mid = new Random().nextInt(right - left + 1) + left;
        swap(nums, left, mid);
        int j = left + 1;
        for (int i = left + 1; i <= right; i++) {
            if (nums[i] < nums[left]) {
                swap(nums, i, j++);
            }
        }
        swap(nums, j - 1, left);
        return j - 1;
    }
    public void quickSort(int[] nums, int left, int right) {
        if (left < right) {
            int mid = partition(nums, left, right);
            quickSort(nums, left, mid - 1);
            quickSort(nums, mid + 1, right);
        }
    }
}
```

```java
}
```

# 299.猜数字游戏

```java
class Solution {
    public String getHint(String secret, String guess) {
        char[] secretList = secret.toCharArray();
        char[] guessList = guess.toCharArray();
        int [] ss = new int[10];
        int [] gg = new int[10];
        int bulls = 0;
        int cows = 0;
        for (int i = 0; i < secretList.length; i++) {
            if (secretList[i] == guessList[i]) {
                bulls++;
            } else {
                ss[secretList[i] - '0']++;
                gg[guessList[i] - '0']++;
            }
        }
        for (int i = 0; i < 10; i++) {
            cows += Math.min(ss[i], gg[i]);
        }
        return bulls + "A" + cows + "B";
    }
}
```
