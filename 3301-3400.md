# 3304.找出第K个字符Ⅰ

```java
class Solution {
    public char kthCharacter(int k) {
        char[] word = new char[]{'a'};
        char[] newWord = getKLenWord(word, k);
        return newWord[k - 1];
    }
    public char[] getKLenWord(char[] word, int k) {
        if (word.length >= k) {
            return word;
        }
        char[] newWord = new char[word.length * 2];
        int len = word.length;
        for (int i = 0; i < word.length; i++) {
            newWord[i] = word[i];
            newWord[i + len] = (char) (word[i] + 1);
        }
        return getKLenWord(newWord, k);
    }
}
```

# 3314.构造最小位运算数组Ⅰ

```java
class Solution {
    public int[] minBitwiseArray(List<Integer> nums) {
        int[] result = new int[nums.size()];
        for (int i = 0; i < result.length; i++) {
            result[i] = -1;
        }
        for (int i = 0; i < nums.size(); i++) {
            int number = nums.get(i);
            for (int j = 0; j < number; j++) {
                if ((j | (j + 1)) == number) {
                    result[i] = j;
                    break;
                }
            }
        }
        return result;
    }
}
```

```java
class Solution {
```

```java
        int jiSum = 0;
        int ouSum = 0;
        for (int i = 0; i < num.length(); i++) {
            int tempNumber = num.charAt(i) - '0';
            if (i % 2 == 0) {
```

# 3349.检测相邻递增子数组Ⅰ

```java
class Solution {
    public boolean hasIncreasingSubarrays(List<Integer> nums, int k) {
        int firstStartIndex = 0;
        while (firstStartIndex < nums.size() - 2 * k + 1) {
            boolean isUp = true;
            for (int i = firstStartIndex; i < firstStartIndex + k - 1; i++) {
                if (nums.get(i + 1) <= nums.get(i)) {
                    isUp = false;
                }
                if (nums.get(i + k + 1) <= nums.get(i + k)) {
                    isUp = false;
                }
            }
            if (isUp) {
                return true;
            }
            firstStartIndex++;
        }
        return false;
    }
}
```

```java
        int left = 0;

                    result++;
                }

    }
```

```java
    public int minimumSumSubarray(List<Integer> nums, int l, int r) {
        int result = 999999;
        for (int i = 0; i < nums.size(); i++) {
            int sum = 0;
            for (int j = i; j  <nums.size(); j++) {
                sum += nums.get(j);
                int len = j - i + 1;
```

```java
class Solution {
    public boolean isPossibleToRearrange(String s, String t, int k) {
        int len = s.length() / k;
        Map<String, Integer> sMap = new HashMap<>();
        Map<String, Integer> tMap = new HashMap<>();
        for (int i = 0; i < s.length(); i += len) {
            String tempStr1 = s.substring(i, i + len);
            String tempStr2 = t.substring(i, i + len);
            sMap.put(tempStr1, sMap.getOrDefault(tempStr1, 0) + 1);
            tMap.put(tempStr2, tMap.getOrDefault(tempStr2, 0) + 1);
        }
        for (String key : sMap.keySet()) {
            int cnt1 = sMap.get(key);
            int cnt2 = tMap.getOrDefault(key, 0);
            if (cnt1 != cnt2) {
                return false;
            }
        }
        return true;
    }
}
```

```java
        Map<Integer, Integer> tempMap = new HashMap<>();
        tempMap.put(events[0][0], events[0][1]);
        for (int i = 1; i < events.length; i++) {
            int tempTime = tempMap.getOrDefault(events[i][0], 0);
            tempMap.put(events[i][0], Math.max(events[i][1] - events[i - 1][1], tempTime));
        }
        int index = -1;
        int maxTime = -1;
        for (Integer key : tempMap.keySet()) {
            int time = tempMap.get(key);
            if (time > maxTime) {
                maxTime = time;

            }

    }
```

```java
        int[][] heng = new int[rectangles.length][2];
        int[][] shu = new int[rectangles.length][2];

        Arrays.sort(num, new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                return Integer.compare(o1[0], o2[0]);
            }
        });

        for (int i = 1; i < num.length; i++) {
            if (num[i][0] < maxR) {
                cnt--;
            }
            maxR = Math.max(maxR, num[i][1]);
        }
```
