# 3304.找出第K个字符Ⅰ

```java
class Solution {
    public char kthCharacter(int k) {
        char[] word = new char[]{'a'};
        char[] newWord = getKLenWord(word, k);
        return newWord[k - 1];
    }
    public char[] getKLenWord(char[] word, int k) {
        if (word.length >= k) {
            return word;
        }
        char[] newWord = new char[word.length * 2];
        int len = word.length;
        for (int i = 0; i < word.length; i++) {
            newWord[i] = word[i];
            newWord[i + len] = (char) (word[i] + 1);
        }
        return getKLenWord(newWord, k);
    }
}
```

# 3314.构造最小位运算数组Ⅰ

```java
class Solution {
    public int[] minBitwiseArray(List<Integer> nums) {
        int[] result = new int[nums.size()];
        for (int i = 0; i < result.length; i++) {
            result[i] = -1;
        }
        for (int i = 0; i < nums.size(); i++) {
            int number = nums.get(i);
            for (int j = 0; j < number; j++) {
                if ((j | (j + 1)) == number) {
                    result[i] = j;
                    break;
                }
            }
        }
        return result;
    }
}
```

# 3330.找到初始输入字符串Ⅰ

```java
class Solution {
    public int possibleStringCount(String word) {
        int result = 1;
        for (int i = 0; i < word.length() - 1; i++) {
            if (word.charAt(i) == word.charAt(i + 1)) {
                char temp = word.charAt(i);
                int index = i;
                while (i < word.length() && temp == word.charAt(i)) {
                    i++;
                }
                result += i - index - 1;
                i--;
            }
        }
        return result;
    }
}
```

# 3340.检查平衡字符串

```java
class Solution {
    public boolean isBalanced(String num) {
        int jiSum = 0;
        int ouSum = 0;
        for (int i = 0; i < num.length(); i++) {
            int tempNumber = num.charAt(i) - '0';
            if (i % 2 == 0) {
                ouSum += tempNumber;
            } else {
                jiSum += tempNumber;
            }
        }
        return jiSum == ouSum;
    }
}
```



# 3345.最小可整除数位乘积Ⅰ

```java
class Solution {
    public int smallestNumber(int n, int t) {
        while (n < 9999) {
            int ji = 1;
            int number = n;
            while (number > 0) {
                ji *= (number % 10);
                number /= 10;
            }
            if (ji % t == 0) {
                return n;
            }
            n++;
        }
        return -1;
    }
}
```

# 3349.检测相邻递增子数组Ⅰ

```java
class Solution {
    public boolean hasIncreasingSubarrays(List<Integer> nums, int k) {
        int firstStartIndex = 0;
        while (firstStartIndex < nums.size() - 2 * k + 1) {
            boolean isUp = true;
            for (int i = firstStartIndex; i < firstStartIndex + k - 1; i++) {
                if (nums.get(i + 1) <= nums.get(i)) {
                    isUp = false;
                }
                if (nums.get(i + k + 1) <= nums.get(i + k)) {
                    isUp = false;
                }
            }
            if (isUp) {
                return true;
            }
            firstStartIndex++;
        }
        return false;
    }
}
```

# 3354.使数组元素等于零

```java
class Solution {
    public int countValidSelections(int[] nums) {
        int left = 0;
        int right = 0;
        int result = 0;
        for (int i = 1; i < nums.length; i++) {
            right += nums[i];
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                if (left == right) {
                    result += 2;
                } else if (Math.abs(left - right) == 1) {
                    result++;
                }
            }
            if (i == nums.length - 1) {
                break;
            }
            left += nums[i];
            right -= nums[i + 1];
        }
        return result;
    }
}
```

# 3364.最小正和子数组

```java
class Solution {
    public int minimumSumSubarray(List<Integer> nums, int l, int r) {
        int result = 999999;
        for (int i = 0; i < nums.size(); i++) {
            int sum = 0;
            for (int j = i; j  <nums.size(); j++) {
                sum += nums.get(j);
                int len = j - i + 1;
                if (len >= l && len <= r && sum > 0) {
                    result = Math.min(result, sum);
                }
            }
        }
        return result == 999999 ? -1 : result;
    }
}
```

# 3375.使数组的值全部为K的最少操作次数

```java
class Solution {
    public int minOperations(int[] nums, int k) {
        Arrays.sort(nums);
        if (nums[0] < k) {
            return -1;
        }
        Set<Integer> resultSet = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            resultSet.add(nums[i]);
        }
        return nums[0] == k ? resultSet.size() - 1 : resultSet.size();
    }
}
```

# 3392.统计符合条件长度为3的子数组数目

```java
class Solution {
    public int countSubarrays(int[] nums) {
        int result = 0;
        for (int i = 0; i < nums.length - 2; i++) {
            if ((nums[i] + nums[i + 2]) * 2 == nums[i + 1]) {
                result++;
            }
        }
        return result;
    }
}
```

# 3396.使数组元素互不相同所需的最少操作次数

```java
class Solution {
    public int minimumOperations(int[] nums) {
        int i = nums.length - 1;
        Set<Integer> tempSet = new HashSet<>();
        for (; i >= 0; i--) {
            if (!tempSet.add(nums[i])) {
                break;
            }
        }
        return i == -1 ? 0 : i / 3 + 1;
    }
}
```
