~~~java
class Solution {
    public int[] twoSum(int[] nums, int target) {
        int len = nums.length;
        int[] temp = new int[len];
        for (int i = 0; i < len; i++) {
            temp[i] = nums[i];
        }
        Arrays.sort(temp);
        int left = 0;
        int right = len - 1;
        while (left < right) {
            if (temp[left] + temp[right] < target) {
                left++;
            } else if (temp[left] + temp[right] > target) {
                right--;
            } else {
                break;
            }
        }
        int[] result = new int[2];
        result[0] = -1;
        result[1] = -1;
        for (int i = 0; i < len; i++) {
            if (nums[i] == temp[left] && result[0] == -1) {
                result[0] = i;
            } else if (nums[i] == temp[right] && result[1] == -1) {
                result[1] = i;
            }
            if (result[0] != -1 && result[1] != -1) {
                return result;
            }
        }
        return result;
    }
}
~~~

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        ListNode head = l1;
        int isNext = 0;
        while (l1 != null && l2 != null) {
            int sum = l1.val + l2.val + isNext;
            l1.val = sum % 10;
            isNext = sum / 10;
            if (l1.next == null || l2.next == null) {
                break;
            }
            l1 = l1.next;
            l2 = l2.next;
        }
        if (l1.next == null && l2.next == null) {
            if (isNext > 0) {
                l1.next = new ListNode(isNext);
            }
            return head;
        }
        if (l1.next == null && l2.next != null) {
            l1.next = l2.next;
        }
        l1 = l1.next;
        while (l1 != null) {
            int sum = l1.val + isNext;
            l1.val = sum % 10;
            isNext = sum / 10;
            if (l1.next == null) {
                break;
            }
            l1 = l1.next;
        }
        if (isNext != 0) {
            l1.next = new ListNode(isNext);
        }
        return head;
    }
}
```

```java
                    startIndex = i;
                    endIndex = j;
```

```java
class Solution {

    char[] stringChar;
    char[] patternChar;

    public boolean isMatch(String s, String p) {
        stringChar = s.toCharArray();
        patternChar = p.toCharArray();
        int slen = stringChar.length;
        int plen = patternChar.length;
        boolean[][] result = new boolean[slen + 1][plen + 1];
        result[0][0] = true;
        for (int i = 0; i <= slen; i++) {
            for (int j = 1; j <= plen; j++) {
                if (patternChar[j - 1] == '*') {
                    if (match(i, j - 1)) {
                        result[i][j] = result[i][j - 2] || result[i - 1][j];
                    } else {
                        result[i][j] = result[i][j - 2];
                    }
                } else {
                    if (match(i, j)) {
                        result[i][j] = result[i - 1][j - 1];
                    }
                }
            }
        }
        return result[slen][plen];
    }

    public boolean match(int i, int j) {
        if (i == 0) {
            return false;
        }
        if (patternChar[j - 1] == '.') {
            return true;
        }
        return stringChar[i - 1] == patternChar[j - 1];
    }
}
```

# 14.最长公共前缀

```java
class Solution {
    public String longestCommonPrefix(String[] strs) {
        Trie node = new Trie();
        for (String str : strs) {
            node.insert(str);
        }
        return node.findLongPrefix();
    }
}

class Trie {

    public Trie[] child;

    public boolean isEnd;

    public Trie() {
        child = new Trie[26];
        isEnd = false;
    }

    public void insert(String word) {
        char[] c = word.toCharArray();
        Trie node = this;
        int len = c.length;
        for (int i = 0; i < len; i++) {
            int index = c[i] - 'a';
            if (node.child[index] == null) {
                node.child[index] = new Trie();
            }
            node = node.child[index];
        }
        node.isEnd = true;
    }

    public String findLongPrefix() {
        Trie node = this;
        StringBuilder result = new StringBuilder();
        while (node != null) {
            if (node.isEnd == true) {
                return result.toString();
            }
            int index = -1;
            for (int i = 0; i < 26; i++) {
                if (node.child[i] != null) {
                    if (index != -1) {
                        return result.toString();
                    }
                    index = i;
                }
            }
            if (index == -1) {
                return result.toString();
            }
            result.append((char) (index + (int)'a'));
            node = node.child[index];
        }
        return result.toString();
    }
}
```



# 15.三数之和

```java
class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        List<List<Integer>> result = new ArrayList<>();
        Arrays.sort(nums);
        int len = nums.length;
        for (int i = 0; i < len - 2; i++) {
            if (nums[i] > 0) {
                return result;
            }
            if (i > 0 && nums[i] == nums[i - 1]) {
                continue;
            }
            int left = i + 1;
            int right = len - 1;
            while (left < right) {
                if (nums[i] + nums[left] + nums[right] > 0) {
                    right--;
                } else if (nums[i] + nums[left] + nums[right] < 0) {
                    left++;
                } else {
                    List<Integer> list = new ArrayList<>();
                    list.add(nums[i]);
                    list.add(nums[left]);
                    list.add(nums[right]);
                    result.add(list);
                    right--;
                    left++;
                    while (left < right && nums[left] == nums[left - 1]) {
                        left++;
                    }
                    while (left < right && nums[right] == nums[right + 1]) {
                        right--;
                    }
                }
            }
        }
        return result;
    }
}
```

```java
        ListNode left = head;
        ListNode right = head;
        for (int i = 0; i < n; i++) {
            right = right.next;
```

```java
 * }
```

```java
class Solution {
    public ListNode reverseKGroup(ListNode head, int k) {

        ListNode node = newHead;

            int number = k;

            ListNode nodeTemp = node.next;
            while (number > 0 && nodeTemp != null) {
                nodeStack.push(nodeTemp);
                nodeTemp = nodeTemp.next;

                ListNode newNode = nodeStack.pop();
                node.next = newNode;
                node = node.next;
            }
            node.next = nodeTemp;
        }
```

```java
    public ListNode rotateRight(ListNode head, int k) {
        int size = 0;
        ListNode node = head;
        while (node != null) {
            size++;
            node = node.next;
        }
        if (k == 0 || head == null || k % size == 0) {

        left.next = null;
        return newHead;
    }
```

```java
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode reverseBetween(ListNode head, int left, int right) {
        if (left == right) {
            return head;
        }
        ListNode newHead = new ListNode(-1, head);
        ListNode leftNode = newHead;
        ListNode rightNode = newHead;
        while (left > 1) {
            leftNode = leftNode.next;
            left--;
        }
        while (right > 0) {
            rightNode = rightNode.next;
            right--;
        }
        ListNode firstTail = leftNode;
        ListNode secondHead = leftNode.next;
        ListNode secondTail = rightNode;
        ListNode ThridHead = rightNode.next;
        secondTail.next = null;
        Stack<ListNode> stack = new Stack<>();
        while (secondHead != null) {
            stack.push(secondHead);
            secondHead = secondHead.next;
        }
        while (!stack.empty()) {
            ListNode newNode = stack.pop();
            firstTail.next = newNode;
            firstTail = firstTail.next;
        }
        firstTail.next = ThridHead;
        return newHead.next;
```
