# 1.两数之和

~~~java
class Solution {
    public int[] twoSum(int[] nums, int target) {
        int len = nums.length;
        int[] temp = new int[len];
        for (int i = 0; i < len; i++) {
            temp[i] = nums[i];
        }
        Arrays.sort(temp);
        int left = 0;
        int right = len - 1;
        while (left < right) {
            if (temp[left] + temp[right] < target) {
                left++;
            } else if (temp[left] + temp[right] > target) {
                right--;
            } else {
                break;
            }
        }
        int[] result = new int[2];
        result[0] = -1;
        result[1] = -1;
        for (int i = 0; i < len; i++) {
            if (nums[i] == temp[left] && result[0] == -1) {
                result[0] = i;
            } else if (nums[i] == temp[right] && result[1] == -1) {
                result[1] = i;
            }
            if (result[0] != -1 && result[1] != -1) {
                return result;
            }
        }
        return result;
    }
}
~~~

# 2.两数相加

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        ListNode head = l1;
        int isNext = 0;
        while (l1 != null && l2 != null) {
            int sum = l1.val + l2.val + isNext;
            l1.val = sum % 10;
            isNext = sum / 10;
            if (l1.next == null || l2.next == null) {
                break;
            }
            l1 = l1.next;
            l2 = l2.next;
        }
        if (l1.next == null && l2.next == null) {
            if (isNext > 0) {
                l1.next = new ListNode(isNext);
            }
            return head;
        }
        if (l1.next == null && l2.next != null) {
            l1.next = l2.next;
        }
        l1 = l1.next;
        while (l1 != null) {
            int sum = l1.val + isNext;
            l1.val = sum % 10;
            isNext = sum / 10;
            if (l1.next == null) {
                break;
            }
            l1 = l1.next;
        }
        if (isNext != 0) {
            l1.next = new ListNode(isNext);
        }
        return head;
    }
}
```

# 14.最长公共前缀

```java
class Solution {
    public String longestCommonPrefix(String[] strs) {
        Trie node = new Trie();
        for (String str : strs) {
            node.insert(str);
        }
        return node.findLongPrefix();
    }
}

class Trie {

    public Trie[] child;

    public boolean isEnd;

    public Trie() {
        child = new Trie[26];
        isEnd = false;
    }

    public void insert(String word) {
        char[] c = word.toCharArray();
        Trie node = this;
        int len = c.length;
        for (int i = 0; i < len; i++) {
            int index = c[i] - 'a';
            if (node.child[index] == null) {
                node.child[index] = new Trie();
            }
            node = node.child[index];
        }
        node.isEnd = true;
    }

    public String findLongPrefix() {
        Trie node = this;
        StringBuilder result = new StringBuilder();
        while (node != null) {
            if (node.isEnd == true) {
                return result.toString();
            }
            int index = -1;
            for (int i = 0; i < 26; i++) {
                if (node.child[i] != null) {
                    if (index != -1) {
                        return result.toString();
                    }
                    index = i;
                }
            }
            if (index == -1) {
                return result.toString();
            }
            result.append((char) (index + (int)'a'));
            node = node.child[index];
        }
        return result.toString();
    }
}
```

```java
    public List<List<Integer>> threeSum(int[] nums) {
        List<List<Integer>> result = new ArrayList<>();
        Arrays.sort(nums);
        int len = nums.length;
        for (int i = 0; i < len - 2; i++) {
            if (nums[i] > 0) {
                return result;
            }
            if (i > 0 && nums[i] == nums[i - 1]) {
                continue;
            }
            int left = i + 1;
            int right = len - 1;
            while (left < right) {
                if (nums[i] + nums[left] + nums[right] > 0) {
                    right--;
                } else if (nums[i] + nums[left] + nums[right] < 0) {
                    left++;
                } else {
                    List<Integer> list = new ArrayList<>();
                    list.add(nums[i]);
                    list.add(nums[left]);
                    list.add(nums[right]);
                    result.add(list);
                    right--;
                    left++;
                    while (left < right && nums[left] == nums[left - 1]) {
                        left++;
                    }
                    while (left < right && nums[right] == nums[right + 1]) {
                        right--;
                    }
                }
            }
        }
        return result;
    }
}
```

```java
        Stack<Character> stack = new Stack<>();
        char[] c = s.toCharArray();
        for (int i = 0; i < c.length; i++) {
            if (c[i] == '(' || c[i] == '[' || c[i] == '{') {
                stack.push(c[i]);
            } else if (c[i] == ')') {
                if (stack.size() == 0 || stack.pop() != '(') {
                    return false;
                }
            } else if (c[i] == ']') {
                if (stack.size() == 0 || stack.pop() != '[') {
                    return false;
                }
            } else if (c[i] == '}') {
                if (stack.size() == 0 || stack.pop() != '{') {
                    return false;
                }
            }
        }
        if (stack.size() != 0) {
            return false;
        }
        return true;
    }
}
```

# 25.K个一组反转链表

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode reverseKGroup(ListNode head, int k) {
        ListNode newHead = new ListNode(-1, head);
        ListNode node = newHead;
        while (node != null) {
            int number = k;
            Stack<ListNode> nodeStack = new Stack<>();
            ListNode nodeTemp = node.next;
            while (number > 0 && nodeTemp != null) {
                nodeStack.push(nodeTemp);
                nodeTemp = nodeTemp.next;
                number--;
            }
            if (number > 0) {
                break;
            }
            while (!nodeStack.empty()) {
                ListNode newNode = nodeStack.pop();
                node.next = newNode;
                node = node.next;
            }
            node.next = nodeTemp;
        }
        return newHead.next;
    }
}
```

# 61.旋转链表

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode rotateRight(ListNode head, int k) {
        int size = 0;
        ListNode node = head;
        while (node != null) {
            size++;
            node = node.next;
        }
        if (k == 0 || head == null || k % size == 0) {
            return head;
        }
        k = k % size;
        ListNode left = head;
        ListNode right = head;
        while (k > 0) {
            right = right.next;
            k--;
        }
        while (right.next != null) {
            right = right.next;
            left = left.next;
        }
        ListNode newHead = left.next;
        right.next = head;
        left.next = null;
        return newHead;
    }
}
```

# 92.反转链表Ⅱ

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode reverseBetween(ListNode head, int left, int right) {
        if (left == right) {
            return head;
        }
        ListNode newHead = new ListNode(-1, head);
        ListNode leftNode = newHead;
        ListNode rightNode = newHead;
        while (left > 1) {
            leftNode = leftNode.next;
            left--;
        }
        while (right > 0) {
            rightNode = rightNode.next;
            right--;
        }
        ListNode firstTail = leftNode;
        ListNode secondHead = leftNode.next;
        ListNode secondTail = rightNode;
        ListNode ThridHead = rightNode.next;
        secondTail.next = null;
        Stack<ListNode> stack = new Stack<>();
        while (secondHead != null) {
            stack.push(secondHead);
            secondHead = secondHead.next;
        }
        while (!stack.empty()) {
            ListNode newNode = stack.pop();
            firstTail.next = newNode;
            firstTail = firstTail.next;
        }
        firstTail.next = ThridHead;
        return newHead.next;
    }
}
```

# 98.验证二叉搜索树

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {

 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isValidBST(TreeNode root) {
        return isTrue(root, Long.MIN_VALUE, Long.MAX_VALUE);
    }
    public boolean isTrue(TreeNode node, long lower, long upper) {
        if (node == null) {
            return true;
        }
        if (node.val >= upper || node.val <= lower) {
            return false;
        }
        return isTrue(node.left, lower, node.val) && isTrue(node.right, node.val, upper);
    }
}
```

# 99.恢复二叉搜索树

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    List<TreeNode> list = new ArrayList<>();
    public void recoverTree(TreeNode root) {
        inorder(root);
        TreeNode x = null;
        TreeNode y = null;
        for (int i = 0; i < list.size() - 1; i++) {
            if (list.get(i).val > list.get(i + 1).val) {
                y = list.get(i + 1);
                if (x == null) {
                    x = list.get(i);
                }
            }
        }
        int temp = x.val;
        x.val = y.val;
        y.val = temp;
    }
    public void inorder(TreeNode node) {
        if (node == null) {
            return;
        }
        inorder(node.left);
        list.add(node);
        inorder(node.right);
    }
}
```
