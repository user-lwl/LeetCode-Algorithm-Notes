# 面试题 16.11.跳水板

```java
class Solution {
    public int[] divingBoard(int shorter, int longer, int k) {
        if (k == 0) {
            return new int[]{};
        }
        Set<Integer> longSet = new HashSet<>();
        List<Integer> longList = new ArrayList<>();
        int len = 0;
        for (int i = 0; i < k; i++) {
            len += shorter;
        }
        longSet.add(len);
        longList.add(len);
        while (k > 0) {
            len += (longer - shorter);
            if (longSet.add(len)) {
                longList.add(len);
            }
            k--;
        }
        int[] result = new int[longList.size()];
        int index = 0;
        for (int i = 0; i < longList.size(); i++) {
            result[index++] = longList.get(i);
        }
        return result;
    }
}
```



# 面试题 16.15.珠玑妙算

```java
class Solution {
    public int[] masterMind(String solution, String guess) {
        int[] solutionList = new int[4];
        int[] guessList = new int[4];
        int equals = 0;
        int same = 0;
        for (int i = 0; i < solution.length(); i++) {
            char solutionChar = solution.charAt(i);
            char guessChar = guess.charAt(i);
            if (solutionChar == guessChar) {
                equals++;
            } else {
                if (solutionChar == 'R') {
                    solutionList[0]++;
                } else if (solutionChar == 'Y') {
                    solutionList[1]++;
                } else if (solutionChar == 'G') {
                    solutionList[2]++;
                } else {
                    solutionList[3]++;
                }
                if (guessChar == 'R') {
                    guessList[0]++;
                } else if (guessChar == 'Y') {
                    guessList[1]++;
                } else if (guessChar == 'G') {
                    guessList[2]++;
                } else {
                    guessList[3]++;
                }
            }
        }
        for (int i = 0; i < solutionList.length; i++) {
            same += Math.min(solutionList[i], guessList[i]);
        }
        return new int[]{equals, same};
    }
}
```



# 面试题 16.17.连续数列

```java
class Solution {
    public int maxSubArray(int[] nums) {
        int len = nums.length;
        int[] result = new int[len];
        int max = nums[0];
        result[0] = nums[0];
        for (int i = 1; i < len; i++) {
            result[i] = Math.max(nums[i], result[i - 1] + nums[i]);
            max = Math.max(result[i], max);
        }
        return max;
    }
}
```



# 面试题 16.25. LRU缓存

```java
class LRUCache {

    class Node {
        public int key;
        public int val;
        public Node prev;
        public Node next;
        Node() {}
        Node(int k, int v) {
            key = k;
            val = v;
        }
    }

    int size;
    int capacity;
    Node head;
    Node tail;
    Map<Integer, Node> map = new HashMap<>();

    public LRUCache(int capacity) {
        size = 0;
        this.capacity = capacity;
        head = new Node();
        tail = new Node();
        head.next = tail;
        tail.prev = head;
    }
    
    public int get(int key) {
        Node node = map.get(key);
        if (node == null) {
            return -1;
        }
        moveToHead(node);
        return node.val;
    }
    
    public void put(int key, int val) {
        Node node = map.get(key);
        if (node == null) {
            node = new Node(key, val);
            addToHead(node);
            map.put(key, node);
            size++;
            if (size > capacity) {
                Node rm = removeTail();
                map.remove(rm.key);
                size--;
            }
        } else {
            node.val = val;
            moveToHead(node);
        }
    }

    public void addToHead(Node node) {
        node.prev = head;
        node.next = head.next;
        head.next.prev = node;
        head.next = node;
    }

    public Node removeTail() {
        Node rmNode = tail.prev;
        removeNode(rmNode);
        return rmNode;
    }

    public void removeNode(Node node) {
        node.prev.next = node.next;
        node.next.prev = node.prev;
    }

    public void moveToHead(Node node) {
        removeNode(node);
        addToHead(node);
    }
}

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache obj = new LRUCache(capacity);
 * int param_1 = obj.get(key);
 * obj.put(key,value);
 */
```
