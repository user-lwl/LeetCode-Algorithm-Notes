# 面试题 16.17.连续数列

```java
class Solution {
    public int maxSubArray(int[] nums) {
        int len = nums.length;
        int[] result = new int[len];
        int max = nums[0];
        result[0] = nums[0];
        for (int i = 1; i < len; i++) {
            result[i] = Math.max(nums[i], result[i - 1] + nums[i]);
            max = Math.max(result[i], max);
        }
        return max;
    }
}
```



# 面试题 16.25. LRU缓存

```java
class LRUCache {

    class Node {
        public int key;
        public int val;
        public Node prev;
        public Node next;
        Node() {}
        Node(int k, int v) {
            key = k;
            val = v;
        }
    }

    int size;
    int capacity;
    Node head;
    Node tail;
    Map<Integer, Node> map = new HashMap<>();

    public LRUCache(int capacity) {
        size = 0;
        this.capacity = capacity;
        head = new Node();
        tail = new Node();
        head.next = tail;
        tail.prev = head;
    }
    
    public int get(int key) {
        Node node = map.get(key);
        if (node == null) {
            return -1;
        }
        moveToHead(node);
        return node.val;
    }
    
    public void put(int key, int val) {
        Node node = map.get(key);
        if (node == null) {
            node = new Node(key, val);
            addToHead(node);
            map.put(key, node);
            size++;
            if (size > capacity) {
                Node rm = removeTail();
                map.remove(rm.key);
                size--;
            }
        } else {
            node.val = val;
            moveToHead(node);
        }
    }

    public void addToHead(Node node) {
        node.prev = head;
        node.next = head.next;
        head.next.prev = node;
        head.next = node;
    }

    public Node removeTail() {
        Node rmNode = tail.prev;
        removeNode(rmNode);
        return rmNode;
    }

    public void removeNode(Node node) {
        node.prev.next = node.next;
        node.next.prev = node.prev;
    }

    public void moveToHead(Node node) {
        removeNode(node);
        addToHead(node);
    }
}

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache obj = new LRUCache(capacity);
 * int param_1 = obj.get(key);
 * obj.put(key,value);
 */
```
