# 1122.数组的相对排序

```java
class Solution {
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
        int[] counting = new int[1001];
        for (int i = 0; i < arr1.length; i++) {
            counting[arr1[i]]++;
        }
        int[] result = new int[arr1.length];
        int index = 0;
        for (int i = 0; i < arr2.length; i++) {
            while (counting[arr2[i]] > 0) {
                result[index++] = arr2[i];
                counting[arr2[i]]--;
            }
        }
        for (int i = 0; i < counting.length; i++) {
            while (counting[i] > 0) {
                result[index++] = i;
                counting[i]--;
            }
        }
        return result;
    }
}
```

# 1128.等价多米诺骨牌对的数量

```java
class Solution {
    public int numEquivDominoPairs(int[][] dominoes) {
        Map<Integer, Integer> resultMap = new HashMap<>();
        int result = 0;
        for (int i = 0; i < dominoes.length; i++) {
            int numberMin = Math.min(dominoes[i][0], dominoes[i][1]);
            int numberMax = Math.max(dominoes[i][0], dominoes[i][1]);
            int key = numberMin * 10 + numberMax;
            int cnt = resultMap.getOrDefault(key, 0);
            result += cnt;
            resultMap.put(key, cnt + 1);
        }
        return result;
    }
}
```

```java
        num[1] = 1;
```

```java
class Solution {
    public int countCharacters(String[] words, String chars) {
        Map<Character, Integer> initMap = new HashMap<>();
        char[] intiCharList = chars.toCharArray();
        for (int i = 0; i < intiCharList.length; i++) {
            initMap.put(intiCharList[i], initMap.getOrDefault(intiCharList[i], 0) + 1);
        }
        int result = 0;
        for (int i = 0; i < words.length; i++) {
            String tempString = words[i];
            char[] tempCharList = tempString.toCharArray();
            Map<Character, Integer> tempMap = new HashMap<>();
            for (int j = 0; j < tempCharList.length; j++) {
                tempMap.put(tempCharList[j], tempMap.getOrDefault(tempCharList[j], 0) + 1);
            }
            boolean isT = true;
            for (Character key : tempMap.keySet()) {
                int tempCnt = tempMap.getOrDefault(key, 0);
                int initCnt = initMap.getOrDefault(key, 0);
                if (tempCnt > initCnt) {
                    isT = false;
                    break;
                }
            }
            if (isT) {
                result += tempCharList.length;
            }
        }
        return result;
    }
```

# 1171.从链表中删去综合值为零的连续节点

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode removeZeroSumSublists(ListNode head) {
        ListNode newHead = new ListNode(0, head);
        ListNode node = newHead;
        Map<Integer, ListNode> map = new HashMap<>();
        int sum = 0;
        while (node != null) {
            sum += node.val;
            map.put(sum, node);
            node = node.next;
        }
        sum = 0;
        node = newHead;
        while (node != null) {
            sum += node.val;
            node.next = map.get(sum).next;
            node = node.next;
        }
        return newHead.next;
    }
}
```

# 1184.公交站间的距离

```java
class Solution {
    public int distanceBetweenBusStops(int[] distance, int start, int destination) {
        int sum = 0;
        int len = 0;
        if (start > destination) {
            int temp = start;
            start = destination;
            destination = temp;
        }
        for (int i = 0; i < distance.length; i++) {
            sum += distance[i];
            if (i >= start && i < destination) {
                len += distance[i];
            }
        }
        return Math.min(len, sum - len);
    }
}
```

```java
class Solution {
    public int maxNumberOfBalloons(String text) {
        String balloon = "balloon";
        char[] balloonChar = balloon.toCharArray();
        Map<Character, Integer> initMap = new HashMap<>();
        for (int i = 0; i < balloonChar.length; i++) {
            initMap.put(balloonChar[i], initMap.getOrDefault(balloonChar[i], 0) + 1);
        }
        char[] tempChar = text.toCharArray();
        Map<Character, Integer> tempMap = new HashMap<>();
        for (int i = 0; i < tempChar.length; i++) {
            tempMap.put(tempChar[i], tempMap.getOrDefault(tempChar[i], 0) + 1);
        }
        int result = 9999;
        for (Character key : initMap.keySet()) {
            int initCnt = initMap.getOrDefault(key, 0);
            int tempCnt = tempMap.getOrDefault(key, 0);
            result = Math.min(result, tempCnt / initCnt);
        }
        return result;
    }
}
```

# 1200.最小绝对差

```java
class Solution {
    public List<List<Integer>> minimumAbsDifference(int[] arr) {
        HashMap<Integer, List<List<Integer>>> tempMap = new HashMap<>();
        Arrays.sort(arr);
        for (int i = 0; i < arr.length - 1; i++) {
            List<Integer> tempArray = new ArrayList<>();
            tempArray.add(arr[i]);
            tempArray.add(arr[i + 1]);
            int temp = arr[i + 1] - arr[i];
            List<List<Integer>> arrArray = tempMap.getOrDefault(temp, new ArrayList<>());
            arrArray.add(tempArray);
            tempMap.put(temp, arrArray);
        }
        int min = Integer.MAX_VALUE;
        for (Integer cha : tempMap.keySet()) {
            min = Math.min(cha, min);
        }
        return tempMap.get(min);
    }
}
```
