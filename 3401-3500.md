# 3402.使每一列严格递增的最少操作次数

```java
class Solution {
    public int minimumOperations(int[][] grid) {
        int result = 0;
        int[] maxList = new int[grid[0].length];
        for (int i = 0; i < grid[0].length; i++) {
            maxList[i] = grid[0][i];
        }
        for (int i = 1; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] > maxList[j]) {
                    maxList[j] = grid[i][j];
                } else {
                    result += maxList[j] - grid[i][j] + 1;
                    maxList[j]++;
                }
            }
        }
        return result;
    }
}
```

# 3407.子字符串匹配模式

```java
class Solution {
    public boolean hasMatch(String s, String p) {
        String[] part = p.split("\\*");
        if (part.length == 0) {
            return true;
        }
        if (part.length == 1) {
            return s.indexOf(part[0]) != -1;
        }
        int oneIndex = s.indexOf(part[0]);
        int twoIndex = s.indexOf(part[1], oneIndex + part[0].length());
        return oneIndex != -1 && twoIndex != -1;
    }
}
```

# 3417.跳过交替单元格的之字形遍历

```java
class Solution {
    public List<Integer> zigzagTraversal(int[][] grid) {
        List<Integer> resultList = new ArrayList<>();
        for (int i = 0; i < grid.length; i++) {
            if (i % 2 == 0) {
                for (int j = 0; j < grid[0].length; j+=2) {
                    resultList.add(grid[i][j]);
                }
            } else {
                for (int j = (grid[0].length % 2 == 0 ? grid[0].length - 1 : grid[0].length - 2); j >= 0; j -= 2) {
                    resultList.add(grid[i][j]);
                }
            }
        }
        return resultList;
    }
}
```

```java
        int result = Math.abs(nums[nums.length - 1] - nums[0]);
```

# 3432.统计元素和差值为偶数的分区方案

```java
class Solution {
    public int countPartitions(int[] nums) {
        int left = nums[0];
        int right = 0;
        for (int i = 1; i < nums.length; i++) {
            right += nums[i];
        }
        int result = 0;
        for (int i = 1; i < nums.length; i++) {
            if ((left - right) % 2 == 0) {
                result++;
            }
            left += nums[i];
            right -= nums[i];
        }
        return result;
    }
}
```

# 3442.奇偶频次间的最大差值Ⅰ

```java
class Solution {
    public int maxDifference(String s) {
        int[] cnt = new int[26];
        for (int i = 0; i < s.length(); i++) {
            char tempChar = s.charAt(i);
            cnt[tempChar - 'a']++;
        }
        int maxJi = 0;
        int minOu = 999;
        for (int i = 0; i < cnt.length; i++) {
            if (cnt[i] == 0) {
                continue;
            }
            if (cnt[i] % 2 == 0) {
                minOu = Math.min(minOu, cnt[i]);
            } else {
                maxJi = Math.max(maxJi, cnt[i]);
            }
        }
        return maxJi - minOu;
    }
}
```

```java
class Solution {

        int sum = 0;

            if ((i - k < 0 || nums[i - k] < nums[i]) && (i + k > nums.length - 1 || nums[i + k] < nums[i])) {

        }
```

# 3456.找出长度为K的特殊子字符串

```java
class Solution {
    public boolean hasSpecialSubstring(String s, int k) {
        for (int i = 0; i < s.length() - k + 1; i++) {
            String tempString = s.substring(i, i + k);
            char temp = s.charAt(i);
            boolean isT = true;
            if (i > 0 && s.charAt(i - 1) == temp) {
                isT = false;
            }
            if (i + k < s.length() && s.charAt(i + k) == temp) {
                isT = false;
            }
            for (int j = 0; j < tempString.length(); j++) {
                if (tempString.charAt(j) != temp) {
                    isT = false;
                }
            }
            if (isT) {
                return true;
            }
        }
        return false;
    }
}
```

# 3461.判断操作后字符串中的数字是否相等Ⅰ

```java
class Solution {
    public boolean hasSameDigits(String s) {
        while (s.length() > 2) {
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < s.length() - 1; i++) {
                sb.append((s.charAt(i) - '0' + s.charAt(i + 1) - '0') % 10);
            }
            s = sb.toString();
        }
        return s.charAt(0) == s.charAt(1);
    }
}
```

```java
class Solution {
    public int numOfUnplacedFruits(int[] fruits, int[] baskets) {
        int cnt = 0;
        for (int i = 0; i < fruits.length; i++) {
            boolean isIn = false;
            for (int j = 0; j < baskets.length; j++) {
                if (baskets[j] >= fruits[i]) {
                    isIn = true;
                    baskets[j] = -1;
                    break;
                }
            }
            if (!isIn) {
                cnt++;
            }
        }
        return cnt;
    }
}
```

# 3483.不同三位偶数的数目

```java
class Solution {
    public int totalNumbers(int[] digits) {
        Set<Integer> resultSet = new HashSet<>();
        for (int i = 0; i < digits.length; i++) {
            for (int j = 0; j < digits.length; j++) {
                for (int k = 0; k < digits.length; k++) {
                    if (i != j && i != k && j != k && digits[k] % 2 == 0 && digits[i] != 0) {
                        int number = digits[i] * 100 + digits[j] * 10 + digits[k];
                        resultSet.add(number);
                    }
                }
            }
        }
        return resultSet.size();
    }
}
```

# 3487.删除后的最大子数组元素和

```java
class Solution {
    public int maxSum(int[] nums) {
        Set<Integer> resultSet = new HashSet<>();
        int sum = 0;
        int count = 0;
        int max = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (resultSet.add(nums[i]) && nums[i] >= 0) {
                sum += nums[i];
                count++;
            }
            max = Math.max(max, nums[i]);
        }
        return count == 0 ? max : sum;
    }
}
```
