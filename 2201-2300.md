```java
class Solution {
    public boolean divideArray(int[] nums) {
        Set<Integer> tempSet = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            if (!tempSet.add(nums[i])) {
                tempSet.remove(nums[i]);
            }
        }
        return tempSet.size() == 0;
    }
}
```

# 2210.统计数组中峰和谷的数量

```java
class Solution {
    public int countHillValley(int[] nums) {
        int result = 0;
        int preNumber = nums[0];
        int thisNumber = nums[1];
        int nextNumber = nums[2];
        for (int i = 2; i < nums.length - 1; i++) {
            if ((thisNumber > preNumber && thisNumber > nextNumber) || (thisNumber < preNumber && thisNumber < nextNumber)) {
                result++;
                preNumber = nums[i - 1];
                thisNumber = nums[i];
                nextNumber = nums[i + 1];
            } else if (thisNumber == nextNumber) {
                nextNumber = nums[i + 1];
            } else {
                preNumber = nums[i - 1];
                thisNumber = nums[i];
                nextNumber = nums[i + 1];
            }
        }
        if ((thisNumber > preNumber && thisNumber > nextNumber) || (thisNumber < preNumber && thisNumber < nextNumber)) {
            result++;
        }
        return result;
    }
}
```

```java
            if (!set1.contains(nums2[i])) {
                listSet2.add(nums2[i]);
            }
        }
        for (int i = 0; i < nums1.length; i++) {
            if (!set2.contains(nums1[i])) {
                listSet1.add(nums1[i]);
            }
        }
        for (Integer key : listSet1) {
            list1.add(key);
        }
        for (Integer key : listSet2) {
            list2.add(key);
        }
        List<List<Integer>> result = new ArrayList<>();
        result.add(list1);
        result.add(list2);
        return result;
    }
```

# 2224.转化时间需要的最小操作数

```java
class Solution {
    public int convertTime(String current, String correct) {
        String[] currentString = current.split(":");
        String[] correctString = correct.split(":");
        int result = (Integer.valueOf(correctString[0]) - Integer.valueOf(currentString[0])) * 60 + (Integer.valueOf(correctString[1]) - Integer.valueOf(currentString[1]));
        int cnt = 0;
        while (result > 0) {
            if (result >= 60) {
                result -= 60;
                cnt++;
            } else if (result >= 15) {
                result -= 15;
                cnt++;
            } else if (result >= 5) {
                result -= 5;
                cnt++;
            } else {
                result -= 1;
                cnt++;
            }
        } 
        return cnt;
    }
}
```

# 2231.按奇偶性交换后的最大数字

```java
class Solution {
    public int largestInteger(int num) {
        String numString = String.valueOf(num);
        char[] numChar = numString.toCharArray();
        List<Integer> jiNumber = new ArrayList<>();
        List<Integer> jiIndex = new ArrayList<>();
        List<Integer> ouNumber = new ArrayList<>();
        List<Integer> ouIndex = new ArrayList<>();
        for (int i = 0; i < numChar.length; i++) {
            int tempInt = numChar[i] - '0';
            if (tempInt % 2 == 0) {
                ouNumber.add(tempInt);
                ouIndex.add(i);
            } else {
                jiNumber.add(tempInt);
                jiIndex.add(i);
            }
        }
        Collections.sort(jiNumber, Collections.reverseOrder());
        Collections.sort(ouNumber, Collections.reverseOrder());
        int[] result = new int[numChar.length];
        for (int i = 0; i < jiNumber.size(); i++) {
            int index = jiIndex.get(i);
            int number = jiNumber.get(i);
            result[index] = number;
        }
        for (int i = 0; i < ouNumber.size(); i++) {
            int index = ouIndex.get(i);
            int number = ouNumber.get(i);
            result[index] = number;
        }
        int resultNumber = 0;
        for (int i = 0; i < result.length; i++) {
            resultNumber = resultNumber * 10 + result[i];
        }
        return resultNumber;
    }
}
```

# 2236.判断根结点是否等于子结点之和

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean checkTree(TreeNode root) {
        return root.val == (root.left.val + root.right.val);
    }
}
```

```java
        List<Integer> result = new ArrayList<>();

        for (int i = 0; i < nums.length; i++) {
            for (int j = 0; j < nums[i].length; j++) {

            }
        }
        int cnt = nums.length;
```

```java
class Solution {
    public String removeDigit(String number, char digit) {
        int result = 0;
        for (int i = 0; i < number.length(); i++) {
            if (number.charAt(i) == digit) {
                if (i < number.length() - 1 && number.charAt(i) < number.charAt(i + 1)) {
                    return number.substring(0, i) + number.substring(i + 1, number.length());
                }
                result = i;
            }
        }
        return number.substring(0, result) + number.substring(result + 1, number.length());
    }
}
```

# 2273.移除字母异位词后的结果数组

```java
class Solution {
    public List<String> removeAnagrams(String[] words) {
        List<char[]> tempList = new ArrayList<>();
        for (int i = 0; i < words.length; i++) {
            char[] tempChar = words[i].toCharArray();
            Arrays.sort(tempChar);
            tempList.add(tempChar);
        }
        List<String> resultList = new ArrayList<>();
        resultList.add(words[0]);
        char[] preList = tempList.get(0);
        for (int i = 1; i < tempList.size(); i++) {
            char[] temp1 = tempList.get(i);
            if (temp1.length != preList.length) {
                resultList.add(words[i]);
                preList = temp1;
                continue;
            }
            for (int j = 0; j < temp1.length; j++) {
                if (temp1[j] != preList[j]) {
                    resultList.add(words[i]);
                    preList = temp1;
                    break;
                }
            }
        }
        return resultList;
    }
}
```

# 2278.字母在字符串中的百分比

```java
class Solution {
    public int percentageLetter(String s, char letter) {
        char[] ch = s.toCharArray();
        int count = 0;
        for (int i = 0; i < ch.length; i++) {
            if (letter == ch[i]) {
                count++;
            }
        }
        return count * 100 / ch.length;
    }
}
```

```java
    public boolean digitCount(String num) {
        int[] cnt = new int[11];
        for (int i = 0; i < num.length(); i++) {

            cnt[temp]++;
```

# 2287.重排字符形成目标字符串

```java
class Solution {
    public int rearrangeCharacters(String s, String target) {
        int[] charList = new int[26];
        for (int i = 0; i < s.length(); i++) {
            char tempChar = s.charAt(i);
            charList[tempChar - 'a']++;
        }
        int result = 999;
        int[] targetCharList = new int[26];
        for (int i = 0; i < target.length(); i++) {
            char tempChar = target.charAt(i);
            targetCharList[tempChar - 'a']++;
        }
        for (int i = 0; i < charList.length; i++) {
            if (targetCharList[i] != 0) {
                result = Math.min(result, charList[i] / targetCharList[i]);
            }
        }
        return result;
    }
}
```
