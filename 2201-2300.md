```java
class Solution {
    public int countHillValley(int[] nums) {
        int result = 0;
        int preNumber = nums[0];
        int thisNumber = nums[1];
        int nextNumber = nums[2];
        for (int i = 2; i < nums.length - 1; i++) {
            if ((thisNumber > preNumber && thisNumber > nextNumber) || (thisNumber < preNumber && thisNumber < nextNumber)) {
                result++;
                preNumber = nums[i - 1];
                thisNumber = nums[i];
                nextNumber = nums[i + 1];
            } else if (thisNumber == nextNumber) {
                nextNumber = nums[i + 1];
            } else {
                preNumber = nums[i - 1];
                thisNumber = nums[i];
                nextNumber = nums[i + 1];
            }
        }
        if ((thisNumber > preNumber && thisNumber > nextNumber) || (thisNumber < preNumber && thisNumber < nextNumber)) {
            result++;
        }
        return result;
    }
}
```

```java
class Solution {
    public int minBitFlips(int start, int goal) {
        List<Integer> startList = new ArrayList<>();
        List<Integer> goalList = new ArrayList<>();
        for (int i = 31; i >= 0; i--) {
            startList.add((start >> i) & 1);
            goalList.add((goal >> i) & 1);
        }
        int result = 0;
        for (int i = 0; i < startList.size(); i++) {
            if (startList.get(i) != goalList.get(i)) {
                result++;
            }
        }
        return result;
    }
}
```

```java
            int[] temp = new int[nums.length - 1];
```

# 2231.按奇偶性交换后的最大数字

```java
class Solution {
    public int largestInteger(int num) {
        String numString = String.valueOf(num);
        char[] numChar = numString.toCharArray();
        List<Integer> jiNumber = new ArrayList<>();
        List<Integer> jiIndex = new ArrayList<>();
        List<Integer> ouNumber = new ArrayList<>();
        List<Integer> ouIndex = new ArrayList<>();
        for (int i = 0; i < numChar.length; i++) {
            int tempInt = numChar[i] - '0';
            if (tempInt % 2 == 0) {
                ouNumber.add(tempInt);
                ouIndex.add(i);
            } else {
                jiNumber.add(tempInt);
                jiIndex.add(i);
            }
        }
        Collections.sort(jiNumber, Collections.reverseOrder());
        Collections.sort(ouNumber, Collections.reverseOrder());
        int[] result = new int[numChar.length];
        for (int i = 0; i < jiNumber.size(); i++) {
            int index = jiIndex.get(i);
            int number = jiNumber.get(i);
            result[index] = number;
        }
        for (int i = 0; i < ouNumber.size(); i++) {
            int index = ouIndex.get(i);
            int number = ouNumber.get(i);
            result[index] = number;
        }
        int resultNumber = 0;
        for (int i = 0; i < result.length; i++) {
            resultNumber = resultNumber * 10 + result[i];
        }
        return resultNumber;
    }
}
```

```java
        int left = 0;
        int right = k - 1;
```

```java
class Solution {
    public List<String> removeAnagrams(String[] words) {
        List<char[]> tempList = new ArrayList<>();
        for (int i = 0; i < words.length; i++) {
            char[] tempChar = words[i].toCharArray();
            Arrays.sort(tempChar);
            tempList.add(tempChar);
        }
        List<String> resultList = new ArrayList<>();
        resultList.add(words[0]);
        char[] preList = tempList.get(0);
        for (int i = 1; i < tempList.size(); i++) {
            char[] temp1 = tempList.get(i);
            if (temp1.length != preList.length) {
                resultList.add(words[i]);
                preList = temp1;
                continue;
            }
            for (int j = 0; j < temp1.length; j++) {
                if (temp1[j] != preList[j]) {
                    resultList.add(words[i]);
                    preList = temp1;
                    break;
                }
            }
        }
        return resultList;
    }
}
```

```java
    public int rearrangeCharacters(String s, String target) {
        int[] charList = new int[26];
        for (int i = 0; i < s.length(); i++) {
            char tempChar = s.charAt(i);
            charList[tempChar - 'a']++;
        }
        int result = 999;
        int[] targetCharList = new int[26];
        for (int i = 0; i < target.length(); i++) {
            char tempChar = target.charAt(i);
            targetCharList[tempChar - 'a']++;
        }
        for (int i = 0; i < charList.length; i++) {
            if (targetCharList[i] != 0) {
                result = Math.min(result, charList[i] / targetCharList[i]);
            }
        }
        return result;
    }
```

# 2293.极大极小游戏

```java
class Solution {
    public int minMaxGame(int[] nums) {
        while (nums.length > 1) {
            int[] tempList = new int[nums.length / 2];
            for (int i = 0; i < tempList.length; i++) {
                int temp1 = nums[i * 2];
                int temp2 = nums[i * 2 + 1];
                int result = -1;
                if (i % 2 == 0) {
                    result = Math.min(temp1, temp2);
                } else {
                    result = Math.max(temp1, temp2);
                }
                tempList[i] = result;
            }
            nums = tempList;
        }
        return nums[0];
    }
}
```

# 2299.强密码检验器Ⅱ

```java
class Solution {
    public boolean strongPasswordCheckerII(String password) {
        if (password.length() < 8) {
            return false;
        }
        boolean haveLowLetter = false;
        boolean haveUpLetter = false;
        boolean haveNumber = false;
        boolean haveSpecial = false;
        String specialString = "!@#$%^&*()-+";
        for (int i = 0; i < password.length(); i++) {
            char temp = password.charAt(i);
            if (i < password.length() - 1 && temp == password.charAt(i + 1)) {
                return false;
            }
            if (Character.isDigit(temp)) {
                haveNumber = true;
            }
            if (temp >= 'a' && temp <= 'z') {
                haveLowLetter = true;
            }
            if (temp >= 'A' && temp <= 'Z') {
                haveUpLetter = true;
            }
            if (specialString.indexOf(temp) != -1) {
                haveSpecial = true;
            }
        }
        return haveLowLetter && haveNumber && haveSpecial && haveUpLetter;
    }
}
```
