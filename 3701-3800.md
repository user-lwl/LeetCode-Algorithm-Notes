# 3702.按位异或非零的最长子序列

```java
class Solution {
    public int longestSubsequence(int[] nums) {
        Arrays.sort(nums);
        if (nums[nums.length - 1] == 0) {
            return 0;
        }
        int result = nums[0];
        int len = nums.length;
        int index = nums.length - 1;
        for (int i = 1; i < nums.length; i++) {
            result ^= nums[i];
        }
        while (result == 0) {
            result ^= nums[index--];
            len--;
        }
        return len;
    }
}
```

# 3707.相等子字符串分数

```java
class Solution {
    public boolean scoreBalance(String s) {
        int[] score = new int[s.length()];
        int sum = 0;
        for (int i = 0; i < s.length(); i++) {
            score[i] = s.charAt(i) - 'a' + 1;
            sum += score[i];
        }
        int index = 0;
        int left = 0;
        int right = sum;
        while (left < right) {
            left += score[index];
            right -= score[index];
            index++;
            if (left == right) {
                return true;
            }
        }
        return false;
    }
}
```

```java
            if (nums[i] == nums[i - 1] + nums[i - 2]) {
```

```java
class Solution {
```

```java
        Set<Integer> tempSet = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            tempSet.add(nums[i]);
        }
        int target = 1;
        while (target > 0) {
            int temp = target * k;

            }
            target++;
        }
        return -1;
    }
```

# 3727.最大交替平方和

```java
class Solution {
    public long maxAlternatingSum(int[] nums) {
        long[] temp = new long[nums.length];
        for (int i = 0; i < nums.length; i++) {
            temp[i] = (long) nums[i] * nums[i];
        }
        Arrays.sort(temp);
        int left = 0;
        int right = temp.length - 1;
        long result = 0;
        while (left < right) {
            result += temp[right--];
            result -= temp[left++];
        }
        if (left == right) {
            result += temp[left];
        }
        return result;
    }
}
```

```java
                }
            }
        }
        return result == 99999 ? -1 : result;
    }
}
```

# 3741.三个相等元素之间的最小距离Ⅱ

```java
class Solution {
    public int minimumDistance(int[] nums) {
        Map<Integer, List<Integer>> tempMap = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            List<Integer> tempList = tempMap.getOrDefault(nums[i], null);
            if (tempList == null) {
                tempList = new ArrayList<>();
                tempMap.put(nums[i], tempList);
            }
            tempList.add(i);
        }
        int result = Integer.MAX_VALUE;
        for (Integer key : tempMap.keySet()) {
            List<Integer> tempList = tempMap.get(key);
            for (int i = 2; i < tempList.size(); i++) {
                int tempLen1 = tempList.get(i);
                int tempLen2 = tempList.get(i - 1);
                int tempLen3 = tempList.get(i - 2);
                int tempRes = Math.abs(tempLen1 - tempLen2) + Math.abs(tempLen1 - tempLen3) + Math.abs(tempLen2 - tempLen3);
                result = Math.min(result, tempRes);
            }
        }
        return result == Integer.MAX_VALUE ? -1 : result;
    }
}
```

# 3769.二进制反射排序

```java
class Solution {
    public int[] sortByReflection(int[] nums) {
        int[][] tempNumber = new int[nums.length][2];
        for (int i = 0; i < nums.length; i++) {
            tempNumber[i][0] = nums[i];
            String tempBin = Integer.toBinaryString(nums[i]);
            StringBuilder sb = new StringBuilder();
            sb.append(tempBin);
            sb.reverse();
            String revBin = sb.toString();
            int revInt = Integer.parseInt(revBin, 2);
            tempNumber[i][1] = revInt;
        }
        Arrays.sort(tempNumber, new Comparator<int[]>() {
            @Override
            public int compare(int[] a1, int[] a2) {
                if (a1[1] == a2[1]) {
                    return Integer.compare(a1[0], a2[0]);
                }
                return Integer.compare(a1[1], a2[1]);
            }
        });
        int[] result = new int[nums.length];
        for (int i = 0; i < result.length; i++) {
            result[i] = tempNumber[i][0];
        }
        return result;
    }
}
```

```java
class Solution {
    public int mirrorDistance(int n) {
        return Math.abs(n - rev(n));
    }

    public int rev(int n) {
        int result = 0;
        while (n > 0) {
            result *= 10;
            result += n % 10;
            n /= 10;
        }
        return result;
    }
}
```
