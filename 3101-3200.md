# 3105.最长的严格递增或递减子数组

```java
class Solution {
    public int longestMonotonicSubarray(int[] nums) {
        int result = 1;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] > nums[i + 1]) {
                for (int j = i + 1; j < nums.length; j++) {
                    if (nums[j - 1] <= nums[j]) {
                        result = Math.max(result, j - i);
                        break;
                    }
                    if (j == nums.length - 1) {
                        result = Math.max(result, j - i + 1);
                        break;
                    }
                }
            } else if (nums[i] < nums[i + 1]) {
                for (int j = i + 1; j < nums.length; j++) {
                    if (nums[j - 1] >= nums[j]) {
                        result = Math.max(result, j - i);
                        break;
                    }
                    if (j == nums.length - 1) {
                        result = Math.max(result, j - i + 1);
                        break;
                    }
                }
            }
        }
        return result;
    }
}
```

# 3136.有效单词

```java
class Solution {
    public boolean isValid(String word) {
        Set<Character> yuanYinSet = new HashSet<>();
        Character[] yuanYin = {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'};
        yuanYinSet.addAll(Arrays.asList(yuanYin));
        char[] stringChar = word.toCharArray();
        int len = stringChar.length;
        boolean isYuan = false;
        boolean isFu = false;
        if (len < 3) {
            return false;
        }
        for (int i = 0; i < len; i++) {
            char temp = stringChar[i];
            if (temp >= '0' && temp <= '9') {
                continue;
            } else if ((temp >= 'a' && temp <= 'z') || (temp >= 'A' && temp <= 'Z')) {
                if (yuanYinSet.contains(temp)) {
                    isYuan = true;
                } else {
                    isFu = true;
                }
            } else {
                return false;
            }
        }
        return isYuan && isFu;
    }
}
```

# 3169.无需开会的工作日

```java
class Solution {
    public int countDays(int days, int[][] meetings) {
        Arrays.sort(meetings, Comparator.comparingInt(a -> a[0]));
        int meetingDay = 0;
        int maxMeetingDay = 0;
        for (int i = 0; i < meetings.length; i++) {
            if (maxMeetingDay >= meetings[i][1]) {
                continue;
            } else if (maxMeetingDay >= meetings[i][0]) {
                meetingDay += meetings[i][1] - maxMeetingDay;
            } else {
                meetingDay += meetings[i][1] - meetings[i][0] + 1;
            }
            maxMeetingDay = Math.max(meetings[i][1], maxMeetingDay);
        }
        int notMeetingDay = days - meetingDay;
        return notMeetingDay;
    }
}
```

# 3194.最小元素和最大元素的最小平均值

```java
class Solution {
    public double minimumAverage(int[] nums) {
        Arrays.sort(nums);
        int startIndex = 0;
        int endIndex = nums.length - 1;
        double min = Integer.MAX_VALUE;
        while (endIndex > startIndex) {
            double middle = (double) (nums[startIndex] + nums[endIndex]) / 2;
            min = Math.min(min, middle);
            endIndex--;
            startIndex++;
        }
        return min;
    }
}
```
