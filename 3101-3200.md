# 3114.替换字符可以得到的最晚时间

```java
class Solution {
    public String findLatestTime(String s) {
        String[] timeStamp = s.split(":");
        String hour = timeStamp[0];
        String minute = timeStamp[1];
        if (hour.charAt(0) == '?' && hour.charAt(1) == '?') {
            hour = "11";
        } else if (hour.charAt(0) == '?') {
            if (hour.charAt(1) <='1') {
                hour = "1" + hour.charAt(1);
            } else {
                hour = "0" + hour.charAt(1);
            }
        } else if (hour.charAt(1) == '?') {
            if (hour.charAt(0) == '0') {
                hour = hour.charAt(0) + "9";
            } else {
                hour = hour.charAt(0) + "1";
            }
        }

        if (minute.charAt(0) == '?' && minute.charAt(1) == '?') {
            minute = "59";
        } else if (minute.charAt(0) == '?') {
            minute = "5" + minute.charAt(1);
        } else if (minute.charAt(1) == '?') {
            minute = minute.charAt(0) + "9";
        }
        return hour + ":" + minute;
    }
}
```

```java
class Solution {
    public boolean canMakeSquare(char[][] grid) {
        for (int i = 0; i < grid.length - 1; i++) {
            for (int j = 0; j < grid[0].length - 1; j++) {
                int bInt = 0;
                if (grid[i][j] == 'B') {
                    bInt++;
                }
                if (grid[i][j + 1] == 'B') {
                    bInt++;
                }
                if (grid[i + 1][j] == 'B') {
                    bInt++;
                }
                if (grid[i + 1][j + 1] == 'B') {
                    bInt++;
                }
                if (!(bInt == 2)) {
                    return true;
                }
            }
        }
        return false;
    }
}
```

# 3136.有效单词

```java
class Solution {
    public boolean isValid(String word) {
        Set<Character> yuanYinSet = new HashSet<>();
        Character[] yuanYin = {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'};
        yuanYinSet.addAll(Arrays.asList(yuanYin));
        char[] stringChar = word.toCharArray();
        int len = stringChar.length;
        boolean isYuan = false;
        boolean isFu = false;
        if (len < 3) {
            return false;
        }
        for (int i = 0; i < len; i++) {
            char temp = stringChar[i];
            if (temp >= '0' && temp <= '9') {
                continue;
            } else if ((temp >= 'a' && temp <= 'z') || (temp >= 'A' && temp <= 'Z')) {
                if (yuanYinSet.contains(temp)) {
                    isYuan = true;
                } else {
                    isFu = true;
                }
            } else {
                return false;
            }
        }
        return isYuan && isFu;
    }
}
```

# 3169.无需开会的工作日

```java
class Solution {
    public int countDays(int days, int[][] meetings) {
        Arrays.sort(meetings, Comparator.comparingInt(a -> a[0]));
        int meetingDay = 0;
        int maxMeetingDay = 0;
        for (int i = 0; i < meetings.length; i++) {
            if (maxMeetingDay >= meetings[i][1]) {
                continue;
            } else if (maxMeetingDay >= meetings[i][0]) {
                meetingDay += meetings[i][1] - maxMeetingDay;
            } else {
                meetingDay += meetings[i][1] - meetings[i][0] + 1;
            }
            maxMeetingDay = Math.max(meetings[i][1], maxMeetingDay);
        }
        int notMeetingDay = days - meetingDay;
        return notMeetingDay;
    }
}
```

# 3174.清除数字

```java
class Solution {
    public String clearDigits(String s) {
        List<Character> tempChar = s.chars().mapToObj(c -> (char) c).collect(Collectors.toList());
        for (int i = 0; i < tempChar.size() - 1; i++) {
            if (tempChar.get(i + 1) >= '0' && tempChar.get(i + 1) <= '9') {
                tempChar.remove(i);
                tempChar.remove(i);
                i -= 2;
                if (i < 0) {
                    i = -1;
                }
            }
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < tempChar.size(); i++) {
            sb.append(tempChar.get(i));
        }
        return sb.toString();
    }
}
```

# 3194.最小元素和最大元素的最小平均值

```java
class Solution {
    public double minimumAverage(int[] nums) {
        Arrays.sort(nums);
        int startIndex = 0;
        int endIndex = nums.length - 1;
        double min = Integer.MAX_VALUE;
        while (endIndex > startIndex) {
            double middle = (double) (nums[startIndex] + nums[endIndex]) / 2;
            min = Math.min(min, middle);
            endIndex--;
            startIndex++;
        }
        return min;
    }
}
```
