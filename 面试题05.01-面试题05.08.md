# 面试题 05.01.插入

```java
class Solution {
    public int insertBits(int N, int M, int i, int j) {
        List<Integer> NList = new ArrayList<>();
        List<Integer> MList = new ArrayList<>();
        int cnt = j - i + 1;
        for (int index = 31; index >= 0; index--) {
            int Ntemp = (N >> index) & 1;
            int Mtemp = (M >> index) & 1;
            NList.add(Ntemp);
            if (index < cnt) {
                MList.add(Mtemp);
            }
        }
        int result = 0;
        int left = NList.size() - j - 1;
        int right = NList.size() - i - 1;
        int indexM = 0;
        for (int index = 0; index < NList.size(); index++) {
            if (index >= left && index <= right) {
                result = result * 2 + MList.get(indexM++);
            } else {
                result = result * 2 + NList.get(index);
            }
        }
        return result;
    }
}
```

# 面试题 05.03.翻转数位

```java
class Solution {
    public int reverseBits(int num) {
        int[] number = new int[32];
        for (int i = 31; i >= 0; i--) {
            number[i] = (num >> i) & 1;
        }
        int result = 0;
        for (int i = 0; i < number.length; i++) {
            if (number[i] == 0) {
                int left = i - 1;
                int right = i + 1;
                int cnt = 1;
                while (left >= 0 && number[left] == 1) {
                    cnt++;
                    left--;
                }
                while (right < number.length && number[right] == 1) {
                    cnt++;
                    right++;
                }
                result = Math.max(result, cnt);
            }
        }
        return result == 0 ? number.length : result;
    }
}
```

# 面试题 05.06.整数转换

```java
class Solution {
    public int convertInteger(int A, int B) {
        int result = 0;
        for (int i = 31; i >= 0; i--) {
            if (((A >> i) & 1) != ((B >> i) & 1)) {
                result++;
            }
        }
        return result;
    }
}
```

# 面试题 05.07.配对交换

```java
class Solution {
    public int exchangeBits(int num) {
        int[] number = new int[32];
        int index = 0;
        for (int i = 31; i > 0; i-=2) {
            int temp1 = (num >> i) & 1;
            int temp2 = (num >> (i - 1)) & 1;
            number[index++] = temp2;
            number[index++] = temp1;
        }
        int result = 0;
        for (int i = 0; i < number.length; i++) {
            result = result * 2 + number[i];
        }
        return result;
    }
}
```
