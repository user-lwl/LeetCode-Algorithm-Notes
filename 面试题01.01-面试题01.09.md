# 面试题 01.01.判定字符是否唯一

```java
class Solution {
    public boolean isUnique(String astr) {
        char[] sChat = astr.toCharArray();
        Set<Character> tempSet = new HashSet<>();
        for (int i = 0; i < sChat.length; i++) {
            Character tempChar = sChat[i];
            if (!tempSet.add(tempChar)) {
                return false;
            }
        }
        return true;
    }
}
```

# 面试题 01.02.判定是否互为字符重排

```java
class Solution {
    public boolean CheckPermutation(String s1, String s2) {
        char[] s1Char = s1.toCharArray();
        char[] s2Char = s2.toCharArray();
        if (s1Char.length != s2Char.length) {
            return false;
        }
        Map<Character, Integer> tempMap = new HashMap<>();
        for (int i = 0; i < s1Char.length; i++) {
            tempMap.put(s1Char[i], tempMap.getOrDefault(s1Char[i], 0) + 1);
        }
        for (int i = 0; i < s2Char.length; i++) {
            int cnt = tempMap.getOrDefault(s2Char[i], 0);
            if (cnt == 0) {
                return false;
            }
            tempMap.put(s2Char[i], cnt - 1);
        }
        return true;
    }
}
```

# 面试题 01.03.URL化

```java
class Solution {
    public String replaceSpaces(String S, int length) {
        char[] sChar = S.toCharArray();
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < length; i++) {
            if (' ' == sChar[i]) {
                sb.append("%20");
            } else {
                sb.append(sChar[i]);
            }
        }
        return sb.toString();
    }
}
```



# 面试题 01.04.回文排列

```java
class Solution {
    public boolean canPermutePalindrome(String s) {
        int[] cnt = new int[256];
        for (int i = 0; i < s.length(); i++) {
            cnt[s.charAt(i)]++;
        }
        int notOu = 0;
        for (int i = 0; i < cnt.length; i++) {
            if (cnt[i] % 2 != 0) {
                notOu++;
            }
            if (notOu > 1) {
                return false;
            }
        }
        return true;
    }
}
```

# 面试题 01.06.字符串压缩

```java
class Solution {
    public String compressString(String S) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < S.length(); i++) {
            char temp = S.charAt(i);
            int startIndex = i;
            while (i < S.length() && S.charAt(i) == temp) {
                i++;
            }
            sb.append(temp).append(i - startIndex);
            i--;
        }
        return sb.length() < S.length() ? sb.toString() : S;
    }
}
```



# 面试题 01.09.字符串轮转

```java
class Solution {
    public boolean isFlipedString(String s1, String s2) {
        if (s1.length() != s2.length()) {
            return false;
        }
        return (s1 + s1).indexOf(s2) != -1;
    }
}
```
