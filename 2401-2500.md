# 2404.出现最频繁的偶数元素

```java
class Solution {
    public int mostFrequentEven(int[] nums) {
        Map<Integer, Integer> resultMap = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] % 2 == 0) {
                resultMap.put(nums[i], resultMap.getOrDefault(nums[i], 0) + 1);
            }
        }
        int result = -1;
        int number = 999999;
        for (Integer key : resultMap.keySet()) {
            int cnt = resultMap.get(key);
            if (cnt > result) {
                result = cnt;
                number = key;
            } else if (cnt == result && key < number) {
                number = key;
            }
        }
        return result == -1 ? -1 : number;
    }
}
```



# 2405.子字符串的最优划分

```java
class Solution {
    public int partitionString(String s) {
        int[] temp = new int[26];
        char[] c = s.toCharArray();
        int max = 1;
        for (int i = 0; i < c.length; i++) {
            if (temp[(int)c[i] - 'a'] == 1) {
                temp = new int[26];
                max++;
            }
            temp[(int)c[i] - 'a']++;
        }
        return max;
    }
}
```

# 2410.运动员和训练师的最大匹配数

```java
class Solution {
    public int matchPlayersAndTrainers(int[] players, int[] trainers) {
        Integer[] playersInteger = Arrays.stream(players).boxed().toArray(Integer[]::new);
        Arrays.sort(playersInteger, new Comparator<Integer>() {
            @Override
            public int compare(Integer a, Integer b) {
                return b - a;
            }
        });
        Integer[] trainersInteger = Arrays.stream(trainers).boxed().toArray(Integer[]::new);
        Arrays.sort(trainersInteger, new Comparator<Integer>() {
            @Override
            public int compare(Integer a, Integer b) {
                return b - a;
            }
        });
        int person = 0;
        int train = 0;
        int max = 0;
        while (person < playersInteger.length && train < trainersInteger.length) {
            if (playersInteger[person] <= trainersInteger[train]) {
                max ++;
                person++;
                train++;
            } else {
                person++;
            }
        }
        return max;
    }
}
```

# 2418.按身高排序

```java
class Solution {
    public String[] sortPeople(String[] names, int[] heights) {
        int[][] heightsNum = new int[heights.length][2];
        for (int i = 0; i < heights.length; i++) {
            heightsNum[i][0] = heights[i];
            heightsNum[i][1] = i;
        }
        Arrays.sort(heightsNum, new Comparator<int[]>() {
            public int compare(int[] o1, int[] o2) {
                return o2[0] - o1[0];
            }
        });
        String[] result = new String[names.length];
        for (int i = 0; i < heightsNum.length; i++) {
            result[i] = names[heightsNum[i][1]];
        }
        return result;
    }
}
```

# 2437.有效时间的数目

```java
class Solution {
    public int countTime(String time) {
        String[] timeStamp = time.split(":");
        String hour = timeStamp[0];
        String second = timeStamp[1];
        int hourCnt = 1;
        int secondCnt = 1;
        if (hour.charAt(0) == '?' && hour.charAt(1) == '?') {
            hourCnt = 24;
        } else if (hour.charAt(0) == '?') {
            if (hour.charAt(1) <= '3') {
                hourCnt = 3;
            } else {
                hourCnt = 2;
            }
        } else if (hour.charAt(1) == '?') {
            if(hour.charAt(0) <= '1') {
                hourCnt = 10;
            } else {
                hourCnt = 4;
            }
        }

        if (second.charAt(0) =='?' && second.charAt(1) == '?') {
            secondCnt = 60;
        } else if (second.charAt(0) == '?') {
            secondCnt = 6;
        } else if (second.charAt(1) == '?') {
            secondCnt = 10;
        }

        return secondCnt * hourCnt;
    }
}
```

# 2441.与对应负数同时存在的最大正整数

```java
class Solution {
    public int findMaxK(int[] nums) {
        Set<Integer> tempSet = new HashSet<>();
        int result = -1;
        Arrays.sort(nums);
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > 0) {
                if (tempSet.contains(nums[i])) {
                    result = Math.max(nums[i], result);
                }
            } else {
                tempSet.add(-nums[i]);
            }
        }
        return result;
    }
}
```

# 2465.不同的平均值数目

```java
class Solution {
    public int distinctAverages(int[] nums) {
        Set<Double> tempSet = new HashSet<>();
        Arrays.sort(nums);
        int leftIndex = 0;
        int rightIndex = nums.length - 1;
        while (leftIndex < rightIndex) {
            Double avg = (double) (nums[leftIndex++] + nums[rightIndex--]) / 2;
            tempSet.add(avg);
        }
        return tempSet.size();
    }
}
```

# 2475.数组中不等三元组的数目

```java
class Solution {
    public int unequalTriplets(int[] nums) {
        Arrays.sort(nums);
        int result = 0;
        int len = nums.length;
        for (int i = 0, j = 0; i < len; i = j) {
            while (j < len && nums[j] == nums[i]) {
                j++;
            }
            result += i * (j - i) * (len - j);
        }
        return result;
    }
}
```



# 2485.找出中枢整数

```java
class Solution {
    int[] prefix;
    public int pivotInteger(int n) {
        prefix = new int[n];
        prefix[0] = 1;
        for (int i = 1; i < n; i++) {
            prefix[i] = prefix[i - 1] + i + 1;
        }
        for (int i = 0; i < n; i++) {
            if (getPrefixSum(i) == getPrefixSum(n - 1) - getPrefixSum(i - 1)) {
                return i + 1;
            }
        }
        return -1;
    }
    public int getPrefixSum(int index) {
        if (index == -1) {
            return 0;
        }
        return prefix[index];
    }
}
```

# 2500.删除每行中的最大值

```java
class Solution {
    public int deleteGreatestValue(int[][] grid) {
        int[] result = new int[grid[0].length];
        for (int i = 0; i < grid.length; i++) {
            Arrays.sort(grid[i]);
            for (int j = 0; j < grid[i].length; j++) {
                result[j] = Math.max(grid[i][j], result[j]);
            }
        }
        int sum = 0;
        for (int i = 0; i < result.length; i++) {
            sum += result[i];
        }
        return sum;
    }
}
```
