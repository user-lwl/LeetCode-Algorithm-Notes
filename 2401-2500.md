```java
class Solution {
    public int mostFrequentEven(int[] nums) {
        Map<Integer, Integer> resultMap = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] % 2 == 0) {
                resultMap.put(nums[i], resultMap.getOrDefault(nums[i], 0) + 1);
            }
        }
        int result = -1;
        int number = 999999;
        for (Integer key : resultMap.keySet()) {
            int cnt = resultMap.get(key);
            if (cnt > result) {
                result = cnt;
                number = key;
            } else if (cnt == result && key < number) {
                number = key;
            }
        }
        return result == -1 ? -1 : number;
    }
}
```

```java
class Solution {
    public int countDaysTogether(String arriveAlice, String leaveAlice, String arriveBob, String leaveBob) {
        int days = Math.min(dateToDays(leaveAlice), dateToDays(leaveBob)) - Math.max(dateToDays(arriveAlice), dateToDays(arriveBob)) + 1;
        return days < 0 ? 0 : days;
    }

    public int dateToDays(String date) {
        int month = Integer.valueOf(date.substring(0, 2));
        int day = Integer.valueOf(date.substring(3));
        int days = 0;
        for (int i = 0; i < month; i++) {
            if (i == 1 || i == 3 || i == 5 || i == 7 || i == 8 || i == 10 || i == 12) {
                days += 31;
            } else if (i == 2) {
                days += 28;
            } else {
                days += 30;
            }
        }
        days += day;
        return days;
    }
}
```

```java
        int maxTime = -1;
        int id = -1;
        for (int i = 0; i < logs.length; i++) {
            int tempId = logs[i][0];

                maxTime = tempTime;

    }
```

```java
class Solution {
    public int unequalTriplets(int[] nums) {
        Arrays.sort(nums);
        int result = 0;
        int len = nums.length;
        for (int i = 0, j = 0; i < len; i = j) {
            while (j < len && nums[j] == nums[i]) {
                j++;
            }
```

# 2496.数组中字符串的最大值

```java
class Solution {
    public int maximumValue(String[] strs) {
        int result = 0;
        for (int i = 0; i < strs.length; i++) {
            boolean haveLetter = false;
            for (int j = 0; j < strs[i].length(); j++) {
                if (Character.isLetter(strs[i].charAt(j))) {
                    haveLetter = true;
                    break;
                }
            }
            if (haveLetter) {
                result = Math.max(result, strs[i].length());
            } else {
                result = Math.max(result, Integer.valueOf(strs[i]));
            }
        }
        return result;
    }
}
```
