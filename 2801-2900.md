# 2815.数组中的最大数对和

```java
class Solution {
    public int maxSum(int[] nums) {
        int result = -1;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                int tempOne = nums[i];
                int tempTwo = nums[j];
                int bigOne = -1;
                int bigTwo = -2;
                while (tempOne > 0) {
                    bigOne = Math.max(bigOne, tempOne % 10);
                    tempOne /= 10;
                }
                while (tempTwo > 0) {
                    bigTwo = Math.max(bigTwo, tempTwo % 10);
                    tempTwo /= 10;
                }
                if (bigOne == bigTwo) {
                    result = Math.max(result, nums[i] + nums[j]);
                }
            }
        }
        return result;
    }
}
```

# 2833.距离原点最远的点

```java
class Solution {
    public int furthestDistanceFromOrigin(String moves) {
        int index = 0;
        int cnt = 0;
        for (int i = 0; i < moves.length(); i++) {
            char tempChar = moves.charAt(i);
            if (tempChar == 'L') {
                index--;
            } else if (tempChar == 'R') {
                index++;
            } else {
                cnt++;
            }
        }
        return Math.abs(index) + cnt;
    }
}
```

# 2843.统计对称整数的数目

```java
class Solution {
    public int countSymmetricIntegers(int low, int high) {
        int result = 0;
        for (int i = low; i <= high; i++) {
            char[] charInt = (String.valueOf(i)).toCharArray();
            if (charInt.length % 2 == 0) {
                int firstSum = 0;
                int lastSum = 0;
                for (int j = 0; j < charInt.length / 2; j++) {
                    firstSum += charInt[j] - '0';
                }
                for (int j = charInt.length / 2; j < charInt.length; j++) {
                    lastSum += charInt[j] - '0';
                }
                if (firstSum == lastSum) {
                    result++;
                }
            }
        }
        return result;
    }
}
```

# 2855.使数组成为递增数组的最少右移次数

```java
class Solution {
    public int minimumRightShifts(List<Integer> nums) {
        int result = -1;
        for (int i = 0; i < nums.size() - 1; i++) {
            if (nums.get(i) > nums.get(i + 1)) {
                if (result == -1) {
                    result = i;
                } else {
                    return -1;
                }
            }
        }
        return result == -1 ? 0 : nums.get(nums.size() - 1) > nums.get(0) ? -1 : nums.size() - 1 - result;
    }
}
```

# 2859.计算K置位下标对应元素的和

```java
class Solution {
    public int sumIndicesWithKSetBits(List<Integer> nums, int k) {
        int result = 0;
        for (int i = 0; i < nums.size(); i++) {
            int cnt = 0;
            for (int j = 31; j >= 0; j--) {
                if (((i >> j) & 1) == 1) {
                    cnt++;
                }
            }
            if (cnt == k) {
                result += nums.get(i);
            }
        }
        return result;
    }
}
```

# 2869.收集元素的最少操作次数

```java
class Solution {
    public int minOperations(List<Integer> nums, int k) {
        Set<Integer> tempSet = new HashSet<>();
        int i = nums.size() - 1;
        for (; i >= 0; i--) {
            if (nums.get(i) <= k) {
                tempSet.add(nums.get(i));
            }
            if (tempSet.size() == k) {
                break;
            }
        }
        return nums.size() - i;
    }
}
```

# 2873.有序三元组中的最大值Ⅰ

```java
class Solution {
    public long maximumTripletValue(int[] nums) {
        long result = Long.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                for (int k = j + 1; k < nums.length; k++) {
                    long temp = (long) (nums[i] - nums[j]) * nums[k];
                    result = Math.max(temp, result);
                }
            }
        }
        return result < 0 ? 0 : result;
    }
}
```

```java
        int num1 = 0;
        int num2 = 0;
        for (int i = 1; i <= n; i++) {
            if (i % m == 0) {
```

# 2899.上一个遍历的整数

```java
class Solution {
    public List<Integer> lastVisitedIntegers(int[] nums) {
        List<Integer> ans = new ArrayList<>();
        List<Integer> seen = new ArrayList<>();
        int k = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != -1) {
                k = 0;
                seen.add(0, nums[i]);
            } else {
                k++;
                if (k <= seen.size()) {
                    ans.add(seen.get(k - 1));
                } else {
                    ans.add(-1);
                }
            }
        }
        return ans;
    }
}
```

# 2900.最长相邻不相等子序列Ⅰ

```java
class Solution {
    public List<String> getLongestSubsequence(String[] words, int[] groups) {
        List<String> result = new ArrayList<>();
        result.add(words[0]);
        for (int i = 1; i < words.length; i++) {
            if (groups[i] != groups[i - 1]) {
                result.add(words[i]);
            }
        }
        return result;
    }
}
```
