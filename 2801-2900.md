# 2808.使循环数组所有元素相等的最少秒数

```java
class Solution {
    public int minimumSeconds(List<Integer> nums) {
        if (nums.size() == 1) {
            return 0;
        }
        List<Integer> tempList = new ArrayList<>(nums);
        tempList.addAll(nums);
        Map<Integer, List<Integer>> tempMap = new HashMap<>();
        for (int i = 0; i < tempList.size(); i++) {
            Integer number = tempList.get(i);
            List<Integer> tempTempList = tempMap.getOrDefault(number, null);
            if (tempTempList == null) {
                tempTempList = new ArrayList<>();
                tempMap.put(number, tempTempList);
            }
            tempTempList.add(i);
        }
        int minMaxLen = (int) 1e9 + 1;
        for (Integer key : tempMap.keySet()) {
            List<Integer> tempTempList = tempMap.get(key);
            int maxLen = 0;
            for (int i = 0; i < tempTempList.size() - 1; i++) {
                int len = tempTempList.get(i + 1) - tempTempList.get(i);
                maxLen = Math.max(maxLen, len);
            }
            minMaxLen = Math.min(minMaxLen, maxLen);
        }
        return minMaxLen / 2;
    }
}
```

# 2815.数组中的最大数对和

```java
class Solution {
    public int maxSum(int[] nums) {
        int result = -1;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                int tempOne = nums[i];
                int tempTwo = nums[j];
                int bigOne = -1;
                int bigTwo = -2;
                while (tempOne > 0) {
                    bigOne = Math.max(bigOne, tempOne % 10);
                    tempOne /= 10;
                }
                while (tempTwo > 0) {
                    bigTwo = Math.max(bigTwo, tempTwo % 10);
                    tempTwo /= 10;
                }
                if (bigOne == bigTwo) {
                    result = Math.max(result, nums[i] + nums[j]);
                }
            }
        }
        return result;
    }
}
```

# 2840.判断通过操作能否让字符串相等Ⅱ

```java
class Solution {
    public boolean checkStrings(String s1, String s2) {
        Map<Character, Integer> jiS1Map = new HashMap<>();
        Map<Character, Integer> ouS1Map = new HashMap<>();
        Map<Character, Integer> jiS2Map = new HashMap<>();
        Map<Character, Integer> ouS2Map = new HashMap<>();
        for (int i = 0; i < s1.length(); i++) {
            Character tempChar1 = s1.charAt(i);
            Character tempChar2 = s2.charAt(i);
            if (i % 2 == 0) {
                ouS1Map.put(tempChar1, ouS1Map.getOrDefault(tempChar1, 0) + 1);
                ouS2Map.put(tempChar2, ouS2Map.getOrDefault(tempChar2, 0) + 1);
            } else {
                jiS1Map.put(tempChar1, jiS1Map.getOrDefault(tempChar1, 0) + 1);
                jiS2Map.put(tempChar2, jiS2Map.getOrDefault(tempChar2, 0) + 1);
            }
        }
        for (Character key : ouS1Map.keySet()) {
            int cnt1 = ouS1Map.get(key);
            int cnt2 = ouS2Map.getOrDefault(key, 0);
            if (cnt1 != cnt2) {
                return false;
            }
        }
        for (Character key : jiS1Map.keySet()) {
            int cnt1 = jiS1Map.get(key);
            int cnt2 = jiS2Map.getOrDefault(key, 0);
            if (cnt1 != cnt2) {
                return false;
            }
        }
        return true;
    }
}
```

# 2843.统计对称整数的数目

```java
class Solution {
    public int countSymmetricIntegers(int low, int high) {
        int result = 0;
        for (int i = low; i <= high; i++) {
            char[] charInt = (String.valueOf(i)).toCharArray();
            if (charInt.length % 2 == 0) {
                int firstSum = 0;
                int lastSum = 0;
                for (int j = 0; j < charInt.length / 2; j++) {
                    firstSum += charInt[j] - '0';
                }
                for (int j = charInt.length / 2; j < charInt.length; j++) {
                    lastSum += charInt[j] - '0';
                }
                if (firstSum == lastSum) {
                    result++;
                }
            }
        }
        return result;
    }
}
```

```java
    public int minimumRightShifts(List<Integer> nums) {
        int result = -1;
        for (int i = 0; i < nums.size() - 1; i++) {
            if (nums.get(i) > nums.get(i + 1)) {
                if (result == -1) {
                    result = i;
                } else {
                    return -1;
                }
            }
        }
```

```java
        for (int i = 0; i < nums.size(); i++) {
```

```java
        Set<Integer> tempSet = new HashSet<>();
        int i = nums.size() - 1;
```

# 2873.有序三元组中的最大值Ⅰ

```java
class Solution {
    public long maximumTripletValue(int[] nums) {
        long result = Long.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                for (int k = j + 1; k < nums.length; k++) {
                    long temp = (long) (nums[i] - nums[j]) * nums[k];
                    result = Math.max(temp, result);
                }
            }
        }
        return result < 0 ? 0 : result;
    }
}
```

```java
class Solution {

        int num1 = 0;
        int num2 = 0;
        for (int i = 1; i <= n; i++) {
```

# 2899.上一个遍历的整数

```java
class Solution {
    public List<Integer> lastVisitedIntegers(int[] nums) {
        List<Integer> ans = new ArrayList<>();
        List<Integer> seen = new ArrayList<>();
        int k = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != -1) {
                k = 0;
                seen.add(0, nums[i]);
            } else {
                k++;
                if (k <= seen.size()) {
                    ans.add(seen.get(k - 1));
                } else {
                    ans.add(-1);
                }
            }
        }
        return ans;
    }
}
```

```java
class Solution {
    public List<String> getLongestSubsequence(String[] words, int[] groups) {
        List<String> result = new ArrayList<>();
        result.add(words[0]);
        for (int i = 1; i < words.length; i++) {

                result.add(words[i]);

        }
```
