# 2824.统计和小于目标的下标对数目

```java
class Solution {
    public int countPairs(List<Integer> nums, int target) {
        Collections.sort(nums);
        int result = 0;
        for (int i = 0; i < nums.size(); i++) {
            int temp1 = nums.get(i);
            for (int j = i + 1; j < nums.size(); j++) {
                int temp2 = nums.get(j);
                if (temp1 + temp2 < target) {
                    result++;
                } else {
                    break;
                }
            }
        }
        return result;
    }
}
```

# 2833.距离原点最远的点

```java
class Solution {
    public int furthestDistanceFromOrigin(String moves) {
        int index = 0;
        int cnt = 0;
        for (int i = 0; i < moves.length(); i++) {
            char tempChar = moves.charAt(i);
            if (tempChar == 'L') {
                index--;
            } else if (tempChar == 'R') {
                index++;
            } else {
                cnt++;
            }
        }
        return Math.abs(index) + cnt;
    }
}
```



# 2843.统计对称整数的数目

```java
class Solution {
    public int countSymmetricIntegers(int low, int high) {
        int result = 0;
        for (int i = low; i <= high; i++) {
            char[] charInt = (String.valueOf(i)).toCharArray();
            if (charInt.length % 2 == 0) {
                int firstSum = 0;
                int lastSum = 0;
                for (int j = 0; j < charInt.length / 2; j++) {
                    firstSum += charInt[j] - '0';
                }
                for (int j = charInt.length / 2; j < charInt.length; j++) {
                    lastSum += charInt[j] - '0';
                }
                if (firstSum == lastSum) {
                    result++;
                }
            }
        }
        return result;
    }
}
```

# 2859.计算K置位下标对应元素的和

```java
class Solution {
    public int sumIndicesWithKSetBits(List<Integer> nums, int k) {
        int result = 0;
        for (int i = 0; i < nums.size(); i++) {
            int cnt = 0;
            for (int j = 31; j >= 0; j--) {
                if (((i >> j) & 1) == 1) {
                    cnt++;
                }
            }
            if (cnt == k) {
                result += nums.get(i);
            }
        }
        return result;
    }
}
```



# 2864.最大二进制奇数

```java
class Solution {
    public String maximumOddBinaryNumber(String s) {
        char[] c = s.toCharArray();
        int num = 0;
        for (int i = 0; i < c.length; i++) {
            if (c[i] == '1') {
                num++;
            }
        }
        String result = "";
        for (int i = 0; i < c.length; i++) {
            if (i == c.length - 1) {
                result += "1";
                return result;
            }
            if (num > 1) {
                result += "1";
                num--;
            } else {
                result += "0";
            }
        }
        return result;
    }
}
```

# 2869.收集元素的最少操作次数

```java
class Solution {
    public int minOperations(List<Integer> nums, int k) {
        Set<Integer> tempSet = new HashSet<>();
        int i = nums.size() - 1;
        for (; i >= 0; i--) {
            if (nums.get(i) <= k) {
                tempSet.add(nums.get(i));
            }
            if (tempSet.size() == k) {
                break;
            }
        }
        return nums.size() - i;
    }
}
```



# 2900.最长相邻不相等子序列Ⅰ

```java
class Solution {
    public List<String> getLongestSubsequence(String[] words, int[] groups) {
        List<String> result = new ArrayList<>();
        result.add(words[0]);
        for (int i = 1; i < words.length; i++) {
            if (groups[i] != groups[i - 1]) {
                result.add(words[i]);
            }
        }
        return result;
    }
}
```
