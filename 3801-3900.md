```java
        int result = 0;
```

```java
class Solution {
    public long countPairs(String[] words) {
        Map<String, Integer> tempMap = new HashMap<>();
        for (int i = 0; i < words.length; i++) {
            String tempStr = words[i];
            int tempIndex = tempStr.charAt(0) - 'a';
            StringBuilder sb = new StringBuilder();
            for (int j = 0; j < tempStr.length(); j++) {
                sb.append((tempStr.charAt(j) - 'a' - tempIndex + 26) % 26).append(",");
            }
            String tempResStr = sb.toString();
            tempMap.put(tempResStr, tempMap.getOrDefault(tempResStr, 0) + 1);
        }
        long result = 0;
        for (String key : tempMap.keySet()) {
            int cnt = tempMap.get(key);
            result += (long) cnt * (cnt - 1) / 2;
        }
        return result;
    }
}
```

```java
        yuanSet.add('a');
        yuanSet.add('e');
        yuanSet.add('i');
        yuanSet.add('o');
```

# 3823.反转一个字符串里的字母后反转特殊字符

```java
class Solution {
    public String reverseByType(String s) {
        char[] sChar = s.toCharArray();
        List<Character> letterList = new ArrayList<>();
        List<Integer> letterIndexList = new ArrayList<>();
        List<Character> dotList = new ArrayList<>();
        List<Integer> dotIndexList = new ArrayList<>();
        for (int i = 0; i < sChar.length; i++) {
            if (Character.isLetter(sChar[i])) {
                letterList.add(sChar[i]);
                letterIndexList.add(i);
            } else {
                dotList.add(sChar[i]);
                dotIndexList.add(i);
            }
        }
        for (int i = 0; i < letterList.size(); i++) {
            sChar[letterIndexList.get(i)] = letterList.get(letterIndexList.size() - 1 - i);
        }
        for (int i = 0; i < dotList.size(); i++) {
            sChar[dotIndexList.get(i)] = dotList.get(dotIndexList.size() - 1 - i);
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < sChar.length; i++) {
            sb.append(sChar[i]);
        }
        return sb.toString();
    }
}
```

```java
        int sum = 0;

        int cnt = nums.length;
```

```java
class Solution {
    public List<Integer> toggleLightBulbs(List<Integer> bulbs) {
        Set<Integer> blubsSet = new HashSet<>();
        List<Integer> blubsList = new ArrayList<>();
        for (int i = 0; i < bulbs.size(); i++) {
            Integer blub = bulbs.get(i);
            if (!blubsSet.add(blub)) {
                blubsSet.remove(blub);
            }
        }
        for (Integer tempBlub : blubsSet) {
            blubsList.add(tempBlub);
        }
        blubsList.sort((s1, s2) -> s1.compareTo(s2));
        return blubsList;
    }
}
```

```java
            tempInt[nums[i]]++;
        }

            }
```

# 3862.找出最小平衡下标

```java
class Solution {
    public int smallestBalancedIndex(int[] nums) {
        long[] left = new long[nums.length];
        long right = 1L;
        left[0] = nums[0];
        for (int i = 1; i < nums.length; i++) {
            left[i] = nums[i] + left[i - 1];
        }
        for (int i = nums.length - 1; i >= 0; i--) {
            if (i == 0) {
                if (right == 0) {
                    return i;
                } else {
                    break;
                }
            }
            if (left[i - 1] == right) {
                return i;
            }
            right *= nums[i];
            if (right > Long.MAX_VALUE || right < 0) {
                right = -1;
            }
        }
        return -1;
    }
}
```

```java
        Map<Integer, Integer> indexMap = new HashMap<>();
        Set<Integer> numberSet = new HashSet<>();
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] % 2 == 1) {
                continue;
            }

                indexMap.put(nums[i], i);
            } else {
                indexMap.remove(nums[i]);
            }

            return -1;
        }

            if (index < minIndex) {
                minIndex = index;
                result = key;
            }
        }
        return result;
    }
}
```

# 3868.通过交换使数组相等的最小花费

```java
class Solution {
    public int minCost(int[] nums1, int[] nums2) {
        Map<Integer, Integer> map1 = new HashMap<>();
        Map<Integer, Integer> map2 = new HashMap<>();
        Set<Integer> allNumSet = new HashSet<>();
        for (int i = 0; i < nums1.length; i++) {
            map1.put(nums1[i], map1.getOrDefault(nums1[i], 0) + 1);
            map2.put(nums2[i], map2.getOrDefault(nums2[i], 0) + 1);
            allNumSet.add(nums1[i]);
            allNumSet.add(nums2[i]);
        }
        int result = 0;
        for (Integer key : allNumSet) {
            int cnt1 = map1.getOrDefault(key, 0);
            int cnt2 = map2.getOrDefault(key, 0);
            if ((cnt1 + cnt2) % 2 == 1) {
                return -1;
            }
            result += Math.abs(cnt1 - cnt2) / 2;
        }
        return result / 2;
    }
}
```
