# 905.按奇偶排序数组

```java
class Solution {
    public int[] sortArrayByParity(int[] nums) {
        int len = nums.length;
        int[] result = new int[len];
        int left = 0;
        int right = len - 1;
        for (int i = 0; i < len; i++) {
            if (nums[i] % 2 == 0) {
                result[left] = nums[i];
                left++;
            } else {
                result[right] = nums[i];
                right--;
            }
        }
        return result;
    }
}
```

```java
class Solution {
    public int[] sortArray(int[] nums) {
        quickSort(nums, 0, nums.length - 1);
        return nums;
    }
    public int partition(int[] nums, int left, int right) {
        int mid = new Random().nextInt(right - left + 1) + left;
        swap(nums, left, mid);
        int j = left + 1;
        for (int i = left + 1; i <= right; i++) {
            if (nums[i] < nums[left]) {
                swap(nums, i, j);
                j++;
            }
        }
        swap(nums, j - 1, left);
        return j - 1;
    }
    public void quickSort(int[] nums, int left, int right) {
        if (left < right) {
            int mid = partition(nums, left, right);
            quickSort(nums, left, mid - 1);
            quickSort(nums, mid + 1, right);
        }
    }
    public void swap(int[] nums, int a, int b) {
        int temp = nums[a];
        nums[a] = nums[b];
        nums[b] = temp;
    }
}
```

```java
class Solution {
    public int[] sortArray(int[] nums) {
        heapSort(nums);
        return nums;
    }
    public void swap(int[] nums, int left, int right) {
        int temp = nums[left];
        nums[left] = nums[right];
        nums[right] = temp;
    }
    public void maxHeapify(int[] nums, int left, int right) {
        int son = left * 2;
        while (son <= right) {
            if (son + 1 <= right && nums[son + 1] > nums[son]) {
                son++;
            }
            if (nums[son] > nums[left]) {
                swap(nums, son, left);
                left = son;
                son = son * 2;
            } else {
                break;
            }
        }
    }
    public void heapSort(int[] nums) {
        int len = nums.length - 1;
        for (int i = len / 2; i >= 0; i--) {
            maxHeapify(nums, i, len);
        }
        for (int i = len; i > 0; i--) {
            swap(nums, 0, i);
            maxHeapify(nums, 0, --len);
        }
    }
}
```

```java
class Solution {
    int[] prefix;
    public int numSubarraysWithSum(int[] nums, int goal) {
        int len = nums.length;
        prefix = new int[len];
        prefix[0] = nums[0];
        for (int i = 1; i < len; i++) {
            prefix[i] = prefix[i - 1] + nums[i];
        }
        int sum = 0;
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 1);
        for (int i = 0; i < len; i++) {
            int count = map.getOrDefault(prefix[i] - goal, 0);
            sum += count;
            map.put(prefix[i], map.getOrDefault(prefix[i], 0) + 1);
        }
        return sum;
    }
    public int getPrefixSum(int index) {
        if (index == -1) {
            return 0;
        }
        return prefix[index];
    }
}
```

# 938.二叉搜索树的范围和

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int result;
    public int rangeSumBST(TreeNode root, int low, int high) {
        result = 0;
        sum(root, low, high);
        return result;
    }
    public void sum(TreeNode node, int low, int high) {
        if (node == null) {
            return;
        }
        if(node.val >= low &&node.val <= high){
            result += node.val;
        }
        sum(node.left, low, high);
        sum(node.right, low, high);
    }
}
```

# 945.使数组唯一的最小增量

```java
class Solution {
    public int minIncrementForUnique(int[] nums) {
        Arrays.sort(nums);
        int len = nums.length;
        int move = 0;
        for (int i = 1; i < len; i++) {
            if (nums[i] <= nums[i - 1]) {
                move += nums[i - 1] - nums[i] + 1;
                nums[i] = nums[i - 1] + 1;
            }
        }
        return move;
    }
}
```

# 965.单值二叉树

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isUnivalTree(TreeNode root) {
        return isTrue(root, root.val);
    }

    public boolean isTrue(TreeNode node, int value) {
        if (node == null) {
            return true;
        }
        if (node.val != value) {
            return false;
        }
        return isTrue(node.left, value) && isTrue(node.right, value);
    }
}
```

# 974.和可被K整除的子数组

```java
class Solution {
    public int subarraysDivByK(int[] nums, int k) {
        int len = nums.length;
        int[] prefix = new int[len];
        prefix[0] = nums[0] % k;
        if (prefix[0] < 0) {
            prefix[0] += k;
        }
        for (int i = 1; i < len; i++) {
            prefix[i] = (prefix[i - 1] + nums[i]) % k;
            if (prefix[i] < 0) {
                prefix[i] += k;
            }
        }
        int result = 0;
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 1);
        for (int i = 0; i < len; i++) {
            int number = map.getOrDefault(prefix[i], 0);
            result += number;
            map.put(prefix[i], number + 1);
        }
        return result;
    }
}
```

# 976.三角形的最大周长

```java
class Solution {
    public int largestPerimeter(int[] nums) {
        int len = nums.length;
        Arrays.sort(nums);
        for (int i = len - 1; i > 1; i--) {
            if (nums[i - 1] + nums[i - 2] > nums[i] && nums[i] - nums[i - 1] < nums[i - 2]) {
                return nums[i] + nums[i - 1] + nums[i - 2];
            }
        }
        return 0;
    }
}
```
