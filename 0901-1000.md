```java
class Solution {
    public int[] sortArrayByParity(int[] nums) {
        int len = nums.length;
        int[] result = new int[len];
        int left = 0;
        int right = len - 1;
        for (int i = 0; i < len; i++) {
            if (nums[i] % 2 == 0) {
                result[left] = nums[i];
                left++;
            } else {
                result[right] = nums[i];
                right--;
            }
        }
        return result;
    }
}
```

```java
class Solution {
    public int[] sortArray(int[] nums) {
        quickSort(nums, 0, nums.length - 1);
        return nums;
    }
    public int partition(int[] nums, int left, int right) {
        int mid = new Random().nextInt(right - left + 1) + left;
        swap(nums, left, mid);
        int j = left + 1;
        for (int i = left + 1; i <= right; i++) {
            if (nums[i] < nums[left]) {
                swap(nums, i, j);
                j++;
            }
        }
        swap(nums, j - 1, left);
        return j - 1;
    }
    public void quickSort(int[] nums, int left, int right) {
        if (left < right) {
            int mid = partition(nums, left, right);
            quickSort(nums, left, mid - 1);
            quickSort(nums, mid + 1, right);
        }
    }
    public void swap(int[] nums, int a, int b) {
        int temp = nums[a];
        nums[a] = nums[b];
        nums[b] = temp;
    }
}
```

```java
class Solution {
    public int[] sortArray(int[] nums) {
        heapSort(nums);
        return nums;
    }
    public void swap(int[] nums, int left, int right) {
        int temp = nums[left];
        nums[left] = nums[right];
        nums[right] = temp;
    }
    public void maxHeapify(int[] nums, int left, int right) {
        int son = left * 2;
        while (son <= right) {
            if (son + 1 <= right && nums[son + 1] > nums[son]) {
                son++;
            }
            if (nums[son] > nums[left]) {
                swap(nums, son, left);
                left = son;
                son = son * 2;
            } else {
                break;
            }
        }
    }
    public void heapSort(int[] nums) {
        int len = nums.length - 1;
        for (int i = len / 2; i >= 0; i--) {
            maxHeapify(nums, i, len);
        }
        for (int i = len; i > 0; i--) {
            swap(nums, 0, i);
            maxHeapify(nums, 0, --len);
        }
    }
}
```

```java
class Solution {
    public boolean hasGroupsSizeX(int[] deck) {
        Map<Integer, Integer> tempMap = new HashMap<>();
        for (int i = 0; i < deck.length; i++) {
            tempMap.put(deck[i], tempMap.getOrDefault(deck[i], 0) + 1);
        }
        int gys = -1;
        for (Integer key : tempMap.keySet()) {
            if (gys == -1) {
                gys = tempMap.get(key);
            } else {
                gys = getGys(gys, tempMap.get(key));
            }
        }
        return gys >= 2;
    }

    public int getGys(int a, int b) {
        return a == 0 ? b : getGys(b % a, a);
    }
}
```

```java
class Solution {
    public String reverseOnlyLetters(String s) {
        char[] sChar = s.toCharArray();
        int left = 0;
        int right = sChar.length - 1;
        while (left < sChar.length && !Character.isLetter(sChar[left])) {
            left++;
        }
        while (right >= 0 && !Character.isLetter(sChar[right])) {
            right--;
        }
        while (left < right) {
            char temp = sChar[left];
            sChar[left] = sChar[right];
            sChar[right] = temp;
            left++;
            right--;
            while (!Character.isLetter(sChar[left])) {
                left++;
            }
            while (!Character.isLetter(sChar[right])) {
                right--;
            }
        }
        return new String(sChar);
    }
}
```

```java
class Solution {
    public boolean isLongPressedName(String name, String typed) {
        char[] nameChar = name.toCharArray();
        char[] typedChar = typed.toCharArray();
        int nameIndex = 0;
        int typedIndex = 0;
        while (typedIndex < typedChar.length) {
            if (nameIndex < nameChar.length && nameChar[nameIndex] == typedChar[typedIndex]) {
                nameIndex++;
                typedIndex++;
            } else if (nameIndex > 0 && nameChar[nameIndex - 1] == typedChar[typedIndex]) {
                typedIndex++;
            } else {
                return false;
            }
        }
        return nameIndex == nameChar.length;
    }
}
```

```java
class Solution {
    public int numUniqueEmails(String[] emails) {
        Set<String> resultSet = new HashSet<>();
        for (int i = 0; i < emails.length; i++) {
            char[] tempEmail = emails[i].toCharArray();
            StringBuilder sb = new StringBuilder();
            boolean isDisable = false;
            boolean isAt = false;
            for (int j = 0; j < tempEmail.length; j++) {
                char tempChar = tempEmail[j];
                if (tempChar == '+') {
                    isDisable = true;
                }
                if (tempChar == '@') {
                    isDisable = false;
                    isAt = true;
                }
                if ((isDisable && !isAt) || (tempChar == '.' && !isAt)) {
                    continue;
                }
                sb.append(tempChar);
            }
            resultSet.add(sb.toString());
        }
        return resultSet.size();
    }
}
```

```java
    public int numSubarraysWithSum(int[] nums, int goal) {
        int len = nums.length;
        prefix = new int[len];
        prefix[0] = nums[0];
        for (int i = 1; i < len; i++) {
            prefix[i] = prefix[i - 1] + nums[i];
        }
        int sum = 0;
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 1);
        for (int i = 0; i < len; i++) {
            int count = map.getOrDefault(prefix[i] - goal, 0);
            sum += count;
            map.put(prefix[i], map.getOrDefault(prefix[i], 0) + 1);
        }
        return sum;
    }
    public int getPrefixSum(int index) {
        if (index == -1) {
            return 0;
        }
        return prefix[index];
    }
}
```

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int result;
    public int rangeSumBST(TreeNode root, int low, int high) {
        result = 0;
        sum(root, low, high);
        return result;
    }
    public void sum(TreeNode node, int low, int high) {
        if (node == null) {
            return;
        }
        if(node.val >= low &&node.val <= high){
            result += node.val;
        }
        sum(node.left, low, high);
        sum(node.right, low, high);
    }
}
```

```java
class Solution {
```

```java
        int len = nums.length;
        int move = 0;
        for (int i = 1; i < len; i++) {

        }
```

```java
class Solution {
    public boolean isAlienSorted(String[] words, String order) {
        int[] index = new int[26];
        for (int i = 0; i < order.length(); ++i) {
            index[order.charAt(i) - 'a'] = i;
        }
        for (int i = 1; i < words.length; i++) {
            boolean valid = false;
            for (int j = 0; j < words[i - 1].length() && j < words[i].length(); j++) {
                int prev = index[words[i - 1].charAt(j) - 'a'];
                int curr = index[words[i].charAt(j) - 'a'];
                if (prev < curr) {
                    valid = true;
                    break;
                } else if (prev > curr) {
                    return false;
                }
            }
            if (!valid) {
                /* 比较两个字符串的长度 */
                if (words[i - 1].length() > words[i].length()) {
                    return false;
                }
            }
        }
        return true;
    }
}
```

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isUnivalTree(TreeNode root) {
        return isTrue(root, root.val);
    }

    public boolean isTrue(TreeNode node, int value) {
        if (node == null) {
            return true;
        }
        if (node.val != value) {
            return false;
        }
        return isTrue(node.left, value) && isTrue(node.right, value);
    }
}
```

```java
class Solution {
    public int subarraysDivByK(int[] nums, int k) {
        int len = nums.length;
        int[] prefix = new int[len];
        prefix[0] = nums[0] % k;
        if (prefix[0] < 0) {
            prefix[0] += k;
        }
        for (int i = 1; i < len; i++) {
            prefix[i] = (prefix[i - 1] + nums[i]) % k;
            if (prefix[i] < 0) {
                prefix[i] += k;
            }
        }
        int result = 0;
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, 1);

            int number = map.getOrDefault(prefix[i], 0);
            result += number;
            map.put(prefix[i], number + 1);
        }
        return result;
    }
}
```

```java
class Solution {
    public int findJudge(int n, int[][] trust) {
        Set<Integer> peopleSet = new HashSet<>();
        Map<Integer, Integer> resultMap = new HashMap<>();
        if (trust.length == 0 && n == 1) {
            return 1;
        }
        for (int i = 0; i < trust.length; i++) {
            int people = trust[i][0];
            int order = trust[i][1];
            peopleSet.add(people);
            resultMap.put(order, resultMap.getOrDefault(order, 0) + 1);
        }
        for (Integer key : resultMap.keySet()) {
            if (resultMap.get(key) == n - 1 && !peopleSet.contains(key)) {
                return key;
            }
        }
        return -1;
    }
}
```

```java
        int RIndexY = -1;

                break;
```
