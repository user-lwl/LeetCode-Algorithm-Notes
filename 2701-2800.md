```java
class Solution {

        int index = num.length();
        for (int i = num.length() - 1; i >= 0; i--) {
            if ('0' != num.charAt(i)) {
                index = i;

            }
        }

    }
```

# 2717.半有序排列

```java
class Solution {
    public int semiOrderedPermutation(int[] nums) {
        int one = -1;
        int n = -1;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 1) {
                one = i;
            } else if (nums[i] == nums.length) {
                n = i;
            }
        }
        return nums.length - 1 - n + one - (one > n ? 1 : 0);
    }
}
```

# 2730.找到最长的半重复子字符串

```java
class Solution {
    public int longestSemiRepetitiveSubstring(String s) {
        int left = 0;
        int right = 0;
        int fixCount = 0;
        int max = 0;
        char[] c = s.toCharArray();
        int len = c.length;
        while (right < len) {
            if (right > 0 && c[right] == c[right - 1]) {
                fixCount++;
            }
            while (fixCount > 1) {
                left++;
                if (left > 0 && c[left] == c[left - 1]) {
                    fixCount--;
                }
            }
            max = Math.max(max, right - left + 1);
            right++;
        }
        return max;
    }
}
```

# 2760.最长奇偶子数组

```java
class Solution {
    public int longestAlternatingSubarray(int[] nums, int threshold) {
        int result = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] % 2 != 0 || nums[i] > threshold) {
                continue;
            }
            int len = 1;
            for (int j = i + 1; j < nums.length; j++) {
                if (nums[j - 1] % 2 == nums[j] % 2 || nums[j] > threshold) {
                    break;
                }
                len++;
            }
            result = Math.max(result, len);
        }
        return result;
    }
}
```



# 2765.最长交替子数组

```java
class Solution {
    public int alternatingSubarray(int[] nums) {
        int result = -1;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] == nums[i + 1] - 1) {
                int startIndex = i;
                i += 2;
                while (i < nums.length && nums[i] == nums[i - 2]) {
                    i++;
                }
                int endIndex = i - 1;
                i -= 2;
                result = Math.max(result, endIndex - startIndex + 1);
            }
        }
        return result;
    }
}
```

# 2778.特殊元素平方和

```java
class Solution {
    public int sumOfSquares(int[] nums) {
        int n = nums.length;
        int result = 0;
        for (int i = 0; i < nums.length; i++) {
            if (n % (i + 1) == 0) {
                result += nums[i] * nums[i];
            }
        }
        return result;
    }
}
```

```java
class Solution {
    public boolean isGood(int[] nums) {
        Arrays.sort(nums);
        if (nums.length == 1) {
            return false;
        }
        for (int i = 0; i < nums.length; i++) {
            if (i == nums.length - 1) {
                if (nums[i] != nums[i - 1]) {
                    return false;
                }
            } else {
                if (nums[i] != i + 1) {
                    return false;
                }
            }
        }
        return true;
    }
}
```

# 2785.将字符串中的元音字母排序

```java
class Solution {
    public String sortVowels(String s) {
        char[] tempChar = s.toCharArray();
        List<Character> letterList = new ArrayList<>();
        List<Integer> indexList = new ArrayList<>();
        Set<Character> yuanSet = new HashSet<>();
        yuanSet.add('a');
        yuanSet.add('e');
        yuanSet.add('i');
        yuanSet.add('o');
        yuanSet.add('u');
        yuanSet.add('A');
        yuanSet.add('E');
        yuanSet.add('I');
        yuanSet.add('O');
        yuanSet.add('U');
        for (int i = 0; i < tempChar.length; i++) {
            if (yuanSet.contains(tempChar[i])) {
                letterList.add(tempChar[i]);
                indexList.add(i);
            }
        }
        Collections.sort(letterList);
        for (int i = 0; i < letterList.size(); i++) {
            tempChar[indexList.get(i)] = letterList.get(i);
        }
        return new String(tempChar);
    }
}
```

# 2788.按分隔符拆分字符串

```java
class Solution {
    public List<String> splitWordsBySeparator(List<String> words, char separator) {
        List<String> resultList = new ArrayList<>();
        for (int i = 0; i < words.size(); i++) {
            String[] tempList = words.get(i).split("\\" + separator);
            for (int j = 0; j < tempList.length; j++) {
                if (tempList[j].length() == 0) {
                    continue;
                }
                resultList.add(tempList[j]);
            }
        }
        return resultList;
    }
}
```

# 2789.合并后数组中的最大元素

```java
class Solution {
    public long maxArrayValue(int[] nums) {
        long sum = nums[nums.length - 1];
        for (int i = nums.length - 2; i >= 0; i--) {
```
