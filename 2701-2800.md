# 2706.购买两块巧克力

```java
class Solution {
    public int buyChoco(int[] prices, int money) {
        Arrays.sort(prices);
        if (prices[0] + prices[1] <= money) {
            return money - prices[0] - prices[1];
        }
        return money;
    }
}
```

# 2730.找到最长的半重复子字符串

```java
class Solution {
    public int longestSemiRepetitiveSubstring(String s) {
        int left = 0;
        int right = 0;
        int fixCount = 0;
        int max = 0;
        char[] c = s.toCharArray();
        int len = c.length;
        while (right < len) {
            if (right > 0 && c[right] == c[right - 1]) {
                fixCount++;
            }
            while (fixCount > 1) {
                left++;
                if (left > 0 && c[left] == c[left - 1]) {
                    fixCount--;
                }
            }
            max = Math.max(max, right - left + 1);
            right++;
        }
        return max;
    }
}
```

```java
        int[] cnt = new int[101];

            cnt[nums[i]]++;
        }

        for (int i = 0; i < cnt.length; i++) {
```

# 2748.美丽下标对的数目

```java
class Solution {
    public int countBeautifulPairs(int[] nums) {
        int result = 0;
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {
                int a = String.valueOf(nums[i]).charAt(0) - '0';
                int b = nums[j] % 10;
                if (gcd(a, b) == 1) {
                    result++;
                }
            }
        }
        return result;
    }
    public int gcd (int a, int b) {
        if (b == 0) {
            return a;
        } else {
            return gcd(b, a % b);
        }
    }
}
```

# 2760.最长奇偶子数组

```java
class Solution {
    public int longestAlternatingSubarray(int[] nums, int threshold) {
        int result = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] % 2 != 0 || nums[i] > threshold) {
                continue;
            }
            int len = 1;
            for (int j = i + 1; j < nums.length; j++) {
                if (nums[j - 1] % 2 == nums[j] % 2 || nums[j] > threshold) {
                    break;
                }
                len++;
            }
            result = Math.max(result, len);
        }
        return result;
    }
}
```



# 2765.最长交替子数组

```java
class Solution {
    public int alternatingSubarray(int[] nums) {
        int result = -1;
        for (int i = 0; i < nums.length - 1; i++) {
            if (nums[i] == nums[i + 1] - 1) {
                int startIndex = i;
                i += 2;
                while (i < nums.length && nums[i] == nums[i - 2]) {
                    i++;
                }
                int endIndex = i - 1;
                i -= 2;
                result = Math.max(result, endIndex - startIndex + 1);
            }
        }
        return result;
    }
}
```

# 2778.特殊元素平方和

```java
class Solution {
    public int sumOfSquares(int[] nums) {
        int n = nums.length;
        int result = 0;
        for (int i = 0; i < nums.length; i++) {
            if (n % (i + 1) == 0) {
                result += nums[i] * nums[i];
            }
        }
        return result;
    }
}
```

# 2784.检查数组是否是好的

```java
class Solution {
    public boolean isGood(int[] nums) {
        Arrays.sort(nums);
        if (nums.length == 1) {
            return false;
        }
        for (int i = 0; i < nums.length; i++) {
            if (i == nums.length - 1) {
                if (nums[i] != nums[i - 1]) {
                    return false;
                }
            } else {
                if (nums[i] != i + 1) {
                    return false;
                }
            }
        }
        return true;
    }
}
```

# 2785.将字符串中的元音字母排序

```java
class Solution {
    public String sortVowels(String s) {
        char[] tempChar = s.toCharArray();
        List<Character> letterList = new ArrayList<>();
        List<Integer> indexList = new ArrayList<>();
        Set<Character> yuanSet = new HashSet<>();
        yuanSet.add('a');
        yuanSet.add('e');
        yuanSet.add('i');
        yuanSet.add('o');
        yuanSet.add('u');
        yuanSet.add('A');
        yuanSet.add('E');
        yuanSet.add('I');
        yuanSet.add('O');
        yuanSet.add('U');
        for (int i = 0; i < tempChar.length; i++) {
            if (yuanSet.contains(tempChar[i])) {
                letterList.add(tempChar[i]);
                indexList.add(i);
            }
        }
        Collections.sort(letterList);
        for (int i = 0; i < letterList.size(); i++) {
            tempChar[indexList.get(i)] = letterList.get(i);
        }
        return new String(tempChar);
    }
}
```



# 2789.合并后数组中的最大元素

```java
class Solution {
    public long maxArrayValue(int[] nums) {
        long sum = nums[nums.length - 1];
        for (int i = nums.length - 2; i >= 0; i--) {
            sum = nums[i] <= sum ? nums[i] + sum : nums[i];
        }
        return sum;
    }
}
```
