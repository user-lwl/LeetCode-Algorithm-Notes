# 310.最小高度树

```java
class Solution {
    public List<Integer> findMinHeightTrees(int n, int[][] edges) {
        List<Integer> result = new ArrayList<>();
        if (n == 1) {
            result.add(0);
            return result;
        }
        List<Integer>[] adj= new List[n];
        for (int i = 0; i < n; i++) {
            adj[i] = new ArrayList<>();
        }
        for (int[] edge : edges) {
            adj[edge[0]].add(edge[1]);
            adj[edge[1]].add(edge[0]);
        }
        int[] parent = new int[n];
        Arrays.fill(parent, -1);
        int x = findLongNode(0, parent, adj);
        int y = findLongNode(x, parent, adj);
        List<Integer> path = new ArrayList<>();
        parent[x] = -1;
        while (y != -1) {
            path.add(y);
            y = parent[y];
        }
        int size = path.size();
        if (size % 2 == 0) {
            result.add(path.get(size / 2 - 1));
        }
        result.add(path.get(size / 2));
        return result;
    }
    public int findLongNode(int n, int[] parent, List<Integer>[] adj) {
        int len = adj.length;
        Queue<Integer> queue = new ArrayDeque<>();
        boolean[] visit = new boolean[len];
        queue.offer(n);
        visit[n] = true;
        int node = -1;
        while (!queue.isEmpty()) {
            node = queue.poll();
            for (int ad : adj[node]) {
                if (!visit[ad]) {
                    visit[ad] = true;
                    parent[ad] = node;
                    queue.offer(ad);
                }
            }
        }
        return node;
    }
}
```

# 322.零钱兑换

```java
class Solution {
    public int coinChange(int[] coins, int amount) {
        int[] dp = new int[amount + 1];
        Arrays.fill(dp, 99999999);
        int len = coins.length;
        dp[0] = 0;
        for (int i = 0; i < len; i++) {
            for (int j = coins[i]; j <= amount; j++) {
                dp[j] = Math.min(dp[j], dp[j - coins[i]] + 1);
            }
        }
        return dp[amount] == 99999999 ? -1 : dp[amount];
    }
}
```

# 324.摆动序列Ⅱ

~~~java
class Solution {
    public void wiggleSort(int[] nums) {
        int len = nums.length;
        int[] temp = new int[len];
        for (int i = 0; i < len; i++) {
            temp[i] = nums[i];
        }
        Arrays.sort(temp);
        int right = len - 1;
        for (int i = 1; i < len; i+=2) {
            nums[i] = temp[right--];
        }
        for (int i = 0; i < len; i+=2) {
            nums[i] = temp[right--];
        }
    }
}
~~~

# 328.奇偶链表

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode oddEvenList(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode newHead1 = head;
        ListNode newHead2 = head.next;
        int index = 0;
        ListNode node = head.next.next;
        ListNode newNode1 = newHead1;
        ListNode newNode2 = newHead2;
        while (node != null) {
            ListNode tempNode = node;
            node = node.next;
            if (index++ % 2 == 0) {
                newNode1.next = tempNode;
                newNode1 = newNode1.next;
            } else {
                newNode2.next = tempNode;
                newNode2 = newNode2.next;
            }
        }
        newNode1.next = newHead2;
        newNode2.next = null;
        return newHead1;
    }
}
```

# 338.比特位计数

```java
class Solution {
    public int[] countBits(int n) {
        int[] result = new int[n + 1];
        result[0] = 0;
        int maxNumber = 0;
        for (int i = 1; i <= n; i++) {
            if (i == (maxNumber == 0 ? 1 : maxNumber) * 2) {
                maxNumber = i;
            }
            result[i] = result[i - maxNumber] + 1;
        }
        return result;
    }
}
```

# 347.前K个高频元素

```java
class Solution {
    public int[] topKFrequent(int[] nums, int k) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int num : nums) {
            map.put(num, map.getOrDefault(num, 0) + 1);
        }
        PriorityQueue<Integer> pQueue = new PriorityQueue<>(new Comparator<Integer>() {
            @Override
            public int compare(Integer n1, Integer n2) {
                return map.get(n1) - map.get(n2);
            }
        });
        for (Integer key : map.keySet()) {
            pQueue.offer(key);
            while (pQueue.size() > k) {
                pQueue.poll();
            }
        }
        int[] result = new int[k];
        for (int i = 0; i < k; i++) {
            result[i] = pQueue.poll();
        }
        return result;
    }
}
```

```java
class Solution {
    public boolean isPerfectSquare(int num) {
        int number = binarySearch(num, 0, num+1) - 1;
        return number * number == num;
    }
    public int binarySearch(int n, int left, int right) {
        while (left + 1 < right) {
            int mid = (left + right) / 2;
            if ((long)mid * (long)mid > (long)n) {
                right = mid;
            } else {
                left = mid;
            }
        }
        return right;
    }
}
```

```java
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }

class Solution {

    int[] list = new int[10001];;
    int size;

    public Solution(ListNode head) {
        size = 0;
        while (head != null) {
            list[size++] = head.val;
            head = head.next;
        }

        Random random = new Random();
```

```java
class Solution {
    public char findTheDifference(String s, String t) {
        char[] sChar = s.toCharArray();
        char[] tChar = t.toCharArray();
        int sLen = sChar.length;
        int tLen = tChar.length;
        int[] sx = new int[26];
        for (int i = 0; i < sx.length; i++) {
            sx[i] = 0;
        }
        for (int i = 0; i < tLen; i++) {
            sx[tChar[i] - 'a']++;
        }
        for (int i = 0; i < sLen; i++) {

                return (char) (i + (int)'a');
            }
        }
        return '.';
    }
}
```

# 392.判断子序列

```java
class Solution {
    public boolean isSubsequence(String s, String t) {
        if (s.length() == 0) {
            return true;
        }
        char[] cs = s.toCharArray();
        char[] ct = t.toCharArray();
        int left = 0;
        for (int i = 0; i < ct.length; i++) {
            if (ct[i] == cs[left]) {
                left++;
            }
            if (left == cs.length) {
                return true;
            }
        }
        return false;
    }
}
```

# 397.整数替换

```java
class Solution {
    public int integerReplacement(int n) {
        return findMinNumber(n, 0);
    }
    public int findMinNumber(long n, int count) {
        if (n == 1) {
            return count;
        }
        count++;
        if (n % 2 == 1) {
            return Math.min(findMinNumber(n + 1, count), findMinNumber(n - 1, count));
        } else {
            return findMinNumber(n / 2, count);
        }
    }
}
```
