# 2315.统计星号

```java
class Solution {
    public int countAsterisks(String s) {
        char[] c = s.toCharArray();
        int result = 0;
        boolean isTrue = true;
        for (int i = 0; i < c.length; i++){
            if (c[i] == '|') {
                isTrue = !isTrue;
                continue;
            }
            if (isTrue && c[i] == '*') {
                result++;
            }
        }
        return result;
    }
}
```

# 2331.计算布尔二叉树的值

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean evaluateTree(TreeNode root) {
        return getResult(root);
    }

    public boolean getResult(TreeNode node) {
        if (node.val == 0) {
            return false;
        } else if (node.val == 1) {
            return true;
        } else if (node.val == 2) {
            return getResult(node.left) || getResult(node.right);
        } else {
            return getResult(node.left) && getResult(node.right);
        }
    }
}
```

# 2335.装满被子需要的最短总时长

```java
class Solution {
    public int fillCups(int[] amount) {
        Arrays.sort(amount);
        int min = amount[0];
        int middle = amount[1];
        int max = amount[2];
        return max > (middle + min) ? max : (max + middle + min + 1) / 2;
    }
}
```

# 2363.合并相似的物品

```java
class Solution {
    public List<List<Integer>> mergeSimilarItems(int[][] items1, int[][] items2) {
        Map<Integer, Integer> resultMap = new HashMap<>();
        for (int i = 0; i < items1.length; i++) {
            Integer value = items1[i][0];
            Integer weight = items1[i][1];
            resultMap.put(value, weight);
        }
        for (int i = 0; i < items2.length; i++) {
            Integer value = items2[i][0];
            Integer weight = items2[i][1];
            resultMap.put(value, resultMap.getOrDefault(value, 0) + weight);
        }
        List<List<Integer>> result = new ArrayList<>();
        int[][] tempResult = new int[resultMap.size()][2];
        int index = 0;
        for (Integer key : resultMap.keySet()) {
            Integer value = key;
            Integer weight = resultMap.get(key);
            tempResult[index][0] = value;
            tempResult[index][1] = weight;
            index++;
        }
        Arrays.sort(tempResult, new Comparator<int[]>() {
            public int compare(int[] o1, int[] o2) {
                return o1[0] - o2[0];
            }
        });
        for (int i = 0; i < tempResult.length; i++) {
            List<Integer> tempList = new ArrayList<>();
            Integer value = tempResult[i][0];
            Integer weight = tempResult[i][1];
            tempList.add(value);
            tempList.add(weight);
            result.add(tempList);
        }
        return result;
    }
}
```

# 2367.算数三元组的数目

```java
class Solution {
    public int arithmeticTriplets(int[] nums, int diff) {
        int len = nums.length;
        int number = 0;
        for (int i = 0; i < len; i++) {
            for (int j = i + 1; j < len; j++) {
                if (nums[j] - nums[i] != diff) {
                    continue;
                }
                for (int k = j + 1; k < len; k++) {
                    if (nums[k] - nums[j] == diff) {
                        number++;
                    }
                }
            }
        }
        return number;
    }
}
```

# 2389.和有限的最长子序列

```java
class Solution {
    public int[] answerQueries(int[] nums, int[] queries) {
        Arrays.sort(nums);
        int[] sum = new int[nums.length];
        sum[0] = nums[0];
        for (int i = 1; i < sum.length; i++) {
            sum[i] = sum[i - 1] + nums[i];
        }
        int[] result = new int[queries.length];
        for (int i = 0; i < queries.length; i++) {
            for (int j = 0; j < sum.length; j++) {
                if (sum[j] > queries[i]) {
                    result[i] = j;
                    break;
                } else if (j == sum.length - 1 && sum[j] <= queries[i]) {
                    result[i] = j + 1;
                    break;
                }
            }
        }
        return result;
    }
}
```
