# 2303.计算应缴税款总额

```java
class Solution {
    public double calculateTax(int[][] brackets, int income) {
        int[][] tempBrackets = new int[brackets.length + 2][brackets.length + 2];
        for (int i = 0; i < brackets.length; i++) {
            tempBrackets[i + 1][0] = brackets[i][0];
            tempBrackets[i + 1][1] = brackets[i][1];
        }
        tempBrackets[tempBrackets.length - 1][0] = 9999;
        double result = 0;
        for (int i = 1; i < tempBrackets.length; i++) {
            int need = Math.min(income, tempBrackets[i][0] - tempBrackets[i - 1][0]);
            double fee = (double) need * tempBrackets[i][1] / 100;
            result += fee;
            income -= need;
            if (income == 0) {
                break;
            }
        }
        return result;
    }
}
```

# 2309.兼具大小写的最好英文字母

```java
class Solution {
    public String greatestLetter(String s) {
        Set<Character> bigSet = new HashSet<>();
        Set<Character> smallSet = new HashSet<>();
        for (int i = 0; i < s.length(); i++) {
            char temp = s.charAt(i);
            if (temp >= 'a' && temp <= 'z') {
                smallSet.add(temp);
            } else {
                bigSet.add(temp);
            }
        }
        char result = (char) 0;
        for (Character key : bigSet) {
            if (smallSet.contains((char) (key + ('a' - 'A')))) {
                result = (char) Math.max((int)key, result);
            }
        }
        return result == 0 ? "" : String.valueOf(result);
    }
}
```

```java
class Solution {
    public String decodeMessage(String key, String message) {
        Map<Character, Character> kvMap = new HashMap<>();
        char word = 'a';
        for (int i = 0; i < key.length(); i++) {
            char temp = key.charAt(i);
            if (temp == ' ') {
                continue;
            }
            if (!kvMap.containsKey(temp)) {
                kvMap.put(temp, word++);
            }
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < message.length(); i++) {
            char result = kvMap.getOrDefault(message.charAt(i), ' ');
            sb.append(result);
        }
        return sb.toString();
    }
```

# 2331.计算布尔二叉树的值

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean evaluateTree(TreeNode root) {
        return getResult(root);
    }

    public boolean getResult(TreeNode node) {
        if (node.val == 0) {
            return false;
        } else if (node.val == 1) {
            return true;
        } else if (node.val == 2) {
            return getResult(node.left) || getResult(node.right);
        } else {
            return getResult(node.left) && getResult(node.right);
        }
    }
}
```

```java
class Solution {
    public List<List<Integer>> mergeSimilarItems(int[][] items1, int[][] items2) {
        Map<Integer, Integer> resultMap = new HashMap<>();
        for (int i = 0; i < items1.length; i++) {
            Integer value = items1[i][0];
            Integer weight = items1[i][1];
            resultMap.put(value, weight);
        }
        for (int i = 0; i < items2.length; i++) {
            Integer value = items2[i][0];
            Integer weight = items2[i][1];
            resultMap.put(value, resultMap.getOrDefault(value, 0) + weight);
        }
        List<List<Integer>> result = new ArrayList<>();
        int[][] tempResult = new int[resultMap.size()][2];
        int index = 0;
        for (Integer key : resultMap.keySet()) {
            Integer value = key;
            Integer weight = resultMap.get(key);
            tempResult[index][0] = value;
            tempResult[index][1] = weight;
            index++;
        }
        Arrays.sort(tempResult, new Comparator<int[]>() {
            public int compare(int[] o1, int[] o2) {
                return o1[0] - o2[0];
            }
        });
        for (int i = 0; i < tempResult.length; i++) {
            List<Integer> tempList = new ArrayList<>();
            Integer value = tempResult[i][0];
            Integer weight = tempResult[i][1];
            tempList.add(value);
            tempList.add(weight);
            result.add(tempList);
        }
        return result;
    }
```

# 2367.算数三元组的数目

```java
class Solution {
    public int arithmeticTriplets(int[] nums, int diff) {
        int len = nums.length;
        int number = 0;
        for (int i = 0; i < len; i++) {
            for (int j = i + 1; j < len; j++) {
                if (nums[j] - nums[i] != diff) {
                    continue;
                }
                for (int k = j + 1; k < len; k++) {
                    if (nums[k] - nums[j] == diff) {
                        number++;
                    }
                }
            }
        }
        return number;
    }
}
```

# 2379.得到K个黑块的最少涂色次数

```java
class Solution {
    public int minimumRecolors(String blocks, int k) {
        char[] blocksCharList = blocks.toCharArray();
        int left = 0;
        int right = k - 1;
        int cnt = 0;
        for (int i = left; i <= right; i++) {
            if (blocksCharList[i] == 'W') {
                cnt++;
            }
        }
        int result = cnt;
        left++;
        right++;
        while (right < blocksCharList.length) {
            if (blocksCharList[left - 1] == 'W') {
                cnt--;
            }
            if (blocksCharList[right] == 'W') {
                cnt++;
            }
            result = Math.min(result, cnt);
            left++;
            right++;
        }
        return result;
    }
}
```

# 2383.赢得比赛需要的最少训练时长

```java
class Solution {
    public int minNumberOfHours(int initialEnergy, int initialExperience, int[] energy, int[] experience) {
        int result = 0;
        for (int i = 0; i < energy.length; i++) {
            if (initialEnergy <= energy[i]) {
                result += (energy[i] - initialEnergy + 1);
                initialEnergy = energy[i] + 1;
            }
            initialEnergy -= energy[i];
            if (initialExperience <= experience[i]) {
                result += (experience[i] - initialExperience + 1);
                initialExperience = experience[i] + 1;
            }
            initialExperience += experience[i];
        }
        return result;
    }
}
```

# 2389.和有限的最长子序列

```java
class Solution {
    public int[] answerQueries(int[] nums, int[] queries) {
        Arrays.sort(nums);
        int[] sum = new int[nums.length];
        sum[0] = nums[0];
        for (int i = 1; i < sum.length; i++) {
            sum[i] = sum[i - 1] + nums[i];
        }
        int[] result = new int[queries.length];
        for (int i = 0; i < queries.length; i++) {
            for (int j = 0; j < sum.length; j++) {
                if (sum[j] > queries[i]) {
                    result[i] = j;
                    break;
                } else if (j == sum.length - 1 && sum[j] <= queries[i]) {
                    result[i] = j + 1;
                    break;
                }
            }
        }
        return result;
    }
}
```

# 2399.检查相同字母间的距离

```java
class Solution {
    public boolean checkDistances(String s, int[] distance) {
        int[] indexList = new int[26];
        for (int i = 0; i < indexList.length; i++) {
            indexList[i] = -1;
        }
        for (int i = 0; i < s.length(); i++) {
            int index = s.charAt(i) - 'a';
            if (indexList[index] == -1) {
                indexList[index] = i;
            } else {
                int len = i - indexList[index] - 1;
                if (distance[index] != len) {
                    return false;
                }
            }
        }
        return true;
    }
}
```
