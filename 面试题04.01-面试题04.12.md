# 面试题 04.02.最小高度树

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

    int[] num;

    public TreeNode sortedArrayToBST(int[] nums) {
        num = nums;
        return buildTree(0, num.length - 1);
    }

    public TreeNode buildTree(int left, int right) {
        if (left > right) {
            return null;
        }
        int index = (left + right) / 2;
        TreeNode node = new TreeNode(num[index]);
        node.left = buildTree(left, index - 1);
        node.right = buildTree(index + 1, right);
        return node;
    }
}
```



# 面试题 04.04.检查平衡性

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isBalanced(TreeNode root) {
        if (root == null) {
            return true;
        }
        return Math.abs(height(root.left) - height(root.right)) <= 1 && isBalanced(root.left) && isBalanced(root.right);
    }
    public int height(TreeNode node) {
        if (node == null) {
            return 0;
        }
        return Math.max(height(node.left), height(node.right)) + 1;
    }
}
```

# 面试题 04.05. 合法二叉搜索树

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isValidBST(TreeNode root) {
        return isT(root, Long.MAX_VALUE, Long.MIN_VALUE);
    }
    public boolean isT(TreeNode node, long max, long min) {
        if (node == null) {
            return true;
        }
        if (node.val >= max || node.val <= min) {
            return false;
        }
        return isT(node.left, node.val, min) && isT(node.right, max, node.val);
    }
}
```
