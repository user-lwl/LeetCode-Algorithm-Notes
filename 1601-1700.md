```java
    public int[] bestCoordinate(int[][] towers, int radius) {
        int[] result = new int[]{0, 0};
        int maxXh = 0;
        int[][] tempX = new int[101][101];
        for (int i = 0; i < towers.length; i++) {
            int x = towers[i][0];
            int y = towers[i][1];
            int qd = towers[i][2];
            for (int j = Math.max(x - radius, 0); j <= x + radius; j++) {
                for (int k = Math.max(y - radius, 0); k <= y + radius; k++) {
                    double d = Math.sqrt((x - j) * (x - j) + (y - k) * (y - k));
                    if (d > radius) {
                        continue;
                    }
                    int xh = (int) (qd / (1 + d));
                    tempX[j][k] += xh;
                    if (tempX[j][k] > maxXh) {
                        maxXh = tempX[j][k];
                        result[0] = j;
                        result[1] = k;
                    } else if (tempX[j][k] == maxXh && (j < result[0] || (j == result[0] && k < result[1]))) {
                        result[0] = j;
                        result[1] = k;
                    }
                }
            }
        }
        return result;
    }
```

```java
        int[] tempCode = new int[code.length * 2];
        System.arraycopy(code, 0, tempCode, 0, code.length);
        System.arraycopy(code, 0, tempCode, code.length, code.length);

        }
        int left = k > 0 ? 1 : code.length + k;
```

# 1696.跳跃游戏Ⅵ

```java
class Solution {
    public int maxResult(int[] nums, int k) {
        int len = nums.length;
        Deque<Integer> queue = new LinkedList<>();
        int[] result = new int[len];
        result[0] = nums[0];
        queue.addLast(0);
        for (int i = 1; i < len; i++) {
            while (!queue.isEmpty() && i - queue.peekFirst() > k) {
                queue.pollFirst();
            }
            result[i] = nums[i] + result[queue.peekFirst()];
            while (!queue.isEmpty() && result[i] >= result[queue.peekLast()]) {
                queue.pollLast();
            }
            queue.addLast(i);
        }
        return result[len - 1];
    }
}
```
