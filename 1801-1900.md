# 1812.判断国际象棋棋盘中一个格子的颜色

```java
class Solution {
    public boolean squareIsWhite(String coordinates) {
        int code = coordinates.charAt(0) - 'a';
        int number = coordinates.charAt(1) - '0';
        return (code + number) % 2 == 0;
    }
```

# 1827.最少操作使数组递增

```java
class Solution {
    public int minOperations(int[] nums) {
        int len = nums.length;
        int result = 0;
        for (int i = 0; i < len - 1; i++) {
            if (nums[i] >= nums[i + 1]) {
                result += nums[i] - nums[i + 1] + 1;
                nums[i + 1] = nums[i] + 1;
            }
        }
        return result;
    }
}
```

# 1832.判断句子是否为全字母句

```java
class Solution {
    public boolean checkIfPangram(String sentence) {
        int[] resultInt = new int[26];
        for (int i = 0; i < sentence.length(); i++) {
            resultInt[sentence.charAt(i) - 'a']++;
        }
        for (int i = 0; i < resultInt.length; i++) {
            if (resultInt[i] == 0) {
                return false;
            }
        }
        return true;
    }
}
```

# 1833.雪糕的最大数量

```java
class Solution {
    public int maxIceCream(int[] costs, int coins) {
        int[] counting = new int[100001];
        for (int i = 0; i < costs.length; i++) {
            counting[costs[i]]++;
        }
        int count = 0;
        for (int i = 0; i < counting.length; i++) {
            while (counting[i] > 0) {
                coins -= i;
                if (coins >= 0) {
                    count++;
                    counting[i]--;
                } else {
                    break;
                }
            }
            if (coins <= 0) {
                break;
            }
        }
        return count;
    }
}
```

# 1854.人口最多的年份

```java
class Solution {
    public int maximumPopulation(int[][] logs) {
        int[] people = new int[101];
        for (int i = 0; i < logs.length; i++) {
            for (int j = logs[i][0]; j < logs[i][1]; j++) {
                people[j - 1950]++;
            }
        }
        int result = 0;
        int max = 0;
        for (int i = 0; i < people.length; i++) {
            if (people[i] > max) {
                result = 1950 + i;
                max = people[i];
            }
        }
        return result;
    }
}
```



# 1859.将句子排序

```java
class Solution {
    public String sortSentence(String s) {
        String[] splitString = s.split(" ");
        Map<Integer, String> resultMap = new HashMap<>();
        StringBuilder sb = new StringBuilder();
        for (String tempString : splitString) {
            String word = tempString.substring(0, tempString.length() - 1);
            Integer index = Integer.valueOf(tempString.substring(tempString.length() - 1));
            resultMap.put(index, word);
        }
        for (int i = 1; i < 10; i++) {
            String tempString = resultMap.getOrDefault(i, null);
            if (tempString == null) {
                return sb.toString();
            }
            if (!sb.isEmpty()) {
                sb.append(" ");
            }
            sb.append(tempString);
        }
        return sb.toString();
    }
}
```



# 1863.找出所有子集的异或总和再求和

```java
class Solution {
    public int subsetXORSum(int[] nums) {
        int sum = 0;
        int len = nums.length;
        for (int i = 0; i < (1<<len); i++) {
            int result = 0;
            for (int j = 0; j < len; j++) {
                if ((i & (1<<j)) != 0) {
                    result ^= nums[j];
                }
            }
            sum += result;
        }
        return sum;
    }
}
```

# 1876.长度为三且各字符不同的子字符串

```java
class Solution {
    public int countGoodSubstrings(String s) {
        char[] c = s.toCharArray();
        int number = 0;
        for (int i = 0; i < c.length - 2; i++) {
            if (c[i] != c[i + 1] && c[i] != c[i + 2] && c[i + 1] != c[i + 2]) {
                number++;
            }
        }
        return number;
    }
}
```

# 1893.检查是否区域内所有整数都被覆盖

```java
class Solution {
    public boolean isCovered(int[][] ranges, int left, int right) {
        int[] result = new int[51];
        for (int i = 0; i < ranges.length; i++) {
            for (int j = ranges[i][0]; j <= ranges[i][1]; j++) {
                result[j]++;
            }
        }
        for (int i = left; i <= right; i++) {
            if (result[i] == 0) {
                return false;
            }
        }
        return true;
    }
}
```

# 1897.重新分配字符串使所有字符串都相等

```java
class Solution {
    public boolean makeEqual(String[] words) {
        int[] tempList = new int[26];
        int n = words.length;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < words[i].length(); j++) {
                tempList[words[i].charAt(j) - 'a']++;
            }
        }
        for (int i = 0; i < tempList.length; i++) {
            if (tempList[i] % n != 0) {
                return false;
            }
        }
        return true;
    }
}
```
