# 1805.字符串中不同整数的数目

```java
class Solution {
    public int numDifferentIntegers(String word) {
        Set<String> numberSet = new HashSet<>();
        for (int i = 0; i < word.length(); i++) {
            char temp = word.charAt(i);
            if (Character.isDigit(temp) ) {
                int startIndex = i;
                while (i < word.length() && Character.isDigit(word.charAt(i))) {
                    i++;
                }
                while (startIndex < i && word.charAt(startIndex) == '0') {
                    startIndex++;
                }
                String tempResult = word.substring(startIndex, i);
                numberSet.add(tempResult);
            }
        }
        return numberSet.size();
    }
}
```

# 1827.最少操作使数组递增

```java
class Solution {
    public int minOperations(int[] nums) {
        int len = nums.length;
        int result = 0;
        for (int i = 0; i < len - 1; i++) {
            if (nums[i] >= nums[i + 1]) {
                result += nums[i] - nums[i + 1] + 1;
                nums[i + 1] = nums[i] + 1;
            }
        }
        return result;
    }
}
```

# 1833.雪糕的最大数量

```java
class Solution {
    public int maxIceCream(int[] costs, int coins) {
        int[] counting = new int[100001];
        for (int i = 0; i < costs.length; i++) {
            counting[costs[i]]++;
        }
        int count = 0;
        for (int i = 0; i < counting.length; i++) {
            while (counting[i] > 0) {
                coins -= i;
                if (coins >= 0) {
                    count++;
                    counting[i]--;
                } else {
                    break;
                }
            }
            if (coins <= 0) {
                break;
            }
        }
        return count;
    }
}
```

```java
            sb.append(shift(s.charAt(i - 1), s.charAt(i) - '0'));
```

```java
class Solution {
    public String sortSentence(String s) {
        String[] splitString = s.split(" ");
        Map<Integer, String> resultMap = new HashMap<>();
        StringBuilder sb = new StringBuilder();
        for (String tempString : splitString) {
            String word = tempString.substring(0, tempString.length() - 1);
            Integer index = Integer.valueOf(tempString.substring(tempString.length() - 1));
            resultMap.put(index, word);
        }
        for (int i = 1; i < 10; i++) {
            String tempString = resultMap.getOrDefault(i, null);
            if (tempString == null) {
                return sb.toString();
            }
            if (!sb.isEmpty()) {
                sb.append(" ");
            }
            sb.append(tempString);
        }
        return sb.toString();
    }
}
```

# 1863.找出所有子集的异或总和再求和

```java
class Solution {
    public int subsetXORSum(int[] nums) {
        int sum = 0;
        int len = nums.length;
        for (int i = 0; i < (1<<len); i++) {
            int result = 0;
            for (int j = 0; j < len; j++) {
                if ((i & (1<<j)) != 0) {
                    result ^= nums[j];
                }
            }
            sum += result;
        }
        return sum;
    }
}
```

# 1869.哪种连续子字符串更长

```java
class Solution {
    public boolean checkZeroOnes(String s) {
        int oneLen = 0;
        int zeroLen = 0;
        for (int i = 0; i < s.length(); i++) {
            char tempChar = s.charAt(i);
            int startIndex = i;
            while (i < s.length() && s.charAt(i) == tempChar) {
                i++;
            }
            if (tempChar == '0') {
                zeroLen = Math.max(zeroLen, i - startIndex);
            } else {
                oneLen = Math.max(oneLen, i - startIndex);
            }
            i--;
        }
        return oneLen > zeroLen;
    }
}
```

# 1886.判断矩阵经轮转后是否一致

```java
class Solution {
    public boolean findRotation(int[][] mat, int[][] target) {
        boolean zero = true;
        boolean nineTen = true;
        boolean oneEightTen = true;
        boolean twoSevenTen = true;
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat.length; j++) {
                // 0
                if (mat[i][j] != target[i][j]) {
                    zero = false;
                }
                // 90
                if (mat[i][j] != target[j][mat.length - 1 - i]) {
                    nineTen = false;
                }
                // 180
                if (mat[i][j] != target[mat.length - 1 - i][mat.length - 1 - j]) {
                    oneEightTen = false;
                }
                // 270
                if (mat[i][j] != target[mat.length - 1 - j][i]) {
                    twoSevenTen = false;
                }
            }
        }
        return zero || oneEightTen || twoSevenTen || nineTen;
    }
}
```
